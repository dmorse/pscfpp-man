56  void Propagator<D>::computeHead()
57  {

59  // Reference to head of this propagator
60  QField& qh = qFields_[0];
61 
62  // Initialize qh field to 1.0 at all grid points
63  int ix;
64  int nx = meshPtr_->size();
65  for (ix = 0; ix < nx; ++ix) {
66  qh[ix] = 1.0;
67  }
68 
69  // Pointwise multiply tail QFields of all sources

71  if (!source(is).isSolved()) {
72  UTIL_THROW("Source not solved in computeHead");

74  QField const& qt = source(is).tail();
75  for (ix = 0; ix < nx; ++ix) {

117  /*
118  * Integrate to calculate monomer concentration for this block
119  */
120  template <int D>
121  double Propagator<D>::computeQ()
122  {
123  // Preconditions
124  if (!isSolved()) {
125  UTIL_THROW("Propagator is not solved.");
126  }
127  if (!hasPartner()) {
128  UTIL_THROW("Propagator has no partner set.");
129  }
130  if (!partner().isSolved()) {
131  UTIL_THROW("Partner propagator is not solved");
132  }
133  QField const& qh = head();
134  QField const& qt = partner().tail();
135  int nx = meshPtr_->size();
136  UTIL_CHECK(qt.capacity() == nx);
137  UTIL_CHECK(qh.capacity() == nx);
138 
139  // Take inner product of head and partner tail fields
140  double Q = 0;
141  for (int i =0; i < nx; ++i) {
142  Q += qh[i]*qt[i];
143  }
144  Q /= double(nx);
145  return Q;
146  }
