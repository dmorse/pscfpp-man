64 void BinaryStructureFactorGrid<D>::setup()
65 {
66 //Allocate variables
67 const int nMonomer = system().mixture().nMonomer();
68 IntVec<D> const & dimensions = system().mesh().dimensions();
69 if (!isInitialized_){
70 wm_.allocate(dimensions);
71 wk_.allocate(dimensions);
72 }
73
74 // Compute Fourier space kMeshDimensions_
75 for (int i = 0; i < D; ++i) {
76 if (i < D - 1) {
77 kMeshDimensions_[i] = dimensions[i];
78 } else {
79 kMeshDimensions_[i] = dimensions[i]/2 + 1;
80 }
81 }
82 kSize_ = 1;
83 for(int i = 0; i < D; ++i) {
84 kSize_ *= kMeshDimensions_[i];
85 }
86
87 nWave_ = kSize_;
88 structureFactors_.allocate(nWave_);
89 accumulators_.allocate(nWave_);
90
91 // Convert real grid to KGrid format
92 qList_.resize(kSize_);
93 RField<D> const & kSq = system().domain().waveList().kSq();
94 HostDArray<cudaReal> kSqHost(kSize_);
95 kSqHost = kSq;
96 MeshIterator<D> itr;
97 itr.setDimensions(kMeshDimensions_);
98 for (itr.begin(); !itr.atEnd(); ++itr){
99 qList_[itr.rank()] = sqrt(double(kSqHost[itr.rank()]));
100 }
101
102 isInitialized_ = true;
103
104 // Clear accumulators
105 for (int i = 0; i < nWave_; ++i) {
106 structureFactors_[i] = 0.0;
107 accumulators_[i].setNSamplePerBlock(nSamplePerBlock_);
108 accumulators_[i].clear();

128 VecOp::addVcVc(wm_, system().w().rgrid(0), 0.5,

143 template <int D>
144 void BinaryStructureFactorGrid<D>::computeStructureFactor()
145 {
146 const double vSystem = system().domain().unitCell().volume();
147 const double vMonomer = system().mixture().vMonomer();
148 double n = vSystem / vMonomer;
149 double chi= system().interaction().chi(0,1);
150 MeshIterator<D> itr;
151 itr.setDimensions(kMeshDimensions_);
152 for (itr.begin(); !itr.atEnd(); ++itr) {
153 // Compute vS(q)
154 structureFactors_[itr.rank()] = n / (chi * chi) * accumulators_[itr.rank()].average() - 1.0/(2.0*chi);
155
156 // Compute S(q)
157 structureFactors_[itr.rank()] /= vMonomer;
158 }
159 }

161 // Average S(k) over k of equal magnitude
162 template <int D>
163 void BinaryStructureFactorGrid<D>::averageStructureFactor()
164 {
165 UTIL_CHECK(qList_.capacity() == structureFactors_.capacity());
166
167 std::map<double, double> SMap;
168 for (int i = 0; i < qList_.capacity(); ++i) {
169 double q = qList_[i];
170 double s = structureFactors_[i];
171 SMap[q] += s;
172 }
173
174 // Average structure factor with same magnitude value of q
175 for (auto& i : SMap) {
176 double q = i.first;
177 double sum = i.second;
178 int count = std::count(qList_.begin(), qList_.end(), q);
179 i.second = sum / count;
180 }
181
182 // Average structure factor for q in range of +-tolerance
183 double tolerance = 1e-5;
184 auto it = SMap.begin();
185 double currentq = it->first;
186 double sumS = it->second;
187 int count = 1;
188 for (++it; it != SMap.end(); ++it) {
189 double key = it->first;
190 double s = it->second;
191 if (std::abs(key - currentq) <= tolerance) {
192 sumS += s;
193 count++;
194 } else {
195 averageSMap_[currentq] = sumS / count;
196 currentq = key;
197 sumS = s;
198 count = 1;
199 }
200 }
201
202 // Add last batch of data (last number within tolerance)
203 averageSMap_[currentq] = sumS / count;
204 }
