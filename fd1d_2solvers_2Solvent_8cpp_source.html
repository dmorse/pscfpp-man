76 /*
77 * Compute concentration, q, phi or mu.
78 */
79 void Solvent::compute(WField const & wField)
80 {
81 int nx = domainPtr_->nx();
82 int i;
83
84 // Initialize to zero
85 for (i = 0; i < nx; ++i) {
86 cField_[i] = 0.0;
87 }
88
89 // Evaluate unnormalized integral and q_
90 double s = size();
91 q_ = 0.0;
92 for (i = 0; i < nx; ++i) {
93 cField_[i] = exp(-s*wField[i]);
94 q_ += cField_[i];
95 }
96 q_ = q_/double(nx);
97
98 // Compute mu_ or phi_ and prefactor
99 double prefactor;
100 if (ensemble_ == Species::Closed) {
101 prefactor = phi_/q_;
102 mu_ = log(prefactor);
103 } else {
104 prefactor = exp(mu_);
105 phi_ = prefactor*q_;
106 }
107
108 // Normalize concentration
109 for (i = 0; i < nx; ++i) {
110 cField_[i] *= prefactor;
111 }
112
113 }
114
115}
