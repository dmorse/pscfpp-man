130 /*
131 * Compute properties of a homogeneous reference system.
132 *
133 * mode == 0 : Composition equals spatial average composition
134 * mode == 1: Chemical potential equal to that of system,
135 * composition guess given at last grid point.
136 * mode == 2: Chemical potential equal to that of system,
137 * composition guess given at first grid point.
138 */
139 void HomogeneousComparison::compute(int mode)
140 {
141 int np = mixture().nPolymer();
142 int ns = mixture().nSolvent();
143 if (!p_.isAllocated()) {
144 p_.allocate(np+ns);
145 }
146 UTIL_CHECK(p_.capacity() == homogeneous().nMolecule());
147
148 if (mode == 0) {
149
150 for (int i = 0; i < np; ++i) {
151 p_[i] = mixture().polymer(i).phi();
152 }
153 homogeneous().setComposition(p_);
154 double xi = 0.0;
155 homogeneous().computeMu(interaction(), xi);
156
157 } else
158 if (mode == 1 || mode == 2) {
159
160 if (!m_.isAllocated()) {
161 m_.allocate(np+ns);
162 }
163 UTIL_CHECK(m_.capacity() == homogeneous().nMolecule());
164 for (int i = 0; i < np; ++i) {
165 m_[i] = mixture().polymer(i).mu();
166 }
167 int ix; // Grid index from which we obtain guess of composition
168 if (mode == 1) {
169 ix = domain().nx() - 1;
170 } else
171 if (mode == 2) {
172 ix = 0;
173 }
174
175 for (int i = 0; i < np; ++i) {
176 p_[i] = 0.0;
177 int nb = mixture().polymer(i).nBlock();
178 for (int j = 0; j < nb; ++j) {
179 p_[i] += mixture().polymer(i).block(j).cField()[ix];
180 }
181 }
182
183 #if 0
184 std::cout << std::endl;
185 std::cout << "Composition at boundary " << std::endl;
186 for (int i = 0; i < np; ++i) {
187 std::cout << "phi[" << i << "] = " << p_[i] << std::endl;
188 }
189 #endif
190
191 double xi = 0.0;
192 homogeneous().computePhi(interaction(), m_, p_, xi);
193
194 } else {
195 UTIL_THROW("Unknown mode in computeHomogeneous");
196 }
197
198 // Compute Helmholtz free energy and pressure
199 homogeneous().computeFreeEnergy(interaction());
200 }

202 /*
203 * Output properties of homogeneous system and free energy difference.
204 *
205 * Mode 0: Outputs fHomo (Helmholtz) and difference df
206 * Mode 1 or 2: Outputs Helhmoltz, pressure and difference dp
207 */
208 void HomogeneousComparison::output(int mode, std::ostream& out)
209 {
210 // Output free energies
211 out << std::endl;
212 if (mode == 0) {
213
214 // Output free energies
215 double fHomo = homogeneous().fHelmholtz();
216 double df = system().fHelmholtz() - fHomo;
217 out << "f (homo) = " << Dbl(fHomo, 18, 11) << std::endl;
218 out << "delta f = " << Dbl(df, 18, 11) << std::endl;
219
220 // Output polymer properties
221 out << std::endl;
222 out << "Polymers: i, mu(homo)[i], phi[i] " << std::endl;
223 for (int i = 0; i < homogeneous().nMolecule(); ++i) {
224 out << i
225 << " " << Dbl(homogeneous().mu(i), 18, 11)
226 << " " << Dbl(homogeneous().phi(i), 18, 11)
227 << std::endl;
228 }
229 out << std::endl;
230
231 } else
232 if (mode == 1 || mode == 2) {
233
234 // Output free energies
235 double fHomo = homogeneous().fHelmholtz();
236 double pHomo = homogeneous().pressure();
237 double pEx = system().pressure() - pHomo;
238 double fEx = system().fHelmholtz() - fHomo;
239 double V = domain().volume()/mixture().vMonomer();
240 double PExV = -1.0*pEx*V;
241 double FExV = fEx*V;
242 out << "f (homo) = " << Dbl(fHomo, 18, 11) << std::endl;
243 out << "p (homo) = " << Dbl(pHomo, 18, 11) << std::endl;
244 out << "f (ex) = " << Dbl(fEx, 18, 11) << std::endl;
245 out << "p (ex) = " << Dbl(pEx, 18, 11) << std::endl;
246 out << "-p(ex)*V = " << Dbl(PExV, 18, 11) << std::endl;
247 out << "f(ex)*V = " << Dbl(FExV, 18, 11) << std::endl;
248
249 // Output polymer properties
250 double dV;
251 out << std::endl;
252 out << "Polymers:" << std::endl;
253 out << " i"
254 << " mu[i] "
255 << " phi(homo)[i] "
256 << " dV[i] "
257 << std::endl;
258 for (int i = 0; i < homogeneous().nMolecule(); ++i) {
259 dV = mixture().polymer(i).phi() - homogeneous().phi(i);
260 dV *= V;
261 out << Int(i,5)
262 << " " << Dbl(homogeneous().mu(i), 18, 11)
263 << " " << Dbl(homogeneous().phi(i), 18, 11)
264 << " " << Dbl(dV, 18, 11)
265 << std::endl;
266 }
267 out << std::endl;
268
269 }
