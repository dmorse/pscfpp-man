79  template <int D>
80  void Mixture<D>::compute(DArray<Mixture<D>::WField> const & wFields,
81  DArray<Mixture<D>::CField>& cFields)
82  {
83  UTIL_CHECK(meshPtr_);
84  UTIL_CHECK(mesh().size() > 0);
85  UTIL_CHECK(nMonomer() > 0);
86  UTIL_CHECK(nPolymer() + nSolvent() > 0);
87  UTIL_CHECK(wFields.capacity() == nMonomer());
88  UTIL_CHECK(cFields.capacity() == nMonomer());
89 
90  int nx = mesh().size();
91  int nm = nMonomer();
92  int i, j, k;
93 
94  // Clear all monomer concentration fields
95  for (i = 0; i < nm; ++i) {
96  UTIL_CHECK(cFields[i].capacity() == nx);
97  UTIL_CHECK(wFields[i].capacity() == nx);
98  for (j = 0; j < nx; ++j) {
99  cFields[i][j] = 0.0;
100  }
101  }
102 
103  // Solve MDE for all polymers
104  for (i = 0; i < nPolymer(); ++i) {
105  polymer(i).compute(wFields);
106  }
107 
108  // Accumulate monomer concentration fields
109  double phi;
110  for (i = 0; i < nPolymer(); ++i) {
111  phi = polymer(i).phi();
112  for (j = 0; j < polymer(i).nBlock(); ++j) {
113  int monomerId = polymer(i).block(j).monomerId();
114  UTIL_CHECK(monomerId >= 0);
115  UTIL_CHECK(monomerId < nm);
116  CField& monomerField = cFields[monomerId];
117  CField& blockField = polymer(i).block(j).cField();
118  for (k = 0; k < nx; ++k) {
119  monomerField[k] += phi * blockField[k];
120  }
121  }
122  }
