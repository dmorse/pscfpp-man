333 /*
334 * Get square magnitude of reciprocal basis vector.
335 */
336 template <int D>
337 double UnitCellBase<D>::ksq(IntVec<D> const & k) const
338 {
339 RealVec<D> g(0.0);
340 RealVec<D> p;
341 for (int i = 0; i < D; ++i) {
342 p.multiply(kBasis_[i], k[i]);
343 g += p;
344 }
345 double value = 0.0;
346 for (int i = 0; i < D; ++i) {
347 value += g[i]*g[i];
348 }
349 return value;
350 }

352 /*
353 * Get magnitude of derivative of square of reciprocal basis vector.
354 */
355 template <int D>
356 double UnitCellBase<D>::dksq(IntVec<D> const & vec, int n) const
357 {
358 double element = 0.0;
359 double value = 0.0;
360
361 for (int p = 0; p < D; ++p){
362 for (int q = 0; q < D; ++q){
363 element = dkkBasis(n, p, q);
364 value += vec[p]*vec[q]*element;
365 }
366 }
367
368 return value;
369 }

376 template <int D>
377 void UnitCellBase<D>::initializeToZero()
378 {
379 // Initialize all elements to zero
380 int i, j, k;
381 for (i = 0; i < D; ++i) {
382 for (j = 0; j < D; ++j) {
383 rBasis_[i][j] = 0.0;
384 kBasis_[i][j] = 0.0;
385 }
386 }
387 for (k = 0; k < 6; ++k){
388 for (i = 0; i < D; ++i) {
389 for (j = 0; j < D; ++j) {
390 drBasis_[k](i,j) = 0.0;
391 dkBasis_[k](i,j) = 0.0;
392 drrBasis_[k](i,j) = 0.0;
393 dkkBasis_[k](i,j) = 0.0;
394 }
395 }

399 /*
400 * Compute quantities involving derivatives.
401 */
402 template <int D>
403 void UnitCellBase<D>::computeDerivatives()
404 {
405 // Compute dkBasis
406 int p, q, r, s, t;
407 for (p = 0; p < nParameter_; ++p) {
408 for (q = 0; q < D; ++q) {
409 for (r = 0; r < D; ++r) {
410
411 // Loop over free indices s, t
412 for (s = 0; s < D; ++s) {
413 for (t = 0; t < D; ++t) {
414 dkBasis_[p](q,r)
415 -= kBasis_[q][s]*drBasis_[p](t,s)*kBasis_[t][r];
416 }
417 }
418 dkBasis_[p](q,r) /= 2.0*Constants::Pi;
419
420 }
421 }
422 }
423
424 // Compute drrBasis and dkkBasis
425 for (p = 0; p < nParameter_; ++p) {
426 for (q = 0; q < D; ++q) {
427 for (r = 0; r < D; ++r) {
428 for (s = 0; s < D; ++s) {
429 drrBasis_[p](q,r) += rBasis_[q][s]*drBasis_[p](r,s);
430 drrBasis_[p](q,r) += rBasis_[r][s]*drBasis_[p](q,s);
431 dkkBasis_[p](q,r) += kBasis_[q][s]*dkBasis_[p](r,s);
432 dkkBasis_[p](q,r) += kBasis_[r][s]*dkBasis_[p](q,s);
433
434 }
435 }
436 }
437 }
