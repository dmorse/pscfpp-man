154  template <int D>
155  void Basis<D>::makeStars(const SpaceGroup<D>& group)
156  {
157  /*
158  * Local containers that hold TWave<D> objects:
159  * list - set of waves of equal norm, compared by indicesDft
160  * star - set of symmetry-related waves, compared by indicesDft
161  * tempStar - temporary star, sorted by descending indicesBz
162  * tempList - temporary list, ordered by star
163  */
164  std::set< TWave<D>, TWaveDftComp<D> > list;
165  std::set< TWave<D>, TWaveDftComp<D> > star;
166  std::vector< TWave<D> > tempStar;
167  GArray< TWave<D> > tempList;
168 
169  typename std::set< TWave<D>, TWaveDftComp<D> >::iterator rootItr;
170  typename std::set< TWave<D>, TWaveDftComp<D> >::iterator setItr;
171  TWave<D> wave;
172  Basis<D>::Star newStar;
173 
174  std::complex<double> coeff;
175  double Gsq;
176  double Gsq_max;
177  double phase_diff;
178  const double twoPi = 2.0*Constants::Pi;
179  const double epsilon = 1.0E-8;
180  IntVec<D> meshDimensions = mesh().dimensions();
181  IntVec<D> rootVecBz; // BZ indices for root of this star
182  IntVec<D> vec; // Indices of temporary wavevector
183  IntVec<D> nVec; // Indices of negation of a wavevector
184  int listId = 0; // id for this list
185  int listBegin = 0; // id of first wave in this list
186  int listEnd = 0; // (id of last wave in this list) + 1
187  int listSize; // listEnd - listBegin
188  int starId = 0; // id for this star
189  int starBegin = 0; // id of first wave in this star
190  int i, j, k;
191  bool cancel;
192 
193  /*
194  * Overview of algorithm:
195  *
196  * Loop over index i of array waves_ {
197  *
198  * Search for end of a "list", contiguous block of waves of equal
199  * magnitude, with indices [listBegin,listEnd-1]. Set newList true.
200  *
201  * // Each list can contain one or more stars.
202  * // Process the newly identified list to identify stars
203  *
204  * If (newList) {
205  *
206  * Copy all waves in range into container std::set<TWave<D>> list
207  *
208  * Set rootItr to first wave in list
209  *
210  * // Loop over stars within list
211  * while (list.size() > 0) {
212  *
213  * // To generate a star from a root wave rootItr,
214  * // loop over symmetry operations of space group.
215  * For each group symmetry operation group[j] {
216  * Compute vec = (rootItr->indicesBz)*group[j]
217  * Set phase = rootItr->indicesBz .dot. group[j].t
218  * Check for cancellation of the star, set cancel flag
219  * Add wave to std::set<TWave> star if not added before
220  * // Here, use of a std::set simplifies test of uniqueness
221  * }
222  *
223  * Copy all waves from star to std::vector<TWave> tempStar
224  * Sort tempStar by indicesBz in descending order
225  * // Here, use of ordered container std::vector allows sorting
226  * For each wave in tempStar {
227  * Append wave to std::vector<TWave> tempList
228  * Erase wave from std::set<TWave> list
229  * // Here, use of a std::set for list simplifies erasure
230  * }
231  *
232  * Initialize a Star object newStar and assign values
233  * to members beginId, endId, size, eigen, cancel
234  *
235  * // Assign values of newStar.invertFlag, rootItr, nextInvert
236  * if (nextInvert == -1) {
237  * // This is the second star in pair
238  * newStar.invertFlag = -1;
239  * nextInvert = 1;
240  * Set rootItr to the first wave in remaining list
241  * } else {
242  * Search for negation of rootItr in this star
243  * if negation is in this star {
244  * newStar.invertFlag = 0
245  * nextInvert = 1;
246  * Set rootItr to the first wave in remaining list
247  * } else
248  * Search for negation of rootItr in this remaining list
249  * if the negation is in the remaining list {
250  * newStar.invertFlag = 1
251  * nextInvert = -1;
252  * set rootItr to negation of current root
253  * }
254  * }
255  *
256  * Append newStar object to DArray<Basis::Star> stars_
257  *
258  * } // end loop over stars in a single list
259  *
260  * For each TWave in tempList {
261  * Copy TWave in tempList to a Basis:Wave in waves_
262  * Assign a complex coefficient of unit norm to the Wave
263  * }
264  * // This overwrites a block of the waves_ array
265  *
266  * // At this point, coefficients of waves have correct
267  * // correct relative phases within a star, but not final
268  * // absolute phases and have unit absolute magnitude.
269  *
270  * } // end processing of one list (waves of equal norm)
271  *
272  * } // End initial processing of all waves and stars
273  *
274  * // Set absolute wave coefficients
275  * For each star in array stars_ {
276  * if star is closed (star.invertFlag == 0) {
277  * if star is cancelled {
278  * set all coefficients to zero
279  * } else {
280  * Set the root to the first wave in the star
281  * Check that negation of root is also in this star
282  * Divide all coefficients by the root coefficient
283  * Require coeffs of root & negation be complex conjugates
284  * }
285  * } else
286  * if (star.invertFlag == 1) {
287  * Set root of this star to the 1st wave in the star
288  * Find negation of root (aka "partner") in next star.
289  * If star is cancelled {
290  * Set coefficients in this star and next to zero
291  * } else {
292  * Divide coefficients in this star by root coefficient
293  * Divide coefficients in next star by partner coefficient
294  * }
295  * }
296  * // Note: If star.invertFlag = -1, do nothing
297  * }
298  *
299  * // For all waves, normalize coefficients and set starId
300  * For each star in stars_ {
301  * For each wave in this star {
302  * Set Wave.starId
303  * Divide coefficient by sqrt(double(star.size))
304  * }
305  * }
306  *
307  * // For all waves, set implicit member and add to look up table
308  * For each wave in waves_ {
309  * Set implicit attribute
310  * Assign waveIds_[rank] = i
311  * }
312  */
313 
314  // Loop over all waves
315  nBasis_ = 0;
316  nBasisWave_ = 0;
317  Gsq_max = waves_[0].sqNorm;
318  for (i = 1; i <= nWave_; ++i) {
319 
320  // Determine if this wave begins a new list
321  bool newList = false;
322  if (i == nWave_) {
323  listEnd = i;
324  listSize = listEnd - listBegin;
325  newList = true;
326  } else {
327  Gsq = waves_[i].sqNorm;
328  if (Gsq > Gsq_max + epsilon) {
329  Gsq_max = Gsq;
330  listEnd = i;
331  listSize = listEnd - listBegin;
332  newList = true;
333  }
334  }
335 
336  // Process completed list of wavectors of equal norm
337  if (newList && listEnd > 0) {
338 
339  // Copy waves of equal norm into std::set "list"
340  list.clear();
341  tempList.clear();
342  for (j = listBegin; j < listEnd; ++j) {
343  wave.indicesDft = waves_[j].indicesDft;
344  wave.indicesBz = waves_[j].indicesBz;
345  wave.sqNorm = waves_[j].sqNorm;
346  if (j > listBegin) {
347  UTIL_CHECK( abs(wave.sqNorm-waves_[j].sqNorm)
348  < 2.0*epsilon );
349  }
350  list.insert(wave);
351  }
352 
353  // On entry to each iteration of the loop over stars,
354  // rootItr and nextInvert are known. The iterator rootItr
355  // points to the wave in the remaining list that will be
356  // used as the root of the next star. The flag nextInvert
357  // is equal to -1 iff the previous star was the first of
358  // a pair that are open under inversion, and is equal
359  // to + 1 otherwise.
360 
361  // Initial values for first star in this list
362  rootItr = list.begin();
363  int nextInvert = 1;
364 
365  // Loop over stars with a list of waves of equal norm,
366  // removing each star from the list as it is identified.
367  // The root of the next star must have been chosen on
368  // entry to each iteration of this loop.
369 
370  while (list.size() > 0) {
371 
372  rootVecBz = rootItr->indicesBz;
373  Gsq = rootItr->sqNorm;
374  cancel = false;
375  star.clear();
376 
377  // Construct a star from root vector, by applying every
378  // symmetry operation in the group to the root wavevector.
379  for (j = 0; j < group.size(); ++j) {
380 
381  // Apply symmetry (i.e., multiply by rotation matrix)
382  // vec = rotated wavevector.
383  vec = rootVecBz*group[j];
384 
385  // Check that rotated vector has same norm as root.
386  UTIL_CHECK(abs(Gsq - unitCell().ksq(vec)) < epsilon);
387 
388  // Initialize TWave object associated with rotated wave
389  wave.sqNorm = Gsq;
390  wave.indicesBz = shiftToMinimum(vec, meshDimensions,
391  *unitCellPtr_);
392  wave.indicesDft = vec;
393  mesh().shift(wave.indicesDft);
394 
395  // Compute phase for coeff. of wave in basis function.
396  // Convention -pi < phase <= pi.
397  wave.phase = 0.0;
398  for (k = 0; k < D; ++k) {
399  wave.phase += rootVecBz[k]*(group[j].t(k));
400  }
401  while (wave.phase > 0.5) {
402  wave.phase -= 1.0;
403  }
404  while (wave.phase <= -0.5) {
405  wave.phase += 1.0;
406  }
407  wave.phase *= twoPi;
408 
409  // Check for cancellation of star: The star is
410  // cancelled if application of any symmetry operation
411  // in the group to the root vector yields a rotated
412  // vector equivalent to the root vector but with a
413  // nonzero phase, creating a contradiction.
414 
415  if (wave.indicesDft == rootItr->indicesDft) {
416  if (abs(wave.phase) > 1.0E-6) {
417  cancel = true;
418  }
419  }
420 
421  // Search for an equivalent wave already in the star.
422  // Note: Equivalent waves have equal DFT indices.
423  setItr = star.find(wave);
424 
425  if (setItr == star.end()) {
426 
427  // If no equivalent wave is found in the star,
428  // then add this wave to the star
429  star.insert(wave);
430 
431  } else {
432 
433  // If an equivalent wave is found, check if the
434  // phases are equivalent. If not, the star is
435  // cancelled.
436 
437  phase_diff = setItr->phase - wave.phase;
438  while (phase_diff > 0.5) {
439  phase_diff -= 1.0;
440  }
441  while (phase_diff <= -0.5) {
442  phase_diff += 1.0;
443  }
444  if (abs(phase_diff) > 1.0E-6) {
445  cancel = true;
446  }
447 
448  }
449 
450  }
451 
452  // Copy all waves from set star to std::vector tempStar
453  tempStar.clear();
454  setItr = star.begin();
455  for ( ; setItr != star.end(); ++setItr) {
456  tempStar.push_back(*setItr);
457  }
458 
459  // Sort tempStar, in descending order by indicesBz.
460  TWaveBzComp<D> waveBzComp;
461  std::sort(tempStar.begin(), tempStar.end(), waveBzComp);
462 
463  // Append contents of tempStar to tempList, erase from list
464  int tempStarSize = tempStar.size();
465  for (j = 0; j < tempStarSize; ++j) {
466  list.erase(tempStar[j]);
467  tempList.append(tempStar[j]);
468  }
469  UTIL_CHECK((int)(tempList.size()+list.size()) == listSize);
470 
471  // If this star is not cancelled, increment the number of
472  // basis functions (nBasis_) and waves in basis (nBasisWave_)
473  if (!cancel) {
474  ++nBasis_;
475  nBasisWave_ += star.size();
476  }
477 
478  // Initialize a Star object
479  newStar.eigen = Gsq;
480  newStar.beginId = starBegin;
481  newStar.endId = newStar.beginId + star.size();
482  newStar.size = star.size();
483  newStar.cancel = cancel;
484  // Note: newStar.starInvert is not yet known
485 
486  // Determine invertFlag, rootItr and nextInvert
487  if (nextInvert == -1) {
488 
489  // If this star is 2nd of a pair related by inversion,
490  // set root of next star to 1st wave of remaining list.
491 
492  newStar.invertFlag = -1;
493  rootItr = list.begin();
494  nextInvert = 1;
495 
496  } else {
497 
498  // If this star is not the 2nd of a pair of partners,
499  // then determine if it is closed under inversion.
500 
501  // Compute negation nVec of root vector in FBZ
502  nVec.negate(rootVecBz);
503 
504  // Shift negation nVec to a DFT mesh
505  (*meshPtr_).shift(nVec);
506 
507  // Search for negation of root vector within this star
508  bool negationFound = false;
509  setItr = star.begin();
510  for ( ; setItr != star.end(); ++setItr) {
511  if (nVec == setItr->indicesDft) {
512  negationFound = true;
513  break;
514  }
515  }
516 
517  if (negationFound) {
518 
519  // If this star is closed under inversion, the root
520  // of next star is the 1st vector of remaining list.
521 
522  newStar.invertFlag = 0;
523  rootItr = list.begin();
524  nextInvert = 1;
525 
526  } else {
527 
528  newStar.invertFlag = 1;
529  nextInvert = -1;
530 
531  // If star is not closed, find negation of root in
532  // the remaining list, and use this negation as the
533  // root of the next star.
534 
535  setItr = list.begin();
536  for ( ; setItr != list.end(); ++setItr) {
537  if (nVec == setItr->indicesDft) {
538  negationFound = true;
539  rootItr = setItr;
540  break;
541  }
542  }
543  // If negationFound, then rootItr->indicesDft = nVec
544 
545  // Failure to find the negation here is an error:
546  // It must be either in this star or remaining list
547 
548  if (!negationFound) {
549  std::cout << "Negation not found for: " << "\n";
550  std::cout << " vec (ft):"
551  << rootItr->indicesDft <<"\n";
552  std::cout << " vec (bz):"
553  << rootItr->indicesBz <<"\n";
554  std::cout << "-vec (dft):" << nVec << "\n";
555  UTIL_CHECK(negationFound);
556  }
557 
558  }
559 
560  }
561 
562  stars_.append(newStar);
563  ++starId;
564  starBegin = newStar.endId;
565 
566  }
567  // End loop over stars within a list.
568 
569  UTIL_CHECK(list.size() == 0);
570  UTIL_CHECK(tempList.size() == listEnd - listBegin);
571 
572  // Copy tempList into corresponding section of waves_,
573  // overwriting the section of waves_ used to create the list.
574  // Compute a complex coefficient of unit norm for each wave.
575  for (j = 0; j < tempList.size(); ++j) {
576  k = j + listBegin;
577  waves_[k].indicesDft = tempList[j].indicesDft;
578  waves_[k].indicesBz = tempList[j].indicesBz;
579  waves_[k].sqNorm = tempList[j].sqNorm;
580  coeff = std::complex<double>(0.0, tempList[j].phase);
581  coeff = exp(coeff);
582  if (abs(imag(coeff)) < 1.0E-6) {
583  coeff = std::complex<double>(real(coeff), 0.0);
584  }
585  if (abs(real(coeff)) < 1.0E-6) {
586  coeff = std::complex<double>(0.0, imag(coeff));
587  }
588  waves_[k].coeff = coeff;
589  }
590 
591  // Processing of list is now complete.
592  // Here, waves_[k].coeff has unit absolute magnitude, and
593  // correct relative phases for waves within a star, but
594  // the coeff may not be unity for the first or last wave in
595  // the star.
596 
597  ++listId;
598  listBegin = listEnd;
599  }
600  // Finished processing a list of waves of equal norm
601 
602  } // End loop over all waves
603  nStar_ = stars_.size();
604  // Complete initial processing of all lists and stars
605 
606  std::complex<double> rootCoeff; // Coefficient of root wave
607  std::complex<double> partCoeff; // Coefficient of partner of root
608  std::complex<double> d;
609  int rootId, partId;
610 
611  // Final processing of coefficients of waves in stars.
612  // Require that the root wave of each star and its negation
613  // have conjugate coefficients, and that each basis function
614  // is normalized.
615  for (i = 0; i < nStar_; ++i) {
616 
617  // Treat open and closed stars separately
618 
619  if (stars_[i].invertFlag == 0) {
620 
621  // Identify root of this star (star i)
622  // Set the root to be the first wave in the star
623 
624  rootId = stars_[i].beginId;
625  stars_[i].waveBz = waves_[rootId].indicesBz;
626 
627  if (stars_[i].cancel) {
628 
629  // If the star is cancelled, set all coefficients to zero
630  std::complex<double> czero(0.0, 0.0);
631  for (j = stars_[i].beginId; j < stars_[i].endId; ++j) {
632  waves_[j].coeff = czero;
633  }
634 
635  } else { // if not cancelled
636 
637  // Compute nVec = negation of root, shifted to DFT mesh
638  nVec.negate(waves_[rootId].indicesBz);
639  (*meshPtr_).shift(nVec);
640 
641  // Find negation of root in this star, set partId to index
642  bool negationFound = false;
643  for (j = stars_[i].beginId; j < stars_[i].endId; ++j) {
644  if (nVec == waves_[j].indicesDft) {
645  partId = j;
646  negationFound = true;
647  break;
648  }
649  }
650 
651  // For invertFlag == 0, failure to find nVec in this
652  // star is a fatal error
653  UTIL_CHECK(negationFound);
654 
655  // Divide all coefficients by the root coefficient
656  rootCoeff = waves_[rootId].coeff;
657  for (j = stars_[i].beginId; j < stars_[i].endId; ++j) {
658  waves_[j].coeff /= rootCoeff;
659  }
660 
661  // Require coefficients of root and negation are conjugates
662  if (partId != rootId) {
663 
664  // Compute common divisor
665  partCoeff = waves_[partId].coeff;
666  d = sqrt(partCoeff);
667  if (abs(imag(d)) > 1.0E-8) {
668  if (imag(d) < 0.0) {
669  d = -d;
670  }
671  for (j=stars_[i].beginId; j < stars_[i].endId; ++j) {
672  waves_[j].coeff /= d;
673  }
674  }
675 
676  }
677 
678  } // end if (cancel) ... else ...
679 
680  // end if (stars_[i].invertFlag == 0)
681  } else
682  if (stars_[i].invertFlag == 1) {
683 
684  // Process a pair of open stars related by inversion.
685 
686  // Preconditions:
687  UTIL_CHECK(stars_[i].size == stars_[i+1].size);
688  UTIL_CHECK(stars_[i].cancel == stars_[i+1].cancel);
689  // UTIL_CHECK(stars_[i+1].invertFlag == -1);
690 
691  // Identify root of this star (star i)
692  // Set the root to be the first wave in the star
693  rootId = stars_[i].beginId;
694  stars_[i].waveBz = waves_[rootId].indicesBz;
695 
696  // Compute nVec = negation of root vector, shifted to DFT mesh
697  nVec.negate(waves_[rootId].indicesBz);
698  (*meshPtr_).shift(nVec);
699 
700  // Seek negation of root wave in the next star (star i+1)

769  } else {

771  }
