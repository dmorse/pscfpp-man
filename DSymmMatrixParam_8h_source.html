129 /*
130 * Read a DMatrix from isteam.
131 */
132 template <class Type>
133 void DSymmMatrixParam<Type>::readValue(std::istream &in)
134 {
135 // Preconditions
136 if (!(matrixPtr_->isAllocated())) {
137 UTIL_THROW("Cannot read unallocated DMatrix");
138 }
139 if (n() != matrixPtr_->capacity1()) {
140 UTIL_THROW("Error: Logical size n() != DMatrix<Type>::capacity1()");
141 }
142 if (n() != matrixPtr_->capacity2()) {
143 UTIL_THROW("Error: Logical size n() != DMatrix<Type>::capacity2()");
144 }
145
146 // Create matrix of int/bool flags for error checking
147 // Initialize all elements to zero
148 DMatrix<int> flags;
149 flags.allocate(n(), n());
150 int i, j;
151 for (i = 0; i < n(); ++i) {
152 for (j = 0; j < n(); ++j) {
153 flags(i, j) = 0;
154 (*matrixPtr_)(i, j) = 0.0;
155 }
156 }
157
158 double value;
159 for (int k = 0; k < n()*(n() + 1)/2; ++k) {
160 in >> i >> j >> value;
161 UTIL_CHECK(flags(i,j) == 0);
162 (*matrixPtr_)(i, j) = value;
163 flags(i, j) = 1;
164 if (i != j) {
165 UTIL_CHECK(flags(j,i) == 0);
166 (*matrixPtr_)(j, i) = value;
167 flags(j, i) = 1;
168 }
169 }
170
171 readEndBracket(in);
172 }

230 * Write a DSymmMatrixParam.
231 */
232 template <class Type>
233 void DSymmMatrixParam<Type>::writeParam(std::ostream &out)
234 {
235 if (isActive()) {
236 // Preconditions
237 if (!(matrixPtr_->isAllocated())) {
238 UTIL_THROW("Cannot read unallocated DMatrix");
239 }
240 if (n() > matrixPtr_->capacity1()) {
241 UTIL_THROW("Error: Logical size n() > DMatrix<Type>::capacity1()");
242 }
243 if (n() > matrixPtr_->capacity2()) {
244 UTIL_THROW("Error: Logical size n() > DMatrix<Type>::capacity2()");
245 }
246
247 if (hasBrackets()) {
248 out << indent() << label_ << std::endl;
249 }
250 Label space("");
251 int i, j;
252 for (i = 0; i < n(); ++i) {
253 for (j = 0; j <= i; ++j) {
254 if (i == 0 && j == 0 && !hasBrackets()) {
255 out << indent() << label_;
256 } else {
257 out << indent() << space;
258 }
259 out << Int(i, 4) << " " << Int(j, 4) << " "
260 << std::right << std::scientific
261 << std::setprecision(Parameter::Precision)
262 << std::setw(Parameter::Width)
263 << (*matrixPtr_)(i, j)
264 << std::endl;
265 }
266 }
267 writeEndBracket(out);
268 }
269 }
