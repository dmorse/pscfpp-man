23 static __global__
24 void makeDksqHelperWave(cudaReal* dksq, const int* waveBz,
25 const cudaReal* dkkBasis,
26 const int* partnerId,
27 const int* selfId,
28 const bool* implicit,
29 int nParams, int kSize,
30 int size, int dim)
31 {
32 // Actual size is nStar*nParams
33 // Each thread does nParams calculation
34 // Big performance hit if thread >= 0.5dimension(n-1)
35 int nThreads = blockDim.x * gridDim.x;
36 int startID = blockIdx.x * blockDim.x + threadIdx.x;
37
38 //loop through the entire array
39 int pId;
40 for (int param = 0; param < nParams; ++param) {
41 for (int i = startID; i < size; i += nThreads) {
42 for (int j = 0; j < dim; ++j) {
43 for (int k = 0; k < dim; ++k) {
44 if (!implicit[i]) {
45 // Not = need += so still need memset
46 dksq[(param * size) + i]
47 += waveBz[selfId[i] * dim + j]
48 * waveBz[ selfId[i] * dim + k]
49 * dkkBasis[k + (j * dim) + (param * dim * dim)];
50 } else {
51 pId = partnerId[i];
52 dksq[(param * size) + i] += waveBz[selfId[pId] * dim + j]
53 * waveBz[selfId[pId] * dim + k]
54 * dkkBasis[k + (j * dim) + (param * dim * dim)];
55 }
56 } //dim
57 } //dim
58 } //size
59 } //nParams
60 }

62 static __global__
63 void makeDksqReduction(cudaReal* dksq, const int* partnerId,
64 int nParams, int kSize, int rSize)
65 {
66 int nThreads = blockDim.x * gridDim.x;
67 int startID = blockIdx.x * blockDim.x + threadIdx.x;
68
69 // Add i in the implicit part into their partner's result
70 int pId;
71 for(int param = 0; param < nParams; ++param) {
72 for (int i = startID + kSize; i < rSize; i += nThreads) {
73 pId = partnerId[i];
74 dksq[(param * rSize) + pId] += dksq[(param * rSize) + i];
75 }
76 }

108 void WaveList<D>::allocate(Mesh<D> const & mesh,
109 UnitCell<D> const & unitCell) {
110
111 kSize_ = 1;
112 rSize_ = mesh.size();
113 nParams_ = unitCell.nParameter();
114 dimensions_ = mesh.dimensions();
115
116 for(int i = 0; i < D; ++i) {
117 if( i < D - 1) {
118 kSize_ *= mesh.dimension(i);
119 } else {
120 kSize_ *= (mesh.dimension(i)/ 2 + 1);
121 }
122 }
123
124 minImage_.allocate(kSize_);
125 gpuErrchk(cudaMalloc((void**) &minImage_d, sizeof(int) * rSize_ * D));

139 dkkBasis = new cudaReal[6 * D * D];

141

147 void WaveList<D>::computeMinimumImages(Mesh<D> const & mesh,
148 UnitCell<D> const & unitCell) {
149 MeshIterator<D> itr(mesh.dimensions());
150 IntVec<D> waveId;
151 IntVec<D> G2;
152 IntVec<D> tempIntVec;
153 int partnerId;
154
155 //min image needs mesh size of them
156 //partner only need kSize of them
157 //does setting iterator over kdim solves thing?
158 int kDimRank = 0;
159 int implicitRank = kSize_;
160 //kDimRank + implicitRank = rSize
161 int* invertedIdTable = new int[rSize_];
162
163 for (itr.begin(); !itr.atEnd(); ++itr) {
164 //if not implicit
165 if(itr.position(D - 1) < mesh.dimension(D-1)/2 + 1) {
166 implicit[kDimRank] = false;
167 selfIdTable[kDimRank] = itr.rank();
168 invertedIdTable[itr.rank()] = kDimRank;
169 kDimRank++;
170 } else {
171 implicit[implicitRank] = true;
172 selfIdTable[implicitRank] = itr.rank();
173 invertedIdTable[itr.rank()] = implicitRank;
174 implicitRank++;
175 }
176 }
177
178 int* tempMinImage = new int[rSize_ * D];
179 for (itr.begin(); !itr.atEnd(); ++itr) {
180 kSq_[itr.rank()] = unitCell.ksq(itr.position());
181
182 #if 0
183 //we get position but set mesh dim to be larger, should be okay
184 shiftToMinimum(itr.position(), mesh.dimensions(), minImage_ + (itr.rank() * D));
185 #endif
186
187 // We get position but set mesh dim to be larger, should be okay
188 // not the most elegant code with repeated copying but reduces
189 // repeated code from pscf
190 waveId = itr.position();
191 tempIntVec = shiftToMinimum(waveId, mesh.dimensions(), unitCell);
192 for(int i = 0; i < D; i++) {
193 (tempMinImage + (itr.rank() * D))[i] = tempIntVec[i];
194 }
195
196 if(itr.position(D - 1) < mesh.dimension(D-1)/2 + 1) {
197 minImage_[invertedIdTable[itr.rank()]] = tempIntVec;
198 }
199
200 for(int j = 0; j < D; ++j) {
201 G2[j] = -waveId[j];
202 }
203 mesh.shift(G2);
204 partnerId = mesh.rank(G2);
205 partnerIdTable[invertedIdTable[itr.rank()]] = invertedIdTable[partnerId];
206 }
207
208 /*std::cout<<"Sum kDimRank implicitRank: "<<kDimRank + (implicitRank-kSize_)<<std::endl;
209 std::cout<<"This is kDimRank sanity check "<<kDimRank<<std::endl;
210 for(int i = 0; i < rSize_; i++) {
211 std::cout<<i<<' '<<selfIdTable[i]<<' '<<partnerIdTable[i]<<' '<<implicit[i]<<std::endl;
212 }*/
213 gpuErrchk(cudaMemcpy(minImage_d, tempMinImage, sizeof(int) * rSize_ * D, cudaMemcpyHostToDevice));
214
215 //partner is much smaller but we keep this for now
216 gpuErrchk(cudaMemcpy(partnerIdTable_d, partnerIdTable, sizeof(int) * mesh.size(), cudaMemcpyHostToDevice));
217
218 gpuErrchk(cudaMemcpy(selfIdTable_d, selfIdTable, sizeof(int) * mesh.size(), cudaMemcpyHostToDevice));
219
220 gpuErrchk(cudaMemcpy(implicit_d, implicit, sizeof(bool) * mesh.size(), cudaMemcpyHostToDevice));
221
222 delete[] tempMinImage;
223 computedKSq(unitCell);
224 }

231 template <int D>
232 void WaveList<D>::computedKSq(UnitCell<D> const & unitCell){
233 //dkkbasis is something determined from unit cell size
234 //min image needs to be on device but okay since its only done once
235 //second to last parameter is number of stars originally
236
237 // GPU resources
238 int nBlocks, nThreads;
239 ThreadGrid::setThreadsLogical(rSize_, nBlocks, nThreads);
240
241
242 int idx;
243 for(int i = 0 ; i < unitCell.nParameter(); ++i) {
244 for(int j = 0; j < D; ++j) {
245 for(int k = 0; k < D; ++k) {
246 idx = k + (j * D) + (i * D * D);
247 dkkBasis[idx] = unitCell.dkkBasis(i, j, k);
248 }
249 }
250 }
251
252 cudaMemcpy(dkkBasis_d, dkkBasis,
253 sizeof(cudaReal) * unitCell.nParameter() * D * D,
254 cudaMemcpyHostToDevice);
255
256 cudaMemset(dkSq_, 0, unitCell.nParameter() * rSize_ * sizeof(cudaReal));
257 makeDksqHelperWave<<<nBlocks, nThreads>>>
258 (dkSq_, minImage_d, dkkBasis_d, partnerIdTable_d,
259 selfIdTable_d, implicit_d, unitCell.nParameter(),
260 kSize_, rSize_, D);
261
262 makeDksqReduction<<<nBlocks, nThreads>>>
263 (dkSq_, partnerIdTable_d, unitCell.nParameter(),
264 kSize_, rSize_);
265 }
