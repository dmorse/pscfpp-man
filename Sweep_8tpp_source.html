95 /*
96 * Create guess for adjustable variables by polynomial extrapolation.
97 */
98 template <int D>
99 void Sweep<D>::extrapolate(double sNew)
100 {
101 UTIL_CHECK(historySize() > 0);
102
103 // If historySize() == 1, do nothing: Use previous system state
104 // as trial for the new state.
105
106 if (historySize() > 1) {
107 UTIL_CHECK(historySize() <= historyCapacity());
108
109 // Does the iterator allow a flexible unit cell ?
110 bool isFlexible = system().iterator().isFlexible();
111
112 // Compute coefficients of polynomial extrapolation to sNew
113 setCoefficients(sNew);
114
115 // Set extrapolated trial w fields
116 double coeff;
117 int nMonomer = system().mixture().nMonomer();
118 int nBasis = system().basis().nBasis();
119 DArray<double>* newFieldPtr;
120 DArray<double>* oldFieldPtr;
121 int i, j, k;
122 for (i=0; i < nMonomer; ++i) {
123 newFieldPtr = &(trial_.field(i));
124
125 // Previous state k = 0 (most recent)
126 oldFieldPtr = &state(0).field(i);
127 coeff = c(0);
128 for (j=0; j < nBasis; ++j) {
129 (*newFieldPtr)[j] = coeff*(*oldFieldPtr)[j];
130 }
131
132 // Previous states k >= 1 (older)
133 for (k = 1; k < historySize(); ++k) {
134 oldFieldPtr = &state(k).field(i);
135 coeff = c(k);
136 for (j=0; j < nBasis; ++j) {
137 (*newFieldPtr)[j] += coeff*(*oldFieldPtr)[j];
138 }
139 }
140 }
141
142 // IfisFlexible, then extrapolate unit cell parameters
143 if (isFlexible) {
144
145 double coeff;
146 double parameter;
147 int nParameter = system().unitCell().nParameter();
148
149 // Append contributions from k= 0 (most recent state)
150 coeff = c(0);
151 unitCellParameters_.clear();
152 for (int i = 0; i < nParameter; ++i) {
153 parameter = state(0).unitCell().parameter(i);
154 unitCellParameters_.append(coeff*parameter);
155 }
156
157 // Add contributions from k > 0 (older states)
158 for (k = 1; k < historySize(); ++k) {
159 coeff = c(k);
160 for (int i = 0; i < nParameter; ++i) {
161 parameter = state(k).unitCell().parameter(i);
162 unitCellParameters_[i] += coeff*parameter;
163 }
164 }
165
166 // Reset trial_.unitCell() object
167 trial_.unitCell().setParameters(unitCellParameters_);
168 }
169
170 // Transfer data from trial_ state to parent system
171 trial_.setSystemState(isFlexible);
172 }
173
174
175 };

241 out.close();
