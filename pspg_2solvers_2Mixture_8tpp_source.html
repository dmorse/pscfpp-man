46 template <int D>
47 void Mixture<D>::setMesh(Mesh<D> const& mesh)
48 {
49 UTIL_CHECK(nMonomer() > 0);
50 UTIL_CHECK(nPolymer()+ nSolvent() > 0);
51 UTIL_CHECK(ds_ > 0);
52
53 meshPtr_ = &mesh;
54
55 // Set discretization for all blocks
56 int i, j;
57 for (i = 0; i < nPolymer(); ++i) {
58 for (j = 0; j < polymer(i).nBlock(); ++j) {
59 polymer(i).block(j).setDiscretization(ds_, mesh);
60 }
61 }
62
63 // Set spatial discretization for solvents
64 if (nSolvent() > 0) {
65 for (int i = 0; i < nSolvent(); ++i) {
66 solvent(i).setDiscretization(mesh);
67 }
68 }
69 }

85 template <int D>
86 void Mixture<D>::setKuhn(int monomerId, double kuhn)
87 {
88 // Set new Kuhn length for relevant Monomer object
89 monomer(monomerId).setKuhn(kuhn);
90
91 // Update kuhn length for all blocks of this monomer type
92 for (int i = 0; i < nPolymer(); ++i) {
93 for (int j = 0; j < polymer(i).nBlock(); ++j) {
94 if (monomerId == polymer(i).block(j).monomerId()) {
95 polymer(i).block(j).setKuhn(kuhn);
96 }
97 }
98 }
99 hasStress_ = false;

103 /*
104 * Compute concentrations (but not total free energy).
105 */
106 template <int D>
107 void Mixture<D>::compute(DArray< RDField<D> > const & wFields,
108 DArray< RDField<D> > & cFields)
109 {
110 UTIL_CHECK(meshPtr_);
111 UTIL_CHECK(mesh().size() > 0);
112 UTIL_CHECK(nMonomer() > 0);
113 UTIL_CHECK(nPolymer() + nSolvent() > 0);
114 UTIL_CHECK(wFields.capacity() == nMonomer());
115 UTIL_CHECK(cFields.capacity() == nMonomer());
116
117 int nMesh = mesh().size();
118 int nm = nMonomer();
119 int i, j;
120
121 // GPU resources
122 int nBlocks, nThreads;
123 ThreadGrid::setThreadsLogical(nMesh, nBlocks, nThreads);
124
125 // Clear all monomer concentration fields
126 for (i = 0; i < nm; ++i) {
127 UTIL_CHECK(cFields[i].capacity() == nMesh);
128 UTIL_CHECK(wFields[i].capacity() == nMesh);
129 assignUniformReal<<<nBlocks, nThreads>>>(cFields[i].cDField(),
130 0.0, nMesh);
131 }
132
133 // Solve MDE for all polymers
134 for (i = 0; i < nPolymer(); ++i) {
135 polymer(i).compute(wFields);
136 }
137
138 // Accumulate monomer concentration fields
139 int monomerId;
140 for (i = 0; i < nPolymer(); ++i) {
141 for (j = 0; j < polymer(i).nBlock(); ++j) {
142 monomerId = polymer(i).block(j).monomerId();
143 UTIL_CHECK(monomerId >= 0);
144 UTIL_CHECK(monomerId < nm);
145 RDField<D>& monomerField = cFields[monomerId];
146 RDField<D>& blockField = polymer(i).block(j).cField();
147 UTIL_CHECK(blockField.capacity()==nMesh);
148 pointWiseAdd<<<nBlocks, nThreads>>>(monomerField.cDField(),
149 blockField.cDField(), nMesh);
150 }
151 }
152
153 // Process solvent species
154 for (i = 0; i < nSolvent(); i++) {
155 monomerId = solvent(i).monomerId();
156 UTIL_CHECK(monomerId >= 0);
157 UTIL_CHECK(monomerId < nm);
158
159 // Compute solvent concentration
160 solvent(i).compute(wFields[monomerId]);
161
162 // Add solvent contribution to relevant monomer concentration
163 RDField<D>& monomerField = cFields[monomerId];
164 RDField<D> const & solventField = solvent(i).concField();
165 UTIL_CHECK(solventField.capacity() == nMesh);
166 pointWiseAdd<<<nBlocks, nThreads>>>(monomerField.cDField(),
167 solventField.cDField(),
168 nMesh);
169
170
171 }
172
173 hasStress_ = false;
174 }

176 /*
177 * Compute Total Stress.
178 */
179 template <int D>
180 void Mixture<D>::computeStress(WaveList<D> const & wavelist)
181 {
182 int i, j;
183
184 // Compute stress for each polymer.
185 for (i = 0; i < nPolymer(); ++i) {
186 polymer(i).computeStress(wavelist);
187 }
188
189 // Accumulate total stress
190 for (i = 0; i < nUnitCellParams_; ++i) {
191 stress_[i] = 0.0;
192 for (j = 0; j < nPolymer(); ++j) {
193 stress_[i] += polymer(j).stress(i);
194 }
195 }
196 // Note: Solvent does not contribute to derivatives of f_Helmholtz
197 // with respect to unit cell parameters at fixed volume fractions.
198
199 hasStress_ = true;
200 }

202 template <int D>
203 bool Mixture<D>::isCanonical()
204 {
205 // Check ensemble of all polymers
206 for (int i = 0; i < nPolymer(); ++i) {
207 if (polymer(i).ensemble() == Species::Open) {
208 return false;
209 }
210 }
211 // Check ensemble of all solvents
212 for (int i = 0; i < nSolvent(); ++i) {
213 if (solvent(i).ensemble() == Species::Open) {
214 return false;
215 }
216 }
217 // Returns true if false was never returned
218 return true;
219 }

221 /*
222 * Combine cFields for each block (and solvent) into one DArray
223 */
224 template <int D>
225 void Mixture<D>::createBlockCRGrid(DArray< RDField<D> > & blockCFields)
226 const
227 {
228 UTIL_CHECK(nMonomer() > 0);
229 UTIL_CHECK(nBlock() + nSolvent() > 0);
230
231 int np = nSolvent() + nBlock();
232 int nx = mesh().size();
233 int i, j;
234
235 UTIL_CHECK(blockCFields.capacity() == nBlock() + nSolvent());
236
237 // GPU resources
238 int nBlocks, nThreads;
239 ThreadGrid::setThreadsLogical(nx, nBlocks, nThreads);
240
241 // Clear all monomer concentration fields, check capacities
242 for (i = 0; i < np; ++i) {
243 UTIL_CHECK(blockCFields[i].capacity() == nx);
244 assignUniformReal<<<nBlocks, nThreads>>>(blockCFields[i].cDField(), 0.0, nx);
245 }
246
247 // Process polymer species
248 int sectionId = -1;
249
250 if (nPolymer() > 0) {
251
252 // Write each block's r-grid data to blockCFields
253 for (i = 0; i < nPolymer(); ++i) {
254 for (j = 0; j < polymer(i).nBlock(); ++j) {
255 sectionId++;
256
257 UTIL_CHECK(sectionId >= 0);
258 UTIL_CHECK(sectionId < np);
259 UTIL_CHECK(blockCFields[sectionId].capacity() == nx);
260
261 const cudaReal* blockField = polymer(i).block(j).cField().cDField();
262 cudaMemcpy(blockCFields[sectionId].cDField(), blockField,
263 mesh().size() * sizeof(cudaReal), cudaMemcpyDeviceToDevice);
264 }
265 }
266 }
267
268 // Process solvent species
269 if (nSolvent() > 0) {
270
271 // Write each solvent's r-grid data to blockCFields
272 for (i = 0; i < nSolvent(); ++i) {
273 sectionId++;
274
275 UTIL_CHECK(sectionId >= 0);
276 UTIL_CHECK(sectionId < np);
277 UTIL_CHECK(blockCFields[sectionId].capacity() == nx);
278
279 const cudaReal* solventField = solvent(i).concField().cDField();
280 cudaMemcpy(blockCFields[sectionId].cDField(), solventField,
281 mesh().size() * sizeof(cudaReal), cudaMemcpyDeviceToDevice);
282 }
283 }
284 }
