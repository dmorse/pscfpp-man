88  void Mixture<D>::compute(DArray<Mixture<D>::WField> const & wFields,
89  DArray<Mixture<D>::CField>& cFields)
90  {
91  UTIL_CHECK(meshPtr_);
92  UTIL_CHECK(mesh().size() > 0);
93  UTIL_CHECK(nMonomer() > 0);
94  UTIL_CHECK(nPolymer() + nSolvent() > 0);
95  UTIL_CHECK(wFields.capacity() == nMonomer());
96  UTIL_CHECK(cFields.capacity() == nMonomer());
97 
98  int nx = mesh().size();
99  int nm = nMonomer();
100  int i, j;
101 
102  // Clear all monomer concentration fields
103  for (i = 0; i < nm; ++i) {
104  UTIL_CHECK(cFields[i].capacity() == nx);
105  UTIL_CHECK(wFields[i].capacity() == nx);
106  //cFields[i][j] = 0.0;
107  assignUniformReal<<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK>>>(cFields[i].cDField(), 0.0, nx);
108  }
109 
110  // Solve MDE for all polymers
111  for (i = 0; i < nPolymer(); ++i) {
112  polymer(i).compute(wFields);
113  }
114 
115  // Accumulate monomer concentration fields
116  for (i = 0; i < nPolymer(); ++i) {
117  for (j = 0; j < polymer(i).nBlock(); ++j) {
118  int monomerId = polymer(i).block(j).monomerId();
119  UTIL_CHECK(monomerId >= 0);
120  UTIL_CHECK(monomerId < nm);
121  CField& monomerField = cFields[monomerId];
122  CField& blockField = polymer(i).block(j).cField();
123  //monomerField[k] += polymer(i).phi() * blockField[k];
124  accumulateConc<<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK>>>(monomerField.cDField(),
125  polymer(i).phi(), blockField.cDField(), nx);
126  }
127  }
