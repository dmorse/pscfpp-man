153  /*
154  * Write incoming q fields for a specified vertex.
155  */
156  void FieldIo::writeVertexQ(int polymerId, int vertexId,
157  std::ostream& out)
158  {
159  Polymer const & polymer = mixture().polymer(polymerId);
160  Vertex const & vertex = polymer.vertex(vertexId);
161  Pair<int> pId;
162  int bId;
163  int dId;
164  int nb = vertex.size();
165  int nx = domain().nx(); // number grid points
166  int i, j;
167  double c, product;
168  for (i = 0; i < nx; ++i) {
169  out << Int(i, 5);
170  product = 1.0;
171  for (j = 0; j < nb; ++j) {
172  pId = vertex.inPropagatorId(j);
173  bId = pId[0]; // blockId
174  dId = pId[1]; // directionId
175  c = polymer.propagator(bId, dId).tail()[i];
176  product *= c;
177  out << " " << Dbl(c, 15, 8);
178  }
179  out << " " << Dbl(product, 15, 8) << std::endl;
180  }

193  /*
194  * Interpolate fields onto new mesh.
195  */
196  void
197  FieldIo::remesh(Array<Field> const & fields, int nx, std::ostream& out)
198  {
199  int nm = mixture().nMonomer();
200  out << "nx " << nx << std::endl;
201  out << "nm " << nm << std::endl;
202 
203  // Output first grid point
204  int i, j;
205  i = 0;
206  out << Int(i, 5);
207  for (j = 0; j < nm; ++j) {
208  out << " " << Dbl(fields[j][i]);
209  }
210  out << std::endl;
211 
212  // Spacing for new grid
213  double dx = (domain().xMax() - domain().xMin())/double(nx-1);
214 
215  double y; // Grid coordinate in old grid
216  double fu; // fractional part of y
217  double fl; // 1.0 - fl
218  double w; // interpolated field value
219  int yi; // Truncated integer part of y
220 
221  // Loop over intermediate points
222  for (i = 1; i < nx -1; ++i) {
223  y = dx*double(i)/domain().dx();
224  yi = y;
225  fu = y - double(yi);
226  fl = 1.0 - fu;
227 
228  out << Int(i, 5);
229  for (j = 0; j < nm; ++j) {
230  w = fl*fields[j][yi] + fu*fields[j][yi+1];
231  out << " " << Dbl(w);
232  }
233  out << std::endl;
234  }
235 
236  // Output last grid point
237  out << Int(nx - 1, 5);
238  i = domain().nx() - 1;
239  for (j = 0; j < nm; ++j) {
240  out << " " << Dbl(fields[j][i]);
241  }
242  out << std::endl;
243 
244  }
