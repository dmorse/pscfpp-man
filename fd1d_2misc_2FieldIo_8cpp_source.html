60 void FieldIo::readFields(DArray<Field>& fields, std::istream& in)
61 {
62 // Read grid dimensions
63 std::string label;
64 in >> label;
65 UTIL_CHECK(label == "nx");
66 int nx;
67 in >> nx;
68 UTIL_CHECK(nx > 0);
69 UTIL_CHECK(nx == domain().nx());
70 in >> label;
71 UTIL_CHECK (label == "nm");
72 int nm;
73 in >> nm;
74 UTIL_CHECK(nm > 0);
75
76 // Check dimensions of fields array
77 UTIL_CHECK(nm == fields.capacity());
78 for (int i = 0; i < nm; ++i) {
79 UTIL_CHECK(nx == fields[i].capacity());
80 }
81
82 // Read fields
83 int i, j, idum;
84 for (i = 0; i < nx; ++i) {
85 in >> idum;
86 UTIL_CHECK(idum == i);
87 for (j = 0; j < nm; ++j) {
88 in >> fields[j][i];
89 }
90 }
91 }

136 /*
137 * Write the concentrations associated with all blocks.
138 */
139 void FieldIo::writeBlockCFields(Mixture const & mixture, std::ostream& out)
140 {
141 int nx = domain().nx(); // number grid points
142 int np = mixture.nPolymer(); // number of polymer species
143 int nb_tot = mixture.nBlock(); // number of blocks in all polymers
144 int ns = mixture.nSolvent(); // number of solvents
145
146 out << "nx " << nx << std::endl;
147 out << "n_block " << nb_tot << std::endl;
148 out << "n_solvent " << ns << std::endl;
149
150 int nb; // number of blocks per polymer
151 int i, j, k, l; // loop indices
152 double c;
153 for (i = 0; i < nx; ++i) {
154 out << Int(i, 5);
155 for (j = 0; j < np; ++j) {
156 nb = mixture.polymer(j).nBlock();
157 for (k = 0; k < nb; ++k) {
158 c = mixture.polymer(j).block(k).cField()[i];
159 out << " " << Dbl(c, 15, 8);
160 }
161 }
162 for (l = 0; l < ns; ++l) {
163 c = mixture.solvent(l).cField()[i];
164 out << " " << Dbl(c, 15, 8);
165 }
166 out << std::endl;
167 }

183 /*
184 * Write incoming q fields for a specified vertex.
185 */
186 void FieldIo::writeVertexQ(Mixture const & mixture,
187 int polymerId, int vertexId,
188 std::ostream& out)
189 {
190 UTIL_CHECK(polymerId >= 0);
191 UTIL_CHECK(polymerId < mixture.nPolymer());
192 Polymer const & polymer = mixture.polymer(polymerId);
193 UTIL_CHECK(vertexId >= 0);
194 UTIL_CHECK(vertexId <= polymer.nBlock());
195 Vertex const & vertex = polymer.vertex(vertexId);
196 int nb = vertex.size(); // number of attached blocks
197 int nx = domain().nx(); // number grid points
198
199 Pair<int> pId;
200 int bId;
201 int dId;
202 int i, j;
203 double c, product;
204 for (i = 0; i < nx; ++i) {
205 out << Int(i, 5);
206 product = 1.0;
207 for (j = 0; j < nb; ++j) {
208 pId = vertex.inPropagatorId(j);
209 bId = pId[0]; // blockId
210 UTIL_CHECK(bId >= 0);
211 dId = pId[1]; // directionId
212 UTIL_CHECK(dId >= 0);
213 UTIL_CHECK(dId <= 1);
214 c = polymer.propagator(bId, dId).tail()[i];
215 out << " " << Dbl(c, 15, 8);
216 product *= c;
217 }
218 out << " " << Dbl(product, 15, 8) << std::endl;
219 }

235 /*
236 * Interpolate fields onto new mesh, write result to outputstream
237 */
238 void
239 FieldIo::remesh(DArray<Field> const & fields, int nx, std::ostream& out)
240 {
241 // Query and check dimensions of fields array
242 int nm = fields.capacity();
243 UTIL_CHECK(nm > 0);
244 for (int i = 0; i < nm; ++i) {
245 UTIL_CHECK(fields[i].capacity() == domain().nx());
246 }
247
248 // Output new grid dimensions
249 out << "nx " << nx << std::endl;
250 out << "nm " << nm << std::endl;
251
252 // Output first grid point
253 int i, j;
254 i = 0;
255 out << Int(i, 5);
256 for (j = 0; j < nm; ++j) {
257 out << " " << Dbl(fields[j][i]);
258 }
259 out << std::endl;
260
261 // Spacing for new grid
262 double dx = (domain().xMax() - domain().xMin())/double(nx-1);
263
264 // Variables used for interpolation
265 double y; // Grid coordinate in old grid
266 double fu; // fractional part of y
267 double fl; // 1.0 - fl
268 double w; // interpolated field value
269 int yi; // Truncated integer part of y
270
271 // Loop over intermediate points
272 for (i = 1; i < nx -1; ++i) {
273 y = dx*double(i)/domain().dx();
274 yi = y;
275 UTIL_CHECK(yi >= 0);
276 UTIL_CHECK(yi + 1 < domain().nx());
277 fu = y - double(yi);
278 fl = 1.0 - fu;
279
280 out << Int(i, 5);
281 for (j = 0; j < nm; ++j) {
282 w = fl*fields[j][yi] + fu*fields[j][yi+1];
283 out << " " << Dbl(w);
284 }
285 out << std::endl;
286 }
287
288 // Output last grid point
289 out << Int(nx - 1, 5);
290 i = domain().nx() - 1;
291 for (j = 0; j < nm; ++j) {
292 out << " " << Dbl(fields[j][i]);
293 }
294 out << std::endl;
295
296 }

308 /*
309 * Interpolate fields onto new mesh.
310 */
311 void
312 FieldIo::extend(DArray<Field> const & fields, int m, std::ostream& out)
313 {
314 // Query and check dimensions of fields array
315 int nm = fields.capacity();
316 UTIL_CHECK(nm > 0);
317 int nx = fields[0].capacity();
318 UTIL_CHECK(nx == domain().nx());
319 if (nm > 1) {
320 for (int i = 0; i < nm; ++i) {
321 UTIL_CHECK(nx == fields[i].capacity());
322 }
323 }
324
325 // Output new grid dimensions
326 out << "nx " << nx + m << std::endl;
327 out << "nm " << nm << std::endl;
328
329 // Loop over existing points;
330 int i, j;
331 for (i = 0; i < nx; ++i) {
332 out << Int(i, 5);
333 for (j = 0; j < nm; ++j) {
334 out << " " << Dbl(fields[j][i]);
335 }
336 out << std::endl;
337 }
338
339 // Add m new points, assign each the last value
340 for (i = nx; i < nx + m; ++i) {
341 out << Int(i, 5);
342 for (j = 0; j < nm; ++j) {
343 out << " " << Dbl(fields[j][nx-1]);
344 }
345 out << std::endl;
346 }
