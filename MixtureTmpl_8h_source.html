246 /*
247 * Read all parameters and initialize.
248 */
249 template <class TP, class TS>
250 void MixtureTmpl<TP,TS>::readParameters(std::istream& in)
251 {
252 // Read monomers
253 read<int>(in, "nMonomer", nMonomer_);
254 monomers_.allocate(nMonomer_);
255 for (int i = 0; i < nMonomer_; ++i) {
256 monomers_[i].setId(i);
257 }
258 readDArray< Monomer >(in, "monomers", monomers_, nMonomer_);
259
260 /*
261 * The input format for a single monomer is defined in the istream
262 * extraction operation (operator >>) for a Pscf::Monomer, in file
263 * pscf/chem/Monomer.cpp. The format consists of the monomer name
264 * string Monomer::name followed by monomer statistical segment
265 * Monomer::kuhn on a single line.
266 */
267
268 // Read nPolymer and (optionally) nSolvent, with nSolvent=0 by default
269 read<int>(in, "nPolymer", nPolymer_);
270 UTIL_CHECK(nPolymer_ > 0);
271
272 // Optionally read nSolvent, with nSolvent=0 by default
273 nSolvent_ = 0;
274 readOptional<int>(in, "nSolvent", nSolvent_);
275
276 // Read polymers
277 if (nPolymer_ > 0) {
278
279 polymers_.allocate(nPolymer_);
280 for (int i = 0; i < nPolymer_; ++i) {
281 readParamComposite(in, polymer(i));
282 }
283
284 // Set statistical segment lengths for all blocks
285 double kuhn;
286 int monomerId;
287 for (int i = 0; i < nPolymer_; ++i) {
288 for (int j = 0; j < polymer(i).nBlock(); ++j) {
289 monomerId = polymer(i).block(j).monomerId();
290 kuhn = monomer(monomerId).kuhn();
291 polymer(i).block(j).setKuhn(kuhn);
292 }
293 }
294
295 }
296
297 // Read solvents
298 if (nSolvent_ > 0) {
299
300 solvents_.allocate(nSolvent_);
301 for (int i = 0; i < nSolvent_; ++i) {
302 readParamComposite(in, solvent(i));
303 }
304
305 }
306
307 }
