21 template <int D>
22 bool
23 SpaceGroup<D>::hasInversionCenter(
24 typename SpaceSymmetry<D>::Translation& center)
25 const
26 {
27 bool isInversion = false;
28 int i, j, k;
29 for (i = 0; i < size(); ++i) {
30 isInversion = true;
31 for (j = 0; j < D; ++j) {
32 for (k = 0; k < D; ++k) {
33 if (j == k) {
34 if ((*this)[i].R(j,k) != -1) isInversion = false;
35 } else {
36 if ((*this)[i].R(j,k) != 0) isInversion = false;
37 }
38 }
39 }
40 if (isInversion) {
41 for (int j = 0; j < D; ++j) {
42 center[j] = (*this)[i].t(j)/2;
43 }
44 return true;
45 }
46 }
47 return false;
48 }
