264 inline
265 bool FFT<D>::hasImplicitInverse(IntVec<D> const & wavevector,
266 IntVec<D> const & meshDimensions)
267 {
268 int i = wavevector[D-1];
269 int d = meshDimensions[D-1];
270 if ((i != 0) && (d - i > d/2 + 1)) {
271 return true;
272 } else {
273 return false;
274 }
275 }
