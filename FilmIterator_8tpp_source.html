72 /*
73 * Output the indices of each flexible lattice parameter, based on
74 * normalVecId and unitCell definitions in param file
75 */
76 template <typename IteratorType>
77 void FilmIterator<2, IteratorType>::setFlexibleParams()
78 {
79 FSArray<int, 6> params;
80
81 // If the lattice system is square, params should be empty. Otherwise,
82 // params should contain only the length of the vector that is not
83 // normalVecId. The length of normalVecId is fixed, and gamma = 90 degrees.
84 if (system().domain().unitCell().lattice() != UnitCell<2>::Square) {
85 if (normalVecId() == 0) {
86 params.append(1);
87 } else { // normalVecId() == 1
88 params.append(0);
89 }
90 }
91
92 if (params.size() == 0) {
93 Log::file() << std::endl
94 << "Warning - None of the lattice parameters are allowed\n"
95 << "to be flexible for this choice of lattice system\n"
96 << "when the system is confined in a thin film."
97 << std::endl;
98 }
99
100 // Store params in flexibleParams_ member of this object
101 setFlexibleParams(params);
102
103 // Pass params into the iterator member of this object
104 iterator().setFlexibleParams(params);
105 }

109 * the thin film constraint (in 2D, we require that gamma = 90°)

111 template <typename IteratorType>

114 RealVec<2> a, b;
115 a = system().domain().unitCell().rBasis(0);
116 b = system().domain().unitCell().rBasis(1);
117
118 double gamma = dot(a,b);

120 UTIL_THROW("ERROR: Lattice basis vectors must be orthogonal when wall is present");

135 /*
136 * Output the indices of each flexible lattice parameter, based on
137 * normalVecId and unitCell definitions in param file
138 */
139 template <typename IteratorType>
140 void FilmIterator<3, IteratorType>::setFlexibleParams()
141 {
142 FSArray<int, 6> params;
143 UnitCell<3>::LatticeSystem lattice = system().domain().unitCell().lattice();
144
145 // params can contain up to 3 lattice parameters: the length of the two
146 // lattice vectors that are not normalVecId, and the angle between them.
147 // The other two angles must be 90 degrees, and the length of normalVecId
148 // is fixed. The crystal system determines which parameters are flexible.
149 if (lattice == UnitCell<3>::Hexagonal ||
150 lattice == UnitCell<3>::Rhombohedral) {
151 UTIL_CHECK(normalVecId() == 2); // this is required for hex/rhombohedral
152 params.append(0);
153 } else if (lattice == UnitCell<3>::Tetragonal) {
154 if (normalVecId() == 2) {
155 params.append(0);
156 } else {
157 params.append(1);
158 }
159 } else if (lattice != UnitCell<3>::Cubic) {
160 // The for-loop below applies to orthorhombic, monoclinic, & triclinic
161 for (int i = 0; i < 3; i++) {
162 if (normalVecId() != i) {
163 params.append(i);
164 }
165 }
166 if (lattice == UnitCell<3>::Monoclinic && normalVecId() == 1) {
167 params.append(3); // beta is flexible if normalVecId == 1
168 } else if (lattice == UnitCell<3>::Triclinic) {
169 params.append(normalVecId() + 3);
170 }
171 }
172
173 if (params.size() == 0) {
174 Log::file() << std::endl
175 << "Warning - None of the lattice parameters are allowed\n"
176 << "to be flexible for this choice of lattice system\n"
177 << "when the system is confined in a thin film."
178 << std::endl;
179 }
180
181 // Store params in flexibleParams_ member of this object
182 setFlexibleParams(params);
183
184 // Pass params into the iterator member of this object
185 iterator().setFlexibleParams(params);
186 }

188 /*
189 * Check that user-defined lattice basis vectors are compatible with
190 * the thin film constraint (in 3D, we require that there be one
191 * lattice basis vector that is orthogonal to the walls, and two that
192 * are parallel to the walls; the orthogonal vector is normalVecId).
193 */
194 template <typename IteratorType>
195 void FilmIterator<3, IteratorType>::checkLatticeVectors() const
196 {
197 RealVec<3> a, b, c;
198 a = system().domain().unitCell().rBasis(0);
199 b = system().domain().unitCell().rBasis(1);
200 c = system().domain().unitCell().rBasis(2);
201 double alpha, beta, gamma;
202 gamma = dot(a,b);
203 beta = dot(a,c);
204 alpha = dot(b,c);
205
206 if (normalVecId() == 0) {
207 if (beta > 1e-8 || gamma > 1e-8) {
208 UTIL_THROW("ERROR: If normalVecId = 0, beta and gamma must be 90 degrees");
209 }
210 } else if (normalVecId() == 1) {
211 if (alpha > 1e-8 || gamma > 1e-8) {
212 UTIL_THROW("ERROR: If normalVecId = 1, alpha and gamma must be 90 degrees");
213 }
214 } else { // normalVecId == 2
215 if (alpha > 1e-8 || beta > 1e-8) {
216 UTIL_THROW("ERROR: If normalVecId = 2, alpha and beta must be 90 degrees");
217 }
218 }
219 }
