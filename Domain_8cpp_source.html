91  void Domain::computeVolume()
92  {
93  if (mode_ == Planar) {
94  volume_ = xMax_ - xMin_;
95  } else
96  if (mode_ == Cylindrical) {
97  volume_ = xMax_*xMax_;
98  if (isShell_) {
99  volume_ -= xMin_*xMin_;
100  }
101  volume_ *= Constants::Pi;
102  } else
103  if (mode_ == Spherical) {
104  volume_ = xMax_*xMax_*xMax_;
105  if (isShell_) {
106  volume_ -= xMin_*xMin_*xMin_;
107  }
108  volume_ *= Constants::Pi*(4.0/3.0);
109  } else {
110  UTIL_THROW("Invalid geometry mode");
111  }
112  }

114  /*
115  * Compute spatial average of a field.
116  */
117  double Domain::spatialAverage(Field const & f) const
118  {
119  UTIL_CHECK(nx_ > 1);
120  UTIL_CHECK(dx_ > 0.0);
121  UTIL_CHECK(xMax_ - xMin_ >= dx_);
122  UTIL_CHECK(nx_ == f.capacity());
123 
124  double sum = 0.0;
125  double norm = 0.0;
126  if (mode_ == Planar) {
127 
128  sum += 0.5*f[0];
129  for (int i = 1; i < nx_ - 1; ++i) {
130  sum += f[i];
131  }
132  sum += 0.5*f[nx_ - 1];
133  norm = double(nx_ - 1);
134 
135  } else
136  if (mode_ == Cylindrical) {
137 
138  double x0 = xMin_/dx_;
139 
140  // First value
141  if (isShell_) {
142  sum += 0.5*x0*f[0];
143  norm += 0.5*x0;
144  } else {
145  sum += f[0]/8.0;
146  norm += 1.0/8.0;
147  }
148 
149  // Interior values
150  double x;
151  for (int i = 1; i < nx_ - 1; ++i) {
152  x = x0 + double(i);
153  sum += x*f[i];
154  norm += x;
155  }
156 
157  // Last value
158  x = x0 + double(nx_-1);
159  sum += 0.5*x*f[nx_-1];
160  norm += 0.5*x;
161 
162  } else
163  if (mode_ == Spherical) {
164 
165  double x0 = xMin_/dx_;
166 
167  // First value
168  if (isShell_) {
169  sum += 0.5*x0*x0*f[0];
170  norm += 0.5*x0*x0;
171  } else {
172  sum += f[0]/24.0;
173  norm += 1.0/24.0;
174  }
175 
176  // Interior values
177  double x;
178  for (int i = 1; i < nx_ - 1; ++i) {
179  x = x0 + double(i);
180  sum += x*x*f[i];
181  norm += x*x;
182  }
183 
184  // Last value
185  x = x0 + double(nx_-1);
186  sum += 0.5*x*x*f[nx_-1];
187  norm += 0.5*x*x;
188 
189  } else {
190  UTIL_THROW("Invalid geometry mode");
191  }
192 
193  return sum/norm;
194  }

200  {
201  // Preconditions
202  UTIL_CHECK(nx_ > 1);
203  UTIL_CHECK(nx_ == f.capacity());
204  UTIL_CHECK(nx_ == g.capacity());
205  if (!work_.isAllocated()) {
206  work_.allocate(nx_);
207  } else {
208  UTIL_ASSERT(nx_ == work_.capacity());
209  }
210 
211  // Compute average of f(x)*g(x)
212  for (int i = 0; i < nx_; ++i) {
