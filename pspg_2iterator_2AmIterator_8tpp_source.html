101 template <int D>
102 void AmIterator<D>::updateU(DMatrix<double> & U, RingBuffer<FieldCUDA> const & resBasis, int nHist)
103 {
104 // Update matrix U by shifting elements diagonally
105 int maxHist = U.capacity1();
106 for (int m = maxHist-1; m > 0; --m) {
107 for (int n = maxHist-1; n > 0; --n) {
108 U(m,n) = U(m-1,n-1);
109 }
110 }
111
112 // Compute U matrix's new row 0 and col 0
113 for (int m = 0; m < nHist; ++m) {
114 double dotprod = computeUDotProd(resBasis,m,0);
115 U(m,0) = dotprod;
116 U(0,m) = dotprod;
117 }
118 }

184 template <int D>
185 void AmIterator<D>::getCurrent(FieldCUDA& curr)
186 {
187 const int nMonomer = system().mixture().nMonomer();
188 const int nMesh = system().mesh().size();
189 const int n = nElements();
190
191 // GPU resources
192 int nBlocks, nThreads;
193 ThreadGrid::setThreadsLogical(nMesh, nBlocks, nThreads);
194
195 // pointer to fields on system
196 DArray<RDField<D>> * const currSys = &system().wFieldsRGrid();
197
198 // loop to unfold the system fields and store them in one long array
199 for (int i = 0; i < nMonomer; i++) {
200 assignReal<<<nBlocks,nThreads>>>(curr.cDField() + i*nMesh, (*currSys)[i].cDField(), nMesh);
201 }
202
203 // if flexible unit cell, also store unit cell parameters
204 if (isFlexible_) {
205 const int nParam = system().unitCell().nParameter();
206 const FSArray<double,6> currParam = system().unitCell().parameters();
207 // convert into a cudaReal array
208 cudaReal* temp = new cudaReal[nParam];
209 for (int k = 0; k < nParam; k++)
210 temp[k] = (cudaReal)scaleStress_*currParam[k];
211
212 // copy paramters to the end of the curr array
213 cudaMemcpy(curr.cDField() + nMonomer*nMesh, temp, nParam*sizeof(cudaReal), cudaMemcpyHostToDevice);
214 delete[] temp;
215 }

229 template <int D>
230 void AmIterator<D>::getResidual(FieldCUDA& resid)
231 {
232 const int n = nElements();
233 const int nMonomer = system().mixture().nMonomer();
234 const int nMesh = system().mesh().size();
235
236 // GPU resources
237 int nBlocks, nThreads;
238 ThreadGrid::setThreadsLogical(nMesh, nBlocks, nThreads);
239
240 // Initialize residuals to zero. Kernel will take care of potential
241 // additional elements (n vs nMesh).
242 assignUniformReal<<<nBlocks, nThreads>>>(resid.cDField(), 0, n);
243
244 // Compute SCF residuals
245 for (int i = 0; i < nMonomer; i++) {
246 int startIdx = i*nMesh;
247 for (int j = 0; j < nMonomer; j++) {
248 pointWiseAddScale<<<nBlocks, nThreads>>>(resid.cDField() + startIdx,
249 system().cFieldRGrid(j).cDField(),
250 system().interaction().chi(i, j),
251 nMesh);
252 pointWiseAddScale<<<nBlocks, nThreads>>>(resid.cDField() + startIdx,
253 system().wFieldRGrid(j).cDField(),
254 -system().interaction().idemp(i, j),
255 nMesh);
256 }
257 }
258
259
260 // If not canonical, account for incompressibility.
261 if (!system().mixture().isCanonical()) {
262 cudaReal factor = 1/(cudaReal)system().interaction().sum_inv();
263 for (int i = 0; i < nMonomer; ++i) {
264 subtractUniform<<<nBlocks, nThreads>>>(resid.cDField() + i*nMesh,
265 factor, nMesh);
266 }
267 } else {
268 for (int i = 0; i < nMonomer; i++) {
269 // Find current average
270 cudaReal average = findAverage(resid.cDField()+i*nMesh, nMesh);
271 // subtract out average to set residual average to zero
272 subtractUniform<<<nBlocks, nThreads>>>(resid.cDField() + i*nMesh,
273 average, nMesh);
274 }
275 }
276
277 // If variable unit cell, compute stress residuals
278 if (isFlexible_) {
279 const int nParam = system().unitCell().nParameter();
280 cudaReal* stress = new cudaReal[nParam];
281
282 for (int i = 0; i < nParam; i++) {
283 stress[i] = (cudaReal)(-1*scaleStress_*system().mixture().stress(i));
284 }
285
286 cudaMemcpy(resid.cDField()+nMonomer*nMesh, stress, nParam*sizeof(cudaReal), cudaMemcpyHostToDevice);
287 }
288 }

290 template <int D>
291 void AmIterator<D>::update(FieldCUDA& newGuess)
292 {
293 const int nMonomer = system().mixture().nMonomer();
294 const int nMesh = system().mesh().size();
295
296 // GPU resources
297 int nBlocks, nThreads;
298 ThreadGrid::setThreadsLogical(nMesh, nBlocks, nThreads);
299
300 // pointer to field objects in system
301 DArray<RDField<D>> * sysfields = &system().wFieldsRGrid();
302
303 // Manually and explicitly set homogeneous components of field if canonical
304 if (system().mixture().isCanonical()) {
305 cudaReal average, wAverage, cAverage;
306 for (int i = 0; i < nMonomer; i++) {
307 // Find current spatial average
308 average = findAverage(newGuess.cDField() + i*nMesh, nMesh);
309
310 // Subtract average from field, setting average to zero
311 subtractUniform<<<nBlocks, nThreads>>>(newGuess.cDField() + i*nMesh, average, nMesh);
312
313 // Compute the new average omega value, add it to all elements
314 wAverage = 0;
315 for (int j = 0; j < nMonomer; j++) {
316 // Find average concentration
317 cAverage = findAverage(system().cFieldRGrid(j).cDField(), nMesh);
318 wAverage += system().interaction().chi(i,j) * cAverage;
319 }
320 addUniform<<<nBlocks, nThreads>>>(newGuess.cDField() + i*nMesh, wAverage, nMesh);
321 }
322 }
323
324 // copy over grid points
325 for (int i = 0; i < nMonomer; i++) {
326 assignReal<<<nBlocks, nThreads>>>
327 ((*sysfields)[i].cDField(), newGuess.cDField() + i*nMesh, nMesh);
328 }
329
330 // if flexible unit cell, update parameters well
331 if (isFlexible_) {
332 FSArray<double,6> parameters;
333 const int nParam = system().unitCell().nParameter();
334 cudaReal* temp = new cudaReal[nParam];
335
336 cudaMemcpy(temp, newGuess.cDField() + nMonomer*nMesh, nParam*sizeof(cudaReal), cudaMemcpyDeviceToHost);
337 for (int i = 0; i < nParam; i++) {
338 parameters.append(1/scaleStress_ * (double)temp[i]);
339 }
340
341 system().unitCell().setParameters(parameters);
342 system().mixture().setupUnitCell(system().unitCell(), system().wavelist());
343 system().wavelist().computedKSq(system().unitCell());
344
345 delete[] temp;
346 }
347
348 // TEMPORARY. PASS THE INPUTS THROUGH A BASIS FILTER.
349 system().fieldIo().convertRGridToBasis(system().wFieldsRGrid(), system().wFields());
350 system().fieldIo().convertRGridToBasis(system().cFieldsRGrid(), system().cFields());
351 system().fieldIo().convertBasisToRGrid(system().wFields(), system().wFieldsRGrid());
352 system().fieldIo().convertBasisToRGrid(system().cFields(), system().cFieldsRGrid());
353
354 }
