460 /*
461 * Reduce rational number to standard reduced form (private).
462 */
463 inline
464 void Rational::reduce ()
465 {
466 UTIL_CHECK(den_ != 0);
467 if (num_ == 0) {
468 den_ = 1;
469 } else {
470 if (den_ < 0) {
471 den_ *= -1;
472 num_ *= -1;
473 }
474 int c = gcd(num_, den_);
475 num_ /= c;
476 den_ /= c;
477 }
478 }
