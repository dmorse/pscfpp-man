72  void MuSweep::outputSummary(std::ostream& out, int i, double s)
73  {
74  int np = mixture().nPolymer();
75  if (homogeneousMode_ == -1) {
76  out << Dbl(s)
77  << Dbl(system().fHelmholtz(), 20, 10)
78  << Dbl(system().pressure(), 20, 10);
79  for (i = 0; i < np - 1; ++i) {
80  out << Dbl(mixture().polymer(i).phi(), 20, 10);
81  }
82  out << std::endl;
83  } else {
84  out << Dbl(s,10);
85  if (homogeneousMode_ == 0) {
86  double dF = system().fHelmholtz()
87  - system().homogeneous().fHelmholtz();
88  out << Dbl(dF, 20, 10);
89  for (int i = 0; i < np - 1; ++i) {
90  out << Dbl(mixture().polymer(i).phi(), 16);
91  }
92  } else {
93  double fEx = system().fHelmholtz()
94  - system().homogeneous().fHelmholtz();
95  double pEx = system().pressure()
96  - system().homogeneous().pressure();
97  double V = domain().volume()/mixture().vMonomer();
98  double fExV = fEx*V;
99  double pExV = pEx*V;
100  out << Dbl(fExV, 20, 10);
101  out << Dbl(pExV, 20, 10);
102  for (int i = 0; i < np; ++i) {
103  out << Dbl(mixture().polymer(i).mu(), 16);
104  }
105  for (int i = 0; i < np - 1; ++i) {
106  out << Dbl(system().homogeneous().phi(i), 16);
107  }
108  double dV;
109  for (int i = 0; i < np - 1; ++i) {
110  dV = mixture().polymer(i).phi()
111  - system().homogeneous().phi(i);
112  dV *= V;
113  out << Dbl(dV, 16);
114  }
115  }
116  out << std::endl;
117  }
118  }
