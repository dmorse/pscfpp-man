51 /*
52 * Initialization at beginning sweep. Set phi0 to current composition.
53 */
54 void CompositionSweep::setup()
55 {
56 Sweep::setup();
57 int np = mixture().nPolymer();
58 int ns = mixture().nSolvent();
59 int i, j;
60 for (i = 0; i < np; ++i) {
61 phi0_[i] = mixture().polymer(i).phi();
62 }
63 if (ns > 0) {
64 for (j = 0; j < ns; ++j) {
65 i = np + j;
66 phi0_[i] = mixture().solvent(j).phi();
67 }
68 }
69 }

71 /*
72 * Set state for specified value of s.
73 */
74 void CompositionSweep::setParameters(double s)
75 {
76 double phi;
77 int np = mixture().nPolymer();
78 int ns = mixture().nSolvent();
79 int i, j;
80 for (int i = 0; i < np; ++i) {
81 phi = phi0_[i] + s*dPhi_[i];
82 mixture().polymer(i).setPhi(phi);
83 }
84 if (ns > 0) {
85 for (j = 0; j < ns; ++j) {
86 i = j + np;
87 phi = phi0_[i] + s*dPhi_[i];
88 mixture().solvent(j).setPhi(phi);
89 }
90 }
91 }

93 void CompositionSweep::outputSummary(std::ostream& out)
94 {
95 double sNew = s(0);
96 int i = nAccept() - 1;
97 int np = mixture().nPolymer();
98 int ns = mixture().nSolvent();
99 if (homogeneousMode_ == -1) {
100 out << Int(i) << Dbl(sNew)
101 << Dbl(system().fHelmholtz(), 20, 10)
102 << Dbl(system().pressure(), 20, 10);
103 for (int j = 0; j < np; ++j) {
104 out << Dbl(mixture().polymer(j).phi(), 20, 10);
105 }
106 if (ns > 0) {
107 for (int j = 0; j < ns; ++j) {
108 out << Dbl(mixture().solvent(j).phi(), 20, 10);
109 }
110 }
111 out << std::endl;
112 } else {
113 out << Int(i) << Dbl(sNew, 10);
114 if (homogeneousMode_ == 0) {
115 double dF = system().fHelmholtz()
116 - system().homogeneous().fHelmholtz();
117 out << Dbl(dF, 20, 10);
118 for (int j = 0; j < np; ++j) {
119 out << Dbl(mixture().polymer(j).phi(), 16);
120 }
121 if (ns > 0) {
122 for (int j = 0; j < ns; ++j) {
123 out << Dbl(mixture().solvent(j).phi(), 16);
124 }
125 }
126 } else {
127 double fEx = system().fHelmholtz()
128 - system().homogeneous().fHelmholtz();
129 double pEx = system().pressure()
130 - system().homogeneous().pressure();
131 double V = domain().volume()/mixture().vMonomer();
132 double fExV = fEx*V;
133 double pExV = pEx*V;
134 out << Dbl(fExV, 20, 10);
135 out << Dbl(pExV, 20, 10);
136
137 // Output chemical potentials
138 for (int j = 0; j < np; ++j) {
139 out << Dbl(mixture().polymer(j).mu(), 16);
140 }
141 if (ns > 0) {
142 for (int j = 0; j < ns; ++j) {
143 out << Dbl(mixture().solvent(j).mu(), 16);
144 }
145 }
146
147 // Output volume fractions
148 for (int j = 0; j < np; ++j) {
149 out << Dbl(system().homogeneous().phi(j), 16);
150 }
151 if (ns > 0) {
152 for (int j = 0; j < ns; ++j) {
153 out << Dbl(mixture().solvent(j).phi(), 16);
154 }
155 }
156
157 double dV;
158 for (int j = 0; j < np; ++j) {
159 dV = mixture().polymer(j).phi()
160 - system().homogeneous().phi(j);
161 dV *= V;
162 out << Dbl(dV, 16);
163 }
164 if (ns > 0) {
165 for (int j = 0; j < ns; ++j) {
166 dV = mixture().solvent(j).phi();
167 dV *= V;
168 out << Dbl(dV, 16);
169 }
170 }
171 }
172 out << std::endl;
173 }
174 }
