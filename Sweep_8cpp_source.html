62  void Sweep::solve()
63  {
64 
65  int nm = mixture().nMonomer();
66  int nx = domain().nx();
67  UTIL_CHECK(nm > 0);
68  UTIL_CHECK(nx > 0);
69 
70  // Allocate memory for solutions
71  if (!wFields0_.isAllocated()) {
72  wFields0_.allocate(nm);
73  for (int i = 0; i < nm; ++i) {
74  wFields0_[i].allocate(nx);
75  }
76  }
77  if (!wFields1_.isAllocated()) {
78  wFields1_.allocate(nm);
79  for (int i = 0; i < nm; ++i) {
80  wFields1_[i].allocate(nx);
81  }
82  }
83 
84  // Compute and output ds
85  double ds = 1.0/double(ns_);
86  double ds0 = ds;
87  std::cout << std::endl;
88  std::cout << "ns = " << ns_ << std::endl;
89  std::cout << "ds = " << ds << std::endl;
90 
91  // Set Sweep object
92  setup();
93 
94  // Open summary file
95  std::ofstream outFile;
96  std::string fileName = baseFileName_;
97  fileName += "log";
98  fileMaster().openOutputFile(fileName, outFile);
99 
100  // Solve for initial state of sweep
101  double s = 0.0;
102  int i = 0;
103  int error;
104  std::cout << std::endl;
105  std::cout << "Begin s = " << s << std::endl;
106  bool isContinuation = false; // False on first step
107  error = system().iterator().solve(isContinuation);
108  if (error) {
109  UTIL_THROW("Failure to converge initial state of sweep");
110  } else {
111  assignFields(wFields0_, wFields());
112  if (homogeneousMode_ >= 0) {
113  comparison_.compute(homogeneousMode_);
114  }
115  fileName = baseFileName_;
116  fileName += toString(i);
117  outputSolution(fileName, s);
118  outputSummary(outFile, i, s);
119  }
120 
121  // Loop over states on path
122  bool finished = false; // Are we finished with the loop?
123  int nPrev = 0; // Number of previous solutions stored
124  double s1 = 0.0; // Value of s at previous solution (if any)
125  while (!finished) {
126  error = 1;
127  while (error) {
128 
129  std::cout << std::endl;
130  std::cout << "Attempt s = " << s + ds << std::endl;
131 
132  // Setup guess for fields
133  if (nPrev == 0) {
134  std::cout << "Zeroth order continuation" << std::endl;
135  } else {
136  // std::cout << "1st order continuation" << std::endl;
137  double f1 = ds/(s - s1);
138  double f0 = 1.0 + f1;
139  // std::cout << " ds = " << ds << std::endl;
140  // std::cout << " s = " << s << ", s1 = " << s1 << std::endl;
141  // std::cout << " f1 = " << f1 << ", f0 = " << f0 << std::endl;
142  int i, j;
143  for (i = 0; i < nm; ++i) {
144  for (j = 0; j < nx; ++j) {
145  wFields()[i][j] = f0*wFields0_[i][j] - f1*wFields1_[i][j];
146  }
147  }
148  }
149 
150  // Attempt solution
151  setState(s+ds);
152  isContinuation = true;
153  error = system().iterator().solve(isContinuation);
154 
155  if (error) {
156 
157  // Upon failure, reset to fields from last converged solution
158  assignFields(wFields(), wFields0_);
159 
160  // Decrease ds by half
161  ds *= 0.50;
162  if (ds < 0.2*ds0) {
163  UTIL_THROW("Step size too small in sweep");
164  }
165 
166  } else {
167 
168  // Upon success, save new field
169  s1 = s;
170  nPrev = 1;
171  assignFields(wFields1_, wFields0_);
172  assignFields(wFields0_, wFields());
173 
174  // Compare to homogeneous reference system
175  if (homogeneousMode_ >= 0) {
176  comparison_.compute(homogeneousMode_);
177  }
178 
179  // Update s and output
180  s += ds;
181  ++i;
182  fileName = baseFileName_;
183  fileName += toString(i);
184  outputSolution(fileName, s);
185  outputSummary(outFile, i, s);
186 
187  }
188  }
189  if (s + ds > 1.0000001) {
190  finished = true;
191  }
192  }
193  }

223  void Sweep::outputSummary(std::ostream& out, int i, double s)
224  {
225  if (homogeneousMode_ == -1) {
226  out << Int(i,5) << Dbl(s)
227  << Dbl(system().fHelmholtz(),16)
228  << Dbl(system().pressure(),16)
229  << std::endl;
230  } else {
231  out << Int(i,5) << Dbl(s)
232  << Dbl(system().fHelmholtz(),16)
233  << Dbl(system().pressure(),16);
234  if (homogeneousMode_ == 0) {
235  double dF = system().fHelmholtz()
236  - system().homogeneous().fHelmholtz();
237  out << Dbl(dF, 16);
238  } else {
239  double dP = system().pressure()
240  - system().homogeneous().pressure();
241  double dOmega = -1.0*dP*domain().volume();
242  out << Dbl(dOmega, 16);
243  }
244  out << std::endl;
245  }

248  void Sweep::assignFields(DArray<System::Field>& lhs,
249  DArray<System::Field> const & rhs) const
250  {
251 
252  int nm = mixture().nMonomer();
253  int nx = domain().nx();
254 
255  UTIL_CHECK(lhs.capacity() == nm);
256  UTIL_CHECK(rhs.capacity() == nm);
257  int i, j;
258  for (i = 0; i < nm; ++i) {
259  UTIL_CHECK(rhs[i].isAllocated());
260  UTIL_CHECK(rhs[i].capacity() == nx);
261  UTIL_CHECK(lhs[i].isAllocated());
262  UTIL_CHECK(lhs[i].capacity() == nx);
263  for (j = 0; j < nx; ++j) {
264  lhs[i][j] = rhs[i][j];
265  }
266  }
267  }
