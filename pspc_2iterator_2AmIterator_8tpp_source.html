115 template <int D>
116 void AmIterator<D>::updateU(DMatrix<double> & U, RingBuffer<FieldCPU> const & resBasis, int nHist)
117 {
118 // Update matrix U by shifting elements diagonally
119 int maxHist = U.capacity1();
120 for (int m = maxHist-1; m > 0; --m) {
121 for (int n = maxHist-1; n > 0; --n) {
122 U(m,n) = U(m-1,n-1);
123 }
124 }
125
126 // Compute U matrix's new row 0 and col 0
127 for (int m = 0; m < nHist; ++m) {
128 double dotprod = computeUDotProd(resBasis,0,m);
129 U(m,0) = dotprod;
130 U(0,m) = dotprod;
131 }
132 }

193 template <int D>
194 void AmIterator<D>::getCurrent(FieldCPU& curr)
195 {
196 // Straighten out fields into linear arrays
197
198 const int nMonomer = system().mixture().nMonomer();
199 const int nBasis = system().basis().nBasis();
200 const DArray<DArray<double>> * currSys = &system().wFields();
201
202
203 for (int i = 0; i < nMonomer; i++) {
204 for (int k = 0; k < nBasis; k++)
205 {
206 curr[i*nBasis+k] = (*currSys)[i][k];
207 }
208 }
209
210 if (isFlexible_) {
211 const int nParam = system().unitCell().nParameter();
212 const FSArray<double,6> currParam = system().unitCell().parameters();
213
214 for (int i = 0; i < nParam; i++) {
215 curr[nMonomer*nBasis + i] = scaleStress_*currParam[i];
216 }
217 }
218
219 return;

233 template <int D>
234 void AmIterator<D>::getResidual(FieldCPU& resid)
235 {
236 const int n = nElements();
237 const int nMonomer = system().mixture().nMonomer();
238 const int nBasis = system().basis().nBasis();
239
240 // Initialize residuals
241 for (int i = 0 ; i < n; ++i) {
242 resid[i] = 0.0;
243 }
244
245 // Compute SCF residual vector elements
246 for (int i = 0; i < nMonomer; ++i) {
247 for (int j = 0; j < nMonomer; ++j) {
248 for (int k = 0; k < nBasis; ++k) {
249 int idx = i*nBasis + k;
250 resid[idx] +=
251 system().interaction().chi(i,j)*system().cField(j)[k] -
252 system().interaction().idemp(i,j)*system().wField(j)[k];
253 }
254 }
255 }
256
257 // If not canonical, account for incompressibility
258 if (!system().mixture().isCanonical()) {
259 for (int i = 0; i < nMonomer; ++i) {
260 resid[i*nBasis] -= 1.0/system().interaction().sum_inv();
261 }
262 } else {
263 // otherwise explicitly set the residual value for the homogeneous components
264 // the homogeneous field component is set explicitly as well in update
265 for (int i = 0; i < nMonomer; ++i) {
266 resid[i*nBasis] = 0.0;
267 }
268 }
269
270 // If variable unit cell, compute stress residuals
271 if (isFlexible_) {
272 const int nParam = system().unitCell().nParameter();
273
274 // Combined -1 factor and stress scaling here. This is okay:
275 // - residuals only show up as dot products (U, v, norm)
276 // or with their absolute value taken (max), so the
277 // sign on a given residual vector element is not relevant
278 // as long as it is consistent across all vectors
279 // - The scaling is applied here and to the unit cell param
280 // storage, so that updating is done on the same scale,
281 // and then undone right before passing to the unit cell.
282
283 for (int i = 0; i < nParam ; i++) {
284 resid[nMonomer*nBasis + i] = scaleStress_ * -1
285 * system().mixture().stress(i);
286 }
287 }

291 template <int D>
292 void AmIterator<D>::update(FieldCPU& newGuess)
293 {
294 // Convert back to field format
295 const int nMonomer = system().mixture().nMonomer();
296 const int nBasis = system().basis().nBasis();
297
298 DArray< DArray<double> > wField;
299 wField.allocate(nMonomer);
300
301 // Restructure in format of monomers, basis functions
302 for (int i = 0; i < nMonomer; i++) {
303 wField[i].allocate(nBasis);
304 for (int k = 0; k < nBasis; k++)
305 {
306 wField[i][k] = newGuess[i*nBasis + k];
307 }
308 }
309 // Manually and explicitly set homogeneous components of field if canonical
310 if (system().mixture().isCanonical()) {
311 for (int i = 0; i < nMonomer; ++i) {
312 wField[i][0] = 0.0; // initialize to 0
313 for (int j = 0; j < nMonomer; ++j) {
314 wField[i][0] +=
315 system().interaction().chi(i,j) * system().cField(j)[0];
316 }
317 }
318 }
319 system().setWBasis(wField);
320
321 if (isFlexible_) {
322 FSArray<double, 6> parameters;
323 const int nParam = system().unitCell().nParameter();
324
325 for (int i = 0; i < nParam; i++) {
326 parameters.append(1/scaleStress_ * newGuess[nMonomer*nBasis + i]);
327 }
328
329 system().setUnitCell(parameters);
330 }
