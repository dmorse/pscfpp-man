70  void FieldIo<D>::readFieldsBasis(std::istream& in,
71  DArray< DArray<double> >& fields)
72  {
73  int nMonomer = fields.capacity();
74  UTIL_CHECK(nMonomer > 0);
75 
76  // Read header
77  FieldIo<D>::readFieldHeader(in);
78  std::string label;
79  in >> label;
80  UTIL_CHECK(label == "N_star");
81  int nStarIn;
82  in >> nStarIn;
83  UTIL_CHECK(nStarIn > 0);
84 
85  // Initialize all field array elements to zero
86  int i, j;
87  int nStarCapacity;
88  for (j = 0; j < nMonomer; ++j) {
89  nStarCapacity = fields[j].capacity();
90  UTIL_CHECK(nStarCapacity >= nStarIn);
91  for (i = 0; i < nStarCapacity; ++i) {
92  fields[j][i] = 0.0;
93  }
94  }
95 
96  DArray<double> temp;
97  temp.allocate(nMonomer);
98 
99  // Loop over stars to read field components
100  IntVec<D> waveIn, waveBz, waveDft;
101  int waveId, starId, nWaveVectors;
102  bool waveExists;
103  for (i = 0; i < nStarIn; ++i) {
104 
105  // Read components for different monomers
106  for (j = 0; j < nMonomer; ++j) {
107  in >> temp [j];
108  }
109 
110  // Read characteristic wave and number of wavectors in star.
111  in >> waveIn;
112  in >> nWaveVectors;
113 
114  // Check if waveIn is in first Brillouin zone (FBZ) for the mesh.
115  waveBz = shiftToMinimum(waveIn, mesh().dimensions(), unitCell());
116  waveExists = (waveIn == waveBz);
117 
118  // If wave is in FBZ, find in basis and set field components
119  if (waveExists) {
120  waveDft = waveBz;
121  mesh().shift(waveDft);
122  waveId = basis().waveId(waveDft);
123  starId = basis().wave(waveId).starId;
124  UTIL_CHECK(basis().star(starId).waveBz == waveBz);
125  if (!basis().star(starId).cancel) {
126  for (j = 0; j < nMonomer; ++j) {
127  fields[j][starId] = temp [j];
128  }
129  }
130  }

507  template <int D>
508  void FieldIo<D>::convertBasisToKGrid(DArray<double> const& in,
509  RFieldDft<D>& out)
510  {
511  // Create Mesh<D> with dimensions of DFT Fourier grid.
512  Mesh<D> dftMesh(out.dftDimensions());
513 
514  typename Basis<D>::Star const* starPtr; // pointer to current star
515  typename Basis<D>::Wave const* wavePtr; // pointer to current wave
516  std::complex<double> component; // coefficient for star
517  std::complex<double> coeff; // coefficient for wave
518  IntVec<D> indices; // dft grid indices of wave
519  int rank; // dft grid rank of wave
520  int is; // star index
521  int iw; // wave index
522 
523  // Initialize all dft coponents to zero
524  for (rank = 0; rank < dftMesh.size(); ++rank) {
525  out[rank][0] = 0.0;
526  out[rank][1] = 0.0;
527  }
528 
529  // Loop over stars, skipping cancelled stars
530  is = 0;
531  while (is < basis().nStar()) {
532  starPtr = &(basis().star(is));
533 
534  if (starPtr->cancel) {
535  ++is;
536  continue;
537  }
538 
539  if (starPtr->invertFlag == 0) {
540 
541  // Make complex coefficient for star basis function
542  component = std::complex<double>(in[is], 0.0);
543 
544  // Loop over waves in closed star
545  for (iw = starPtr->beginId; iw < starPtr->endId; ++iw) {
546  wavePtr = &basis().wave(iw);
547  if (!wavePtr->implicit) {
548  coeff = component*(wavePtr->coeff);
549  indices = wavePtr->indicesDft;
550  rank = dftMesh.rank(indices);
551  out[rank][0] = coeff.real();
552  out[rank][1] = coeff.imag();
553  }
554  }
555  ++is;
556 
557  } else
558  if (starPtr->invertFlag == 1) {
559 
560  // Make complex component for first star
561  component = std::complex<double>(in[is], -in[is+1]);
562  component /= sqrt(2.0);
563 
564  // Loop over waves in first star
565  starPtr = &(basis().star(is));
566  for (iw = starPtr->beginId; iw < starPtr->endId; ++iw) {
567  wavePtr = &basis().wave(iw);
568  if (!(wavePtr->implicit)) {
569  coeff = component*(wavePtr->coeff);
570  indices = wavePtr->indicesDft;
571  rank = dftMesh.rank(indices);
572  out[rank][0] = coeff.real();
573  out[rank][1] = coeff.imag();
574  }
575  }
576 
577  // Loop over waves in second star
578  starPtr = &(basis().star(is+1));
579  UTIL_CHECK(starPtr->invertFlag == -1);
580  component = conj(component);
581  for (iw = starPtr->beginId; iw < starPtr->endId; ++iw) {
582  wavePtr = &basis().wave(iw);
583  if (!(wavePtr->implicit)) {
584  coeff = component*(wavePtr->coeff);
585  indices = wavePtr->indicesDft;
586  rank = dftMesh.rank(indices);
587  out[rank][0] = coeff.real();
588  out[rank][1] = coeff.imag();
589  }
590  }
591 
592  // Increment is by 2 (two stars were processed)
593  is += 2;
594 
595  } else {
596 
597  UTIL_THROW("Invalid invertFlag value");
598 
599  }
600 
601  }
602 
603  }

605  template <int D>
606  void FieldIo<D>::convertKGridToBasis(RFieldDft<D> const& in,
607  DArray<double>& out)
608  {
609  // Create Mesh<D> with dimensions of DFT Fourier grid.
610  Mesh<D> dftMesh(in.dftDimensions());
611 
612  typename Basis<D>::Star const* starPtr; // pointer to current star
613  typename Basis<D>::Wave const* wavePtr; // pointer to current wave
614  std::complex<double> component; // coefficient for star
615  IntVec<D> indices; // dft grid indices of wave
616  int rank; // dft grid rank of wave
617  int is; // star index
618  int iw; // wave id, within star
619  bool isImplicit;
620 
621  // Initialize all components to zero
622  for (is = 0; is < basis().nStar(); ++is) {
623  out[is] = 0.0;
624  }
625 
626  // Loop over stars
627  is = 0;
628  while (is < basis().nStar()) {
629  starPtr = &(basis().star(is));
630 
631  if (starPtr->cancel) {
632  ++is;
633  continue;
634  }
635 
636  if (starPtr->invertFlag == 0) {
637 
638  // Choose a characteristic wave that is not implicit.
639  // Start with the first, alternately searching from
640  // the beginning and end of star.
641  isImplicit = true;
642  iw = 0;
643  while (isImplicit) {
644  UTIL_CHECK(iw <= (starPtr->size)/2);
645  wavePtr = &basis().wave(starPtr->beginId + iw);
646  if (wavePtr->implicit) {
647  wavePtr = &basis().wave(starPtr->endId - 1 - iw);
648  }
649  isImplicit = wavePtr->implicit;
650  ++iw;
651  }
652  UTIL_CHECK(wavePtr->starId == is);
653  indices = wavePtr->indicesDft;
654  rank = dftMesh.rank(indices);
655 
656  // Compute component value
657  component = std::complex<double>(in[rank][0], in[rank][1]);
658  component /= wavePtr->coeff;
659  UTIL_CHECK(abs(component.imag()) < 1.0E-8);
660  out[is] = component.real();
661  ++is;
662 
663  } else
664  if (starPtr->invertFlag == 1) {
665 
666  // Identify a characteristic wave that is not implicit:
667  // Either first wave of 1st star or last wave of 2nd star.
668  wavePtr = &basis().wave(starPtr->beginId);
669  if (wavePtr->implicit) {
670  starPtr = &(basis().star(is+1));
671  UTIL_CHECK(starPtr->invertFlag == -1);
672  wavePtr = &basis().wave(starPtr->endId-1);
673  UTIL_CHECK(!(wavePtr->implicit));
674  }
675  indices = wavePtr->indicesDft;
676  rank = dftMesh.rank(indices);
677 
678  // Compute component value
679  component = std::complex<double>(in[rank][0], in[rank][1]);
680  UTIL_CHECK(abs(wavePtr->coeff) > 1.0E-8);
681  component /= wavePtr->coeff;
682  component *= sqrt(2.0);
683  out[is] = component.real();
684  out[is+1] = -component.imag();
685 
686  is += 2;
687  } else {
688  UTIL_THROW("Invalid invertFlag value");
689  }
690 
691  } // loop over star index is
692  }
