83 /*
84 * Compute concentration, q, phi or mu.
85 */
86 template <int D>
87 void Solvent<D>::compute(WField const & wField)
88 {
89 int nx = meshPtr_->size(); // Number of grid points
90
91 // Initialize cField_ to zero
92 for (int i = 0; i < nx; ++i) {
93 cField_[i] = 0.0;
94 }
95
96 // Evaluate unnormalized integral and q_
97 double s = size();
98 q_ = 0.0;
99 for (int i = 0; i < nx; ++i) {
100 cField_[i] = exp(-s*wField[i]);
101 q_ += cField_[i];
102 }
103 q_ = q_/double(nx);
104
105 // Compute mu_ or phi_ and prefactor
106 double prefactor;
107 if (ensemble_ == Species::Closed) {
108 prefactor = phi_/q_;
109 mu_ = log(prefactor);
110 } else {
111 prefactor = exp(mu_);
112 phi_ = prefactor*q_;
113 }
114
115 // Normalize concentration
116 for (int i = 0; i < nx; ++i) {
117 cField_[i] *= prefactor;
118 }
119
120 }
121
122}
