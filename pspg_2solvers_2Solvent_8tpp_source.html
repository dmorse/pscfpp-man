83 /*
84 * Compute concentration, q, phi or mu.
85 */
86 template <int D>
87 void Solvent<D>::compute(WField const & wField)
88 {
89 int nx = meshPtr_->size(); // Number of grid points
90
91 // GPU Resources
92 int nBlocks, nThreads;
93 ThreadGrid::setThreadsLogical(nx, nBlocks, nThreads);
94
95 // Initialize concField_ to zero
96 assignUniformReal<<<nBlocks, nThreads>>>(concField_.cDField(), 0, nx);
97
98 // Evaluate unnormalized integral and q_
99 double s = size();
100 q_ = 0.0;
101 assignExp<<<nBlocks, nThreads>>>(concField_.cDField(), wField.cDField(), s, nx);
102 q_ = (double)gpuSum(concField_.cDField(),nx);
103 q_ = q_/double(nx);
104
105 // Compute mu_ or phi_ and prefactor
106 double prefactor;
107 if (ensemble_ == Species::Closed) {
108 prefactor = phi_/q_;
109 mu_ = log(prefactor);
110 } else {
111 prefactor = exp(mu_);
112 phi_ = prefactor*q_;
113 }
114
115 // Normalize concentration
116 scaleReal<<<nBlocks, nThreads>>>(concField_.cDField(), prefactor, nx);
117
118 }
