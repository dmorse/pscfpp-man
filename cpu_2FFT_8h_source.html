269 template <int D>
270 inline
271 bool FFT<D>::hasImplicitInverse(IntVec<D> const & wavevector,
272 IntVec<D> const & meshDimensions)
273 {
274 int i = wavevector[D-1];
275 int d = meshDimensions[D-1];
276 if ((i != 0) && (d - i > d/2 + 1)) {
277 return true;
278 } else {
279 return false;
280 }
281 }
