234  template <int D>
235  double Propagator<D>::computeQ()
236  {
237  // Preconditions
238  if (!isSolved()) {
239  UTIL_THROW("Propagator is not solved.");
240  }
241  if (!hasPartner()) {
242  UTIL_THROW("Propagator has no partner set.");
243  }
244  if (!partner().isSolved()) {
245  UTIL_THROW("Partner propagator is not solved");
246  }
247  const cudaReal * qh = head();
248  const cudaReal * qt = partner().tail();
249  int nx = meshPtr_->size();
250 
251  // Take inner product of head and partner tail fields
252  // cannot reduce assuming one propagator, qh == 1
253  // polymers are divided into blocks midway through
254  double Q = 0;
255 
256  Q = innerProduct(qh, qt, nx);
257  Q /= double(nx);
258  return Q;
259  }
