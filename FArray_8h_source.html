179 template <typename Data, int Capacity>
180 FArray<Data, Capacity>&
181 FArray<Data, Capacity>::operator=(FArray<Data, Capacity> const & other)
182 {
183
184 // Check for self assignment
185 if (this == &other) return *this;
186
187 // Copy elements

189 data_[i] = other[i];
190 }
