49  void NrIterator::allocate()
50  {
51  int nm = mixture().nMonomer(); // number of monomer types
52  int nx = domain().nx(); // number of grid points
53  UTIL_CHECK(nm > 0);
54  UTIL_CHECK(nx > 0);
55  int nr = nm*nx; // number of residual components
56  if (isAllocated_) {
57  UTIL_CHECK(cArray_.capacity() == nm);
58  UTIL_CHECK(residual_.capacity() == nr);
59  } else {
60  cArray_.allocate(nm);
61  wArray_.allocate(nm);
62  residual_.allocate(nr);
63  jacobian_.allocate(nr, nr);
64  residualNew_.allocate(nr);
65  dOmega_.allocate(nr);
66  wFieldsNew_.allocate(nm);
67  cFieldsNew_.allocate(nm);
68  for (int i = 0; i < nm; ++i) {
69  wFieldsNew_[i].allocate(nx);
70  cFieldsNew_[i].allocate(nx);
71  }
72  solver_.allocate(nr);
73  isAllocated_ = true;

77  void NrIterator::computeResidual(Array<WField> const & wFields,
78  Array<CField> const & cFields,
79  Array<double>& residual)
80  {
81  int nm = mixture().nMonomer(); // number of monomer types
82  int nx = domain().nx(); // number of grid points
83  int i; // grid point index
84  int j; // monomer indices
85  int ir; // residual index
86 
87  // Loop over grid points
88  for (i = 0; i < nx; ++i) {
89 
90  // Copy volume fractions at grid point i to cArray_
91  for (j = 0; j < nm; ++j) {
92  cArray_[j] = cFields[j][i];
93  }
94 
95  // Compute w fields, without Langrange multiplier, from c fields
96  system().interaction().computeW(cArray_, wArray_);
97 
98  // Initial residual = wPredicted(from above) - actual w
99  for (j = 0; j < nm; ++j) {
100  ir = j*nx + i;
101  residual[ir] = wArray_[j] - wFields[j][i];
102  }
103 
104  // Residuals j = 1, ..., nm-1 are differences from component j=0
105  for (j = 1; j < nm; ++j) {
106  ir = j*nx + i;
107  residual[ir] = residual[ir] - residual[i];
108  }
109 
110  // Residual for component j=0 then imposes incompressiblity
111  residual[i] = -1.0;
112  for (j = 0; j < nm; ++j) {
113  residual[i] += cArray_[j];
114  }
115  }
116 
117  /*
118  * Note: In canonical ensemble, the spatial integral of the incompressiblity
119  * residual is guaranteed to be zero, as a result of how volume fractions are
120  * computed in SCFT. One of the nx incompressibility constraints is thus
121  * redundant. To avoid this redundancy, replace the incompressibility residual
122  * at the last grid point by a residual that requires the w field for the last
123  * monomer type at the last grid point to equal zero.
124  */
125 
126  if (isCanonical_) {
127  residual[nx-1] = wFields[nm-1][nx-1];
128  }
129 
130  }

135  void NrIterator::computeJacobian()
136  {
137  // std::cout << "Begin computeJacobian ... ";
138  int nm = mixture().nMonomer(); // number of monomer types
139  int nx = domain().nx(); // number of grid points
140  int i; // monomer index
141  int j; // grid point index
142 
143  // Copy system().wFields to wFieldsNew.
144  for (i = 0; i < nm; ++i) {
145  for (j = 0; j < nx; ++j) {
146  UTIL_CHECK(nx == wFieldsNew_[i].capacity());
147  UTIL_CHECK(nx == system().wField(i).capacity());
148  wFieldsNew_[i][j] = system().wField(i)[j];
149  }
150  }
151 
152  // Compute jacobian, column by column
153  double delta = 0.001;
154  int nr = nm*nx; // number of residual elements
155  int jr; // jacobian row index
156  int jc = 0; // jacobian column index
157  for (i = 0; i < nm; ++i) {
158  for (j = 0; j < nx; ++j) {
159  wFieldsNew_[i][j] += delta;
160  mixture().compute(wFieldsNew_, cFieldsNew_);
161  computeResidual(wFieldsNew_, cFieldsNew_, residualNew_);
162  for (jr = 0; jr < nr; ++jr) {
163  jacobian_(jr, jc) =
164  (residualNew_[jr] - residual_[jr])/delta;
165  }
166  wFieldsNew_[i][j] = system().wField(i)[j];
167  ++jc;
168  }
169  }
170 
171  // Decompose Jacobian matrix
172  solver_.computeLU(jacobian_);
173 
174  // std::cout << "Finish computeJacobian" << std::endl;
175  }

177  void NrIterator::incrementWFields(Array<WField> const & wOld,
178  Array<double> const & dW,
179  Array<WField> & wNew)
180  {
181  int nm = mixture().nMonomer(); // number of monomers types
182  int nx = domain().nx(); // number of grid points
183  int i; // monomer index
184  int j; // grid point index
185  int k = 0; // residual element index
186 
187  // Add dW
188  for (i = 0; i < nm; ++i) {
189  for (j = 0; j < nx; ++j) {
190  wNew[i][j] = wOld[i][j] - dW[k];
191  ++k;
192  }
193  }
194 
195  // If canonical, shift such that last element is exactly zero
196  if (isCanonical_) {
197  double shift = wNew[nm-1][nx-1];
198  for (i = 0; i < nm; ++i) {
199  for (j = 0; j < nx; ++j) {
200  wNew[i][j] -= shift;
201  }
202  }
203  }

207  double NrIterator::residualNorm(Array<double> const & residual) const
208  {
209  int nm = mixture().nMonomer(); // number of monomer types
210  int nx = domain().nx(); // number of grid points
211  int nr = nm*nx; // number of residual components
212  double value, norm;
213  norm = 0.0;
214  for (int ir = 0; ir < nr; ++ir) {
215  value = fabs(residual[ir]);
216  if (value > norm) {
217  norm = value;
218  }
219  }
220  return norm;

223  int NrIterator::solve(bool isContinuation)
224  {
225  int nm = mixture().nMonomer(); // number of monomer types
226  int np = mixture().nPolymer(); // number of polymer species
227  int nx = domain().nx(); // number of grid points
228  int nr = nm*nx; // number of residual elements
229 
230  // Allocate memory if needed or, if allocated, check array sizes.
231  allocate();
232 
233  // Determine if isCanonical (iff all species ensembles are closed)
234  isCanonical_ = true;
235  Species::Ensemble ensemble;
236  for (int i = 0; i < np; ++i) {
237  ensemble = mixture().polymer(i).ensemble();
238  if (ensemble == Species::Unknown) {
239  UTIL_THROW("Unknown species ensemble");
240  }
241  if (ensemble == Species::Open) {
242  isCanonical_ = false;
243  }
244  }
245 
246  // If isCanonical, shift so that last element is zero.
247  // Note: This is one of the residuals in this case.
248  if (isCanonical_) {
249  double shift = wFields()[nm-1][nx-1];
250  int i, j;
251  for (i = 0; i < nm; ++i) {
252  for (j = 0; j < nx; ++j) {
253  wFields()[i][j] -= shift;
254  }
255  }
256  }
257 
258  // Compute initial residual vector and norm
259  mixture().compute(system().wFields(), system().cFields());
260  computeResidual(system().wFields(), system().cFields(), residual_);
261  double norm = residualNorm(residual_);
262 
263  // Set Jacobian status
264  newJacobian_ = false;
265  if (!isContinuation) {
266  needsJacobian_ = true;
267  }
268 
269  // Iterative loop
270  double normNew;
271  int i, j, k;
272  for (i = 0; i < 100; ++i) {
273  std::cout << "iteration " << i
274  << " , error = " << norm
275  << std::endl;
276 
277  if (norm < epsilon_) {
278  std::cout << "Converged" << std::endl;
279  system().computeFreeEnergy();
280  // Success
281  return 0;
282  }
283 
284  if (needsJacobian_) {
285  std::cout << "Computing jacobian" << std::endl;;
286  computeJacobian();
287  newJacobian_ = true;
288  needsJacobian_ = false;
289  }
290 
291  // Compute Newton-Raphson increment dOmega_
292  solver_.solve(residual_, dOmega_);
293 
294  // Try full Newton-Raphson update
295  incrementWFields(system().wFields(), dOmega_, wFieldsNew_);
296  mixture().compute(wFieldsNew_, cFieldsNew_);
297  computeResidual(wFieldsNew_, cFieldsNew_, residualNew_);
298  normNew = residualNorm(residualNew_);
299 
300  // Decrease increment if necessary
301  j = 0;
302  while (normNew > norm && j < 3) {
303  std::cout << " decreasing increment, error = "
304  << normNew << std::endl;
305  needsJacobian_ = true;
306  for (k = 0; k < nr; ++k) {
307  dOmega_[k] *= 0.66666666;
308  }
309  incrementWFields(system().wFields(), dOmega_, wFieldsNew_);
310  mixture().compute(wFieldsNew_, cFieldsNew_);
311  computeResidual(wFieldsNew_, cFieldsNew_, residualNew_);
312  normNew = residualNorm(residualNew_);
313  ++j;
314  }
315 
316  // If necessary, try reversing direction
317  if (normNew > norm) {
318  std::cout << " reversing increment, norm = "
319  << normNew << std::endl;
320  needsJacobian_ = true;
321  for (k = 0; k < nr; ++k) {
322  dOmega_[k] *= -1.000;
323  }
324  incrementWFields(system().wFields(), dOmega_, wFieldsNew_);
325  mixture().compute(wFieldsNew_, cFieldsNew_);
326  computeResidual(wFieldsNew_, cFieldsNew_, residualNew_);
327  normNew = residualNorm(residualNew_);
328  }
329 
330  // Accept or reject update
331  if (normNew < norm) {
332 
333  // Update system fields and residual vector
334  for (j = 0; j < nm; ++j) {
335  for (k = 0; k < nx; ++k) {
336  system().wField(j)[k] = wFieldsNew_[j][k];
337  system().cField(j)[k] = cFieldsNew_[j][k];
338  }
339  }
340  for (j = 0; j < nr; ++j) {
341  residual_[j] = residualNew_[j];
342  }
343  newJacobian_ = false;
344  if (!needsJacobian_) {
345  if (normNew/norm > 0.5) {
346  needsJacobian_ = true;
347  }
348  }
349  norm = normNew;
350  } else {
351  std::cout << "Iteration failed, norm = "
352  << normNew << std::endl;
353  if (newJacobian_) {
354  return 1;
355  std::cout << "Unrecoverable failure " << std::endl;
356  } else {
357  std::cout << "Try rebuilding Jacobian" << std::endl;
358  needsJacobian_ = true;
359  }
360  }
361 
362  }
363 
364  // Failure
365  return 1;
366  }
