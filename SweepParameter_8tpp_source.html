51 /*
52 * Read type, set nId and allocate id_ array.
53 */
54 template <int D>
55 void SweepParameter<D>::readParamType(std::istream& in)
56 {
57 std::string buffer;
58 in >> buffer;
59 std::transform(buffer.begin(), buffer.end(),
60 buffer.begin(), ::tolower);
61
62 if (buffer == "block" || buffer == "block_length") {
63 type_ = Block;
64 nID_ = 2; // polymer and block identifiers
65 } else if (buffer == "chi") {
66 type_ = Chi;
67 nID_ = 2; // two monomer type identifiers
68 } else if (buffer == "kuhn") {
69 type_ = Kuhn;
70 nID_ = 1; // monomer type identifier
71 } else if (buffer == "phi_polymer") {
72 type_ = Phi_Polymer;
73 nID_ = 1; //species identifier.
74 } else if (buffer == "phi_solvent") {
75 type_ = Phi_Solvent;
76 nID_ = 1; //species identifier.
77 } else if (buffer == "mu_polymer") {
78 type_ = Mu_Polymer;
79 nID_ = 1; //species identifier.
80 } else if (buffer == "mu_solvent") {
81 type_ = Mu_Solvent;
82 nID_ = 1; //species identifier.
83 } else if (buffer == "solvent" || buffer == "solvent_size") {
84 type_ = Solvent;
85 nID_ = 1; //species identifier.
86 } else {
87 UTIL_THROW("Invalid SweepParameter::ParamType value");
88 }
89
90 if (id_.isAllocated()) id_.deallocate();
91 id_.allocate(nID_);
92
93 }

149 template <int D>
150 double SweepParameter<D>::get_()
151 {
152 if (type_ == Block) {
153 return systemPtr_->mixture().polymer(id(0)).block(id(1)).length();
154 } else if (type_ == Chi) {
155 return systemPtr_->interaction().chi(id(0),id(1));
156 } else if (type_ == Kuhn) {
157 return systemPtr_->mixture().monomer(id(0)).kuhn();
158 } else if (type_ == Phi_Polymer) {
159 return systemPtr_->mixture().polymer(id(0)).phi();
160 } else if (type_ == Phi_Solvent) {
161 return systemPtr_->mixture().solvent(id(0)).phi();
162 } else if (type_ == Mu_Polymer) {
163 return systemPtr_->mixture().polymer(id(0)).mu();
164 } else if (type_ == Mu_Solvent) {
165 return systemPtr_->mixture().solvent(id(0)).mu();
166 } else if (type_ == Solvent) {
167 return systemPtr_->mixture().solvent(id(0)).size();
168 } else {
169 UTIL_THROW("This should never happen.");
170 }
171 }

173 template <int D>
174 void SweepParameter<D>::set_(double newVal)
175 {
176 if (type_ == Block) {
177 systemPtr_->mixture().polymer(id(0)).block(id(1)).setLength(newVal);
178 } else if (type_ == Chi) {
179 systemPtr_->interaction().setChi(id(0), id(1), newVal);
180 } else if (type_ == Kuhn) {
181 systemPtr_->mixture().setKuhn(id(0), newVal);
182 } else if (type_ == Phi_Polymer) {
183 systemPtr_->mixture().polymer(id(0)).setPhi(newVal);
184 } else if (type_ == Phi_Solvent) {
185 systemPtr_->mixture().solvent(id(0)).setPhi(newVal);
186 } else if (type_ == Mu_Polymer) {
187 systemPtr_->mixture().polymer(id(0)).setMu(newVal);
188 } else if (type_ == Mu_Solvent) {
189 systemPtr_->mixture().solvent(id(0)).setMu(newVal);
190 } else if (type_ == Solvent) {
191 systemPtr_->mixture().solvent(id(0)).setSize(newVal);
192 } else {
193 UTIL_THROW("This should never happen.");
194 }

220 SweepParameter<D>& param)
221 {
222 // Read the parameter type.
223 param.readParamType(in);
224 // Read the identifiers associated with this parameter type.
225 for (int i = 0; i < param.nID_; ++i) {
226 in >> param.id_[i];
227 }
228 // Read in the range in the parameter to sweep over
229 in >> param.change_;
230
231 return in;
