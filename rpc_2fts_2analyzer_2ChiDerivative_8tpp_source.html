41 template <int D>
42 double ChiDerivative<D>::compute()
43 {
44 UTIL_CHECK(system().w().hasData());
45
46 // For AB diblock
47 const int nMonomer = system().mixture().nMonomer();
48 UTIL_CHECK(nMonomer == 2);
49
50 const double vSystem = system().domain().unitCell().volume();
51 const double vMonomer = system().mixture().vMonomer();
52 const double nMonomerSystem = vSystem / vMonomer;
53 const int meshSize = system().domain().mesh().size();
54 double chi= system().interaction().chi(0,1);
55
56 /*
57 * Compute field Hamiltonian per monomer.
58 * The fieldHamitonian is calculated in the computeHamiltonian() function,
59 * located in rpc/fts/Simulator.tpp
60 */
61 if (!system().c().hasData()) {
62 system().compute();
63 }
64 if (!simulator().hasWc()){
65 simulator().computeWc();
66 }
67 if (!simulator().hasHamiltonian()) {
68 simulator().computeHamiltonian();
69 }
70 double hField = simulator().fieldHamiltonian()/nMonomerSystem;
71
72 // Compute derivative of free energy with repect to chi_bare per monomer
73 double dfdchi = -(hField - 0.5*simulator().sc(nMonomer - 1))/chi + 1.0/4.0;
74
75 dfdchi *= nMonomerSystem;
76
77 // With N term
78 dfdchi += double(meshSize)/(2.0 * chi);
79
80 return dfdchi;
81 }
