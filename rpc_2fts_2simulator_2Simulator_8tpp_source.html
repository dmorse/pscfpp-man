92 * Allocate required memory.
93 */
94 template <int D>
95 void Simulator<D>::allocate()
96 {
97 UTIL_CHECK(!isAllocated_);
98
99 const int nMonomer = system().mixture().nMonomer();
100
101 // Allocate projected chi matrix chiP_ and associated arrays
102 chiP_.allocate(nMonomer, nMonomer);
103 chiEvals_.allocate(nMonomer);
104 chiEvecs_.allocate(nMonomer, nMonomer);
105 sc_.allocate(nMonomer);
106
107 // Allocate memory for eignevector components of w and c fields
108 wc_.allocate(nMonomer);
109 cc_.allocate(nMonomer);
110 const IntVec<D> dimensions = system().domain().mesh().dimensions();
111 for (int i = 0; i < nMonomer; ++i) {
112 wc_[i].allocate(dimensions);
113 cc_[i].allocate(dimensions);
114 }
115
116 // Allocate memory for components of d (functional derivative)
117 dc_.allocate(nMonomer-1);
118 for (int i = 0; i < nMonomer - 1; ++i) {
119 dc_[i].allocate(dimensions);
120 }
121
122 // Allocate state_, if necessary.
123 if (!state_.isAllocated) {
124 const IntVec<D> dimensions = system().domain().mesh().dimensions();
125 state_.allocate(nMonomer, dimensions);
126 }
127
128 isAllocated_ = true;
129 }

181 /*
182 * Compute field theoretic Hamiltonian H[W].
183 */
184 template <int D>
185 void Simulator<D>::computeHamiltonian()
186 {
187 UTIL_CHECK(isAllocated_);
188 UTIL_CHECK(system().w().hasData());
189 UTIL_CHECK(system().hasCFields());
190 UTIL_CHECK(hasWc_);
191 hasHamiltonian_ = false;
192
193 Mixture<D> const & mixture = system().mixture();
194 Domain<D> const & domain = system().domain();
195
196 const int nMonomer = mixture.nMonomer();
197 const int meshSize = domain.mesh().size();
198
199 const int np = mixture.nPolymer();
200 const int ns = mixture.nSolvent();
201 double phi, mu;
202
203 // Compute polymer ideal gas contributions to lnQ
204 double lnQ = 0.0;
205 if (np > 0) {
206 Polymer<D> const * polymerPtr;
207 double length;
208 for (int i = 0; i < np; ++i) {
209 polymerPtr = &mixture.polymer(i);
210 phi = polymerPtr->phi();
211 mu = polymerPtr->mu();
212 if (PolymerModel::isThread()) {
213 length = polymerPtr->length();
214 } else {
215 length = (double)polymerPtr->nBead();
216 }
217 // Recall: mu = ln(phi/q)
218 if (phi > 1.0E-08) {
219 lnQ += phi*( -mu + 1.0 )/length;
220 }
221 }
222 }
223
224 // Compute solvent ideal gas contributions to lnQ
225 if (ns > 0) {
226 Solvent<D> const * solventPtr;
227 double size;
228 for (int i = 0; i < ns; ++i) {
229 solventPtr = &mixture.solvent(i);
230 phi = solventPtr->phi();
231 mu = solventPtr->mu();
232 size = solventPtr->size();
233 // Recall: mu = ln(phi/q)
234 if (phi > 1.0E-8) {
235 lnQ += phi*( -mu + 1.0 )/size;
236 }
237 }
238 }
239
240 // Subtract average of pressure field wc_[nMonomer-1]
241 RField<D> const & Wc = wc_[nMonomer-1];
242 for (int i = 0; i < meshSize; ++i) {
243 lnQ += Wc[i]/double(meshSize);
244 }
245 // lnQ now contains a value per monomer
246
247 // Initialize field contribution HW
248
249 // Compute quadratic field contribution to HW
250 double HW = 0.0;
251 double prefactor, w, s;
252 int i, j;
253 for (j = 0; j < nMonomer - 1; ++j) {
254 RField<D> const & Wc = wc_[j];
255 prefactor = -0.5*double(nMonomer)/chiEvals_[j];
256 s = sc_[j];
257 for (i = 0; i < meshSize; ++i) {
258 w = Wc[i] - s;
259 HW += prefactor*w*w;
260 }
261 }
262
263 // Normalize HW to equal a value per monomer
264 HW /= double(meshSize);
265
266 // Add constant term K/2 per monomer (K=s=e^{T}chi e/M^2)
267 HW += 0.5*sc_[nMonomer - 1];
268
269 // Compute number of monomers in the system (nMonomerSystem)
270 const double vSystem = domain.unitCell().volume();
271 const double vMonomer = mixture.vMonomer();
272 const double nMonomerSystem = vSystem / vMonomer;
273
274 // Compute final Hamiltonian components
275 fieldHamiltonian_ = nMonomerSystem * HW;
276 idealHamiltonian_ = -1.0 * nMonomerSystem * lnQ;
277 hamiltonian_ = idealHamiltonian_ + fieldHamiltonian_;
278
279 if (hasPerturbation()) {
280 perturbationHamiltonian_ = perturbation().hamiltonian(hamiltonian_);
281 hamiltonian_ += perturbationHamiltonian_;
282 } else {
283 perturbationHamiltonian_ = 0.0;
284 }
285
286 hasHamiltonian_ = true;
287 }

289 template <int D>
290 void Simulator<D>::analyzeChi()
291 {
292 UTIL_CHECK(isAllocated_);
293
294 const int nMonomer = system().mixture().nMonomer();
295 DMatrix<double> const & chi = system().interaction().chi();
296 double d = 1.0/double(nMonomer);
297 int i, j, k;
298
299 // Compute orthogonal projection matrix P
300 DMatrix<double> P;
301 P.allocate(nMonomer, nMonomer);
302 for (i = 0; i < nMonomer; ++i) {
303 for (j = 0; j < nMonomer; ++j) {
304 P(i,j) = -d;
305 }
306 P(i,i) += 1.0;
307 }
308
309 // Compute T = chi*P (temporary matrix)
310 DMatrix<double> T;
311 T.allocate(nMonomer, nMonomer);
312 for (i = 0; i < nMonomer; ++i) {
313 for (j = 0; j < nMonomer; ++j) {
314 T(i, j) = 0.0;
315 for (k = 0; k < nMonomer; ++k) {
316 T(i,j) += chi(i,k)*P(k,j);
317 }
318 }
319 }
320
321 // Compute chiP = = P*chi*P = P*T
322 DMatrix<double> chiP;
323 chiP.allocate(nMonomer, nMonomer);
324 for (i = 0; i < nMonomer; ++i) {
325 for (j = 0; j < nMonomer; ++j) {
326 chiP(i, j) = 0.0;
327 for (k = 0; k < nMonomer; ++k) {
328 chiP(i,j) += P(i,k)*T(k,j);
329 }
330 }
331 }
332
333 // Eigenvalue calculations use data structures and
334 // functions from the Gnu Scientific Library (GSL)
335
336 // Allocate GSL matrix A that will hold a copy of chiP
337 gsl_matrix* A = gsl_matrix_alloc(nMonomer, nMonomer);
338
339 // Copy DMatrix<double> chiP to gsl_matrix A
340 for (i = 0; i < nMonomer; ++i) {
341 for (j = 0; j < nMonomer; ++j) {
342 gsl_matrix_set(A, i, j, chiP(i, j));
343 }
344 }
345
346 // Compute eigenvalues and eigenvectors of chiP (or A)
347 gsl_eigen_symmv_workspace* work = gsl_eigen_symmv_alloc(nMonomer);
348 gsl_vector* Avals = gsl_vector_alloc(nMonomer);
349 gsl_matrix* Avecs = gsl_matrix_alloc(nMonomer, nMonomer);
350 int error;
351 error = gsl_eigen_symmv(A, Avals, Avecs, work);
352 UTIL_CHECK(error == 0);
353
354 // Requirements:
355 // - A has exactly one zero eigenvalue, with eigenvector (1,...,1)
356 // - All other eigenvalues must be negative.
357
358 // Copy eigenpairs with non-null eigenvalues
359 int iNull = -1; // index for null eigenvalue
360 int nNull = 0; // number of null eigenvalue
361 k = 0; // re-ordered index for non-null eigenvalue
362 double val;
363 for (i = 0; i < nMonomer; ++i) {
364 val = gsl_vector_get(Avals, i);
365 if (std::abs(val) < 1.0E-8) {
366 ++nNull;
367 iNull = i;
368 UTIL_CHECK(nNull <= 1);
369 } else {
370 chiEvals_[k] = val;
371 UTIL_CHECK(val < 0.0);
372 for (j = 0; j < nMonomer; ++j) {
373 chiEvecs_(k, j) = gsl_matrix_get(Avecs, j, i);
374 }
375 if (chiEvecs_(k, 0) < 0.0) {
376 for (j = 0; j < nMonomer; ++j) {
377 chiEvecs_(k, j) = -chiEvecs_(k, j);
378 }
379 }
380 ++k;
381 }
382 }
383 UTIL_CHECK(nNull == 1);
384 UTIL_CHECK(iNull >= 0);
385
386 // Set eigenpair with zero eigenvalue
387 i = nMonomer - 1;
388 chiEvals_[i] = 0.0;
389 for (j = 0; j < nMonomer; ++j) {
390 chiEvecs_(i, j) = gsl_matrix_get(Avecs, j, iNull);
391 }
392 if (chiEvecs_(i, 0) < 0) {
393 for (j = 0; j < nMonomer; ++j) {
394 chiEvecs_(i, j) = -chiEvecs_(i, j);
395 }
396 }
397
398 // Normalize all eigenvectors so that the sum of squares = nMonomer
399 double vec, norm, prefactor;
400 for (i = 0; i < nMonomer; ++i) {
401 norm = 0.0;
402 for (j = 0; j < nMonomer; ++j) {
403 vec = chiEvecs_(i, j);
404 norm += vec*vec;
405 }
406 prefactor = sqrt( double(nMonomer)/norm );
407 for (j = 0; j < nMonomer; ++j) {
408 chiEvecs_(i, j) *= prefactor;
409 }
410 }
411
412 // Check final eigenvector is (1, ..., 1)
413 for (j = 0; j < nMonomer; ++j) {
414 UTIL_CHECK(std::abs(chiEvecs_(nMonomer-1, j) - 1.0) < 1.0E-8);
415 }
416
417 // Compute vector s in monomer basis
418 DArray<double> s;
419 s.allocate(nMonomer);
420 for (i = 0; i < nMonomer; ++i) {
421 s[i] = 0.0;
422 for (j = 0; j < nMonomer; ++j) {
423 s[i] += chi(i,j);
424 }
425 s[i] = s[i]/double(nMonomer);
426 }
427
428 // Compute components of s in eigenvector basis -> sc_
429 for (i = 0; i < nMonomer; ++i) {
430 sc_[i] = 0.0;
431 for (j = 0; j < nMonomer; ++j) {
432 sc_[i] += chiEvecs_(i,j)*s[j];
433 }
434 sc_[i] = sc_[i]/double(nMonomer);
435 }
436
437 #if 0
438 // Debugging output
439 for (i = 0; i < nMonomer; ++i) {
440 Log::file() << "Eigenpair " << i << "\n";
441 Log::file() << "value = " << chiEvals_[i] << "\n";
442 Log::file() << "vector = [ ";
443 for (j = 0; j < nMonomer; ++j) {
444 Log::file() << chiEvecs_(i, j) << " ";
445 }
446 Log::file() << "]\n";
447 Log::file() << " sc[i] = " << sc_[i] << std::endl;
448 }
449 #endif
450
451 }

453 /*
454 * Compute the eigenvector components of the w fields, using the
455 * eigenvectors chiEvecs_ of the projected chi matrix as a basis.
456 */
457 template <int D>
458 void Simulator<D>::computeWc()
459 {
460 UTIL_CHECK(isAllocated_);
461
462 const int nMonomer = system().mixture().nMonomer();
463 const int meshSize = system().domain().mesh().size();
464 int i, j, k;
465
466 // Loop over eigenvectors (j is an eigenvector index)
467 for (j = 0; j < nMonomer; ++j) {
468
469 // Loop over grid points to zero out field wc_[j]
470 RField<D>& Wc = wc_[j];
471 for (i = 0; i < meshSize; ++i) {
472 Wc[i] = 0.0;
473 }
474
475 // Loop over monomer types (k is a monomer index)
476 for (k = 0; k < nMonomer; ++k) {
477 double vec = chiEvecs_(j, k)/double(nMonomer);
478
479 // Loop over grid points
480 RField<D> const & Wr = system().w().rgrid(k);
481 for (i = 0; i < meshSize; ++i) {
482 Wc[i] += vec*Wr[i];
483 }
484
485 }
486 }
487
488 hasWc_ = true;
489 }

491 /*
492 * Compute the eigenvector components of the c-fields, using the
493 * eigenvectors chiEvecs_ of the projected chi matrix as a basis.
494 */
495 template <int D>
496 void Simulator<D>::computeCc()
497 {
498 // Preconditions
499 UTIL_CHECK(isAllocated_);
500 UTIL_CHECK(system().w().hasData());
501 UTIL_CHECK(system().hasCFields());
502
503 const int nMonomer = system().mixture().nMonomer();
504 const int meshSize = system().domain().mesh().size();
505 int i, j, k;
506
507 // Loop over eigenvectors (i is an eigenvector index)
508 for (i = 0; i < nMonomer; ++i) {
509
510 // Set cc_[i] to zero
511 RField<D>& Cc = cc_[i];
512 for (k = 0; k < meshSize; ++k) {
513 Cc[k] = 0.0;
514 }
515
516 // Loop over monomer types
517 for (j = 0; j < nMonomer; ++j) {
518 RField<D> const & Cr = system().c().rgrid(j);
519 double vec = chiEvecs_(i, j);
520
521 // Loop over grid points
522 for (k = 0; k < meshSize; ++k) {
523 Cc[k] += vec*Cr[k];
524 }
525
526 }
527 }
528
529 hasCc_ = true;
530 }

532 /*
533 * Compute d fields, i.e., functional derivatives of H[W].
534 */
535 template <int D>
536 void Simulator<D>::computeDc()
537 {
538 // Preconditions
539 UTIL_CHECK(isAllocated_);
540 if (!hasWc_) computeWc();
541 if (!hasCc_) computeCc();
542
543 // Local constants and variables
544 const int meshSize = system().domain().mesh().size();
545 const int nMonomer = system().mixture().nMonomer();
546 const double vMonomer = system().mixture().vMonomer();
547 const double a = 1.0/vMonomer;
548 double b, s;
549 int i, k;
550
551 // Compute derivatives for standard Hamiltonian
552 // Loop over composition eigenvectors (exclude the last)
553 for (i = 0; i < nMonomer - 1; ++i) {
554 RField<D>& Dc = dc_[i];
555 RField<D> const & Wc = wc_[i];
556 RField<D> const & Cc = cc_[i];
557 b = -1.0*double(nMonomer)/chiEvals_[i];
558 s = sc_[i];
559 // Loop over grid points
560 for (k = 0; k < meshSize; ++k) {
561 Dc[k] = a*( b*(Wc[k] - s) + Cc[k] );
562 }
563 }
564
565 // Add derivatives arising from a perturbation (if any).
566 if (hasPerturbation()) {
567 perturbation().incrementDc(dc_);
568 }
569
570 hasDc_ = true;
571 }

573 /*
574 * Save the current state prior to a next move.
575 *
576 * Invoked before each move.
577 */
578 template <int D>
579 void Simulator<D>::saveState()
580 {
581 UTIL_CHECK(system().w().hasData());
582 UTIL_CHECK(hasWc());
583 UTIL_CHECK(state_.isAllocated);
584 UTIL_CHECK(!state_.hasData);
585
586 // Set fields
587 int nMonomer = system().mixture().nMonomer();
588
589 for (int i = 0; i < nMonomer; ++i) {
590 state_.w[i] = system().w().rgrid(i);
591 state_.wc[i] = wc(i);
592 }
593
594 // Save cc based on ccSavePolicy
595 if (state_.needsCc) {
596 UTIL_CHECK(hasCc());
597 UTIL_CHECK(state_.cc.isAllocated());
598 for (int i = 0; i < nMonomer; ++i) {
599 state_.cc[i] = cc(i);
600 }
601 }
602
603 // Save dc based on dcSavePolicy
604 if (state_.needsDc) {
605 UTIL_CHECK(hasDc());
606 UTIL_CHECK(state_.dc.isAllocated());
607 for (int i = 0; i < nMonomer - 1; ++i) {
608 state_.dc[i] = dc(i);
609 }
610 }
611
612 // Save Hamiltonian based on hamiltonianSavePolicy
613 if (state_.needsHamiltonian){
614 UTIL_CHECK(hasHamiltonian());
615 state_.hamiltonian = hamiltonian();
616 state_.idealHamiltonian = idealHamiltonian();
617 state_.fieldHamiltonian = fieldHamiltonian();
618 state_.perturbationHamiltonian = perturbationHamiltonian();
619 }
620
621 if (hasPerturbation()) {
622 perturbation().saveState();
623 }
624
625 state_.hasData = true;
626 }

628 /*
629 * Restore a saved fts state.
630 *
631 * Invoked after an attempted Monte-Carlo move is rejected
632 * or an fts move fails to converge
633 */
634 template <int D>
635 void Simulator<D>::restoreState()
636 {
637 UTIL_CHECK(state_.isAllocated);
638 UTIL_CHECK(state_.hasData);
639 const int nMonomer = system().mixture().nMonomer();
640
641 // Restore fields
642 system().setWRGrid(state_.w);
643
644 // Restore Hamiltonian and components
645 if (state_.needsHamiltonian){
646 hamiltonian_ = state_.hamiltonian;
647 idealHamiltonian_ = state_.idealHamiltonian;
648 fieldHamiltonian_ = state_.fieldHamiltonian;
649 perturbationHamiltonian_ = state_.perturbationHamiltonian;
650 hasHamiltonian_ = true;
651 }
652
653 for (int i = 0; i < nMonomer; ++i) {
654 wc_[i] = state_.wc[i];
655 }
656 hasWc_ = true;
657
658 if (state_.needsCc) {
659 for (int i = 0; i < nMonomer; ++i) {
660 cc_[i] = state_.cc[i];
661 }
662 hasCc_ = true;
663 }
664
665 if (state_.needsDc) {
666 for (int i = 0; i < nMonomer - 1; ++i) {
667 dc_[i] = state_.dc[i];
668 }
669 hasDc_ = true;
670 }
671
672 if (hasPerturbation()) {
673 perturbation().restoreState();
674 }
675
676 state_.hasData = false;
677 }
