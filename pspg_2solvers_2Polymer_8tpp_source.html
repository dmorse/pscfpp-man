80 // Initialize stress_ to 0

84
85 for (int i = 0; i < nBlock(); ++i) {
86 prefactor = exp(mu_)/length();
87 block(i).computeStress(wavelist, prefactor);
88
89 for (int j=0; j < nParams_; ++j){
90 stress_ [j] += block(i).stress(j);
