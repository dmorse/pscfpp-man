54 // Set or reset the state of existing Mesh<D>
55 template <int D>
56 void Mesh<D>::setDimensions(IntVec<D> const & dimensions)
57 {
58 int i;
59 for (i = 0; i < D; ++i) {
60 if (dimensions[i] <= 0) {
61 UTIL_THROW("Mesh dimensions must be positive");
62 }
63 }
64
65 dimensions_ = dimensions;
66 offsets_[D -1] = 1;
67 for (i = D - 1; i > 0; --i) {
68 offsets_[i-1] = offsets_[i]*dimensions_[i];
69 }
70 size_ = offsets_[0]*dimensions_[0];
71 }

74 template <int D>
75 int Mesh<D>::rank(IntVec<D> const & position) const
76 {
77 int i;
78 int result = 0;
79 for (i = 0; i < D - 1; ++i) {
80 assert(position[i] >= 0);
81 assert(position[i] < dimensions_[i]);
82 result += position[i]*offsets_[i];
83 }
84 assert(position[i] >= 0);
85 assert(position[i] < dimensions_[i]);
86 result += position[i];
87 return result;

93 {

97 int i;

134 int Mesh<D>::shift(int& coordinate, int i) const
135 {
136 int shift;
137 if (coordinate >= 0) {
138 shift = coordinate/dimensions_[i];
139 } else {
140 shift = -1 + ((coordinate+1)/dimensions_[i]);
141 }
