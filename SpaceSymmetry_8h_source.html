367 * Equality operator.
368 */
369 template <int D>
370 bool operator == (const SpaceSymmetry<D>& A, const SpaceSymmetry<D>& B)
371 {
372 int i, j;
373 for (i = 0; i < D; ++i) {
374 if (A.t_[i] != B.t_[i]) {
375 return false;
376 }
377 for (j = 0; j < D; ++j) {
378 if (A.R_(i, j) != B.R_(i,j)) {
379 return false;
380 }
381 }
382 }
383 return true;
384 }

386 /*
387 * Group multipication operator for SpaceSymmetry<D> objects.
388 */
389 template <int D>
390 SpaceSymmetry<D>
391 operator * (const SpaceSymmetry<D>& A, const SpaceSymmetry<D>& B)
392 {
393 SpaceSymmetry<D> C;
394 int i, j, k;
395
396 // Compute rotation matrix (matrix product)
397 for (i = 0; i < D; ++i) {
398 for (j = 0; j < D; ++j) {
399 C.R_(i, j) = 0;
400 for (k = 0; k < D; ++k) {
401 C.R_(i, j) += A.R_(i, k)*B.R_(k, j);
402 }
403 }
404 }
405
406 // Compute translation vector
407 for (i = 0; i < D; ++i) {
408 C.t_[i] = A.t_[i];
409 }
410 for (i = 0; i < D; ++i) {
411 for (j = 0; j < D; ++j) {
412 C.t_[i] += A.R_(i, j)*B.t_[j];
413 }
414 }
415 C.normalize();
416
417 return C;
418 }
