29  * Read chi matrix from file.
30  */
31  void ChiInteraction::readParameters(std::istream& in)
32  {
33  UTIL_CHECK(nMonomer() > 0);
34  chi_.allocate(nMonomer(), nMonomer());
35  chiInverse_.allocate(nMonomer(), nMonomer());
36  idemp_.allocate(nMonomer(), nMonomer());
37  readDSymmMatrix(in, "chi", chi_, nMonomer());
38 
39  if (nMonomer() == 2) {
40  double det = chi_(0,0)*chi_(1, 1) - chi_(0,1)*chi_(1,0);
41  double norm = chi_(0,0)*chi_(0, 0) + chi_(1,1)*chi_(1,1)
42  + 2.0*chi_(0,1)*chi_(1,0);
43  if (fabs(det/norm) < 1.0E-8) {
44  UTIL_THROW("Singular chi matrix");
45  }
46  chiInverse_(0,1) = -chi_(0,1)/det;
47  chiInverse_(1,0) = -chi_(1,0)/det;
48  chiInverse_(1,1) = chi_(0,0)/det;
49  chiInverse_(0,0) = chi_(1,1)/det;
50 
51  } else {
52  LuSolver solver;
53  solver.allocate(nMonomer());
54  solver.computeLU(chi_);
55  solver.inverse(chiInverse_);
56  }
57 
58  double sum = 0;
59  int i, j, k;
60 
61  for (i = 0; i < nMonomer(); ++i) {
62  idemp_(0,i) = 0;
63  for (j = 0; j < nMonomer(); ++j) {
64  idemp_(0,i) -= chiInverse_(j,i);
65  }
66  sum -= idemp_(0,i);
67  for (k = 0; k < nMonomer(); ++k) { //row
68  idemp_(k,i) = idemp_(0,i);
69  }
70  }
71 
72  for (i = 0; i < nMonomer(); ++i) { //row
73  for (j = 0; j < nMonomer(); ++j) { //coloumn
74  idemp_(i,j) /= sum;
75  }
76  idemp_(i,i) +=1 ;
77  }
78 
79  sum_inv_ = sum;

87  {

117  void
118  ChiInteraction::computeC(Array<double> const & w,
119  Array<double>& c, double& xi)
120  const
121  {
122  double sum1 = 0.0;
123  double sum2 = 0.0;
124  int i, j;
125  for (i = 0; i < nMonomer(); ++i) {
126  for (j = 0; j < nMonomer(); ++j) {
127  sum1 += chiInverse_(i, j)*w[j];
128  sum2 += chiInverse_(i, j);
129  }
130  }
131  xi = (sum1 - 1.0)/sum2;

133  c[i] = 0;
