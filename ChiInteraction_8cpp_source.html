44 void ChiInteraction::updateMembers()
45 {
46 UTIL_CHECK(chi_.isAllocated());
47 UTIL_CHECK(chiInverse_.isAllocated());
48 UTIL_CHECK(idemp_.isAllocated());
49
50 if (nMonomer() == 2) {
51 double det = chi_(0,0)*chi_(1, 1) - chi_(0,1)*chi_(1,0);
52 double norm = chi_(0,0)*chi_(0, 0) + chi_(1,1)*chi_(1,1)
53 + 2.0*chi_(0,1)*chi_(1,0);
54 if (fabs(det/norm) < 1.0E-8) {
55 UTIL_THROW("Singular chi matrix");
56 }
57 chiInverse_(0,1) = -chi_(0,1)/det;
58 chiInverse_(1,0) = -chi_(1,0)/det;
59 chiInverse_(1,1) = chi_(0,0)/det;
60 chiInverse_(0,0) = chi_(1,1)/det;
61
62 } else {
63 LuSolver solver;
64 solver.allocate(nMonomer());
65 solver.computeLU(chi_);
66 solver.inverse(chiInverse_);
67 }
68
69 double sum = 0;
70 int i, j, k;
71
72 for (i = 0; i < nMonomer(); ++i) {
73 idemp_(0,i) = 0;
74 for (j = 0; j < nMonomer(); ++j) {
75 idemp_(0,i) -= chiInverse_(j,i);
76 }
77 sum -= idemp_(0,i);
78 for (k = 0; k < nMonomer(); ++k) { //row
79 idemp_(k,i) = idemp_(0,i);
80 }
81 }
82
83 for (i = 0; i < nMonomer(); ++i) { //row
84 for (j = 0; j < nMonomer(); ++j) { //coloumn
85 idemp_(i,j) /= sum;
86 }
87 idemp_(i,i) += 1.0 ;
88 }
89
90 sum_inv_ = sum;
91
92 }

105 /*
106 * Compute and return excess Helmholtz free energy per monomer.
107 */
108 double ChiInteraction::fHelmholtz(Array<double> const & c) const
109 {
110 int i, j;
111 double sum = 0.0;
112 for (i = 0; i < nMonomer(); ++i) {
113 for (j = 0; j < nMonomer(); ++j) {
114 sum += chi_(i, j)* c[i]*c[j];
115 }
116 }
117 return 0.5*sum;
118 }

129 w[i] = 0.0;

131 w[i] += chi_(i, j)* c[j];

136 /*
137 * Compute concentrations and xi from chemical potentials.
138 */
139 void
140 ChiInteraction::computeC(Array<double> const & w,
141 Array<double>& c, double& xi)
142 const
143 {
144 double sum1 = 0.0;
145 double sum2 = 0.0;
146 int i, j;
147 for (i = 0; i < nMonomer(); ++i) {
148 for (j = 0; j < nMonomer(); ++j) {
149 sum1 += chiInverse_(i, j)*w[j];
150 sum2 += chiInverse_(i, j);
151 }
152 }
153 xi = (sum1 - 1.0)/sum2;
154 for (i = 0; i < nMonomer(); ++i) {
155 c[i] = 0;
156 for (j = 0; j < nMonomer(); ++j) {
157 c[i] += chiInverse_(i, j)*( w[j] - xi );
158 }
159 }
160 }
