56 /*
57 * Read iterator and wall data from parameter file
58 */
59 template <int D, typename IteratorType>
60 void FilmIteratorBase<D, IteratorType>::readParameters(std::istream& in)
61 {
62 // Make iterator_ a child paramComponent of this object, so that it
63 // will be read/written correctly to/from param file with correct
64 // indentation
65 setParent(iterator_,false);
66 addComponent(iterator_,false);
67
68 // Read the Iterator parameters
69 iterator_.readParameters(in);
70
71 // Then, read required data defining the walls
72 read(in, "normalVecId", normalVecId_);
73 read(in, "interfaceThickness", t_);
74 read(in, "wallThickness", T_);
75
76 // Make sure inputs are valid
77 if (normalVecId_ > D || normalVecId_ < 0) {
78 UTIL_THROW("bad value for normalVecId, must be in [0,D)");
79 }
80 if (t_ > T_) {
81 UTIL_THROW("wallThickness must be larger than interfaceThickness");
82 }
83 if ((T_ <= 0) || (t_ <= 0)) {
84 UTIL_THROW("wallThickness and interfaceThickness must be >0");
85 }
86
87 // Allocate chi_ and set it to zero before optionally reading it in
88 int nm = system().mixture().nMonomer();
89 chi_.allocate(nm,2);
90 for (int i = 0; i < nm; i++) {
91 for (int j = 0; j < 2; j++) {
92 chi_(i,j) = 0.0;
93 }
94 }
95 readOptionalDMatrix(in, "chi", chi_, nm, 2);
96
97 // Read optional filenames if user wants to write wall basis / rgrid
98 // files
99 readOptional(in, "wallBasisOutFile", wallBasisOutFile_);
100 readOptional(in, "wallRGridOutFile", wallRGridOutFile_);
101 }

105 * input is compatible with a film constraint, and create the mask /

114
115 // Ensure that the space group symmetry is compatible with the wall

128 writeWallFieldRGrid(wallRGridOutFile_);

131 // If lattice parameters are flexible, determine which parameters
132 // are allowed to vary, store them in this object, and pass them
133 // into iterator_.
134 if (isFlexible()) {
135 setFlexibleParams();
136 }

154 /*
155 * Generate the concentration field for the walls and pass it into
156 * iterator. Adjust phi values of species in the system to accommodate
157 * the volume occupied by the wall. Finally, generate the external
158 * fields that are created by the wall, if needed (if the wall is not
159 * athermal), and provide them to the iterator.
160 */
161 template <int D, typename IteratorType>
162 void FilmIteratorBase<D, IteratorType>::generateWallFields()
163 {
164 UTIL_CHECK(interfaceThickness() > 0);
165 UTIL_CHECK(wallThickness() > interfaceThickness());
166
167 // Ensure that unit cell is compatible with wall
168 checkLatticeVectors();
169
170 int nb = system().domain().basis().nBasis();
171
172 // Get the length L of the lattice basis vector normal to the walls
173 RealVec<D> a;
174 a = system().domain().unitCell().rBasis(normalVecId_);
175 double norm_sqd; // norm squared
176 for (int i = 0; i < D; i++) {
177 norm_sqd = a[i]*a[i];
178 }
179 double L(sqrt(norm_sqd));
180
181 // Create a 3 element vector 'dim' that contains the grid dimensions.
182 // If system is 2D (1D), then the z (y & z) dimensions are set to 1.
183 IntVec<3> dim;
184 for (int ind = 0; ind < 3; ind++) {
185 if (ind < D) {
186 dim[ind] = system().domain().mesh().dimensions()[ind];
187 } else {
188 dim[ind] = 1;
189 }
190 }
191
192 // Generate an r-grid representation of the walls
193 DArray< RField<D> > rGrid;
194 rGrid.allocate(1);
195 rGrid[0].allocate(system().domain().mesh().dimensions());
196 int x, y, z;
197 int counter = 0;
198 FArray<int,3> coords;
199 double d, rho;
200
201 for (x = 0; x < dim[0]; x++) {
202 coords[0] = x;
203 for (y = 0; y < dim[1]; y++) {
204 coords[1] = y;
205 for (z = 0; z < dim[2]; z++) {
206 coords[2] = z;
207
208 // Get the distance 'd' traveled along the lattice basis
209 // vector that is normal to the walls
210 d = coords[normalVecId_] * L / dim[normalVecId_];
211
212 // Calculate wall volume fraction (rho) at gridpoint (x,y,z)
213 rho = 0.5*(1+tanh(4*(((.5*(T_-L))+fabs(d-(L/2)))/t_)));
214 rGrid[0][counter++] = rho;
215 }
216 }
217 }
218
219 // Convert r-grid representation of walls to basis format, pass into
220 // iterator
221 DArray< DArray <double> > cField;
222 cField.allocate(1);
223 cField[0].allocate(nb);
224 system().domain().fieldIo().convertRGridToBasis(rGrid, cField);
225 iterator_.setMask(cField[0]);
226
227 // Store lattice parameters associated with this maskBasis
228 parameters_ = system().domain().unitCell().parameters();
229
230 // Adjust phi values for all species in the system to account for
231 // the volume that is occupied by the wall
232 adjustPhiVals();
233
234 // Generate external fields if needed
235 generateExternalFields();
236 }

238 /*
239 * Update the concentration field for the walls (the mask in the
240 * iterator) if the lattice parameters have been updated since mask was
241 * last generated.
242 *
243 * Update the external fields if the wall/polymer chi parameters have
244 * been updated since external fields were last generated.
245 */
246 template <int D, typename IteratorType>
247 void FilmIteratorBase<D, IteratorType>::updateWallFields()
248 {
249 // Ensure that the space group symmetry is compatible with the wall
250 checkSpaceGroup();
251
252 // Check if system lattice parameters are different than parameters_
253 FSArray<double, 6> sysParams =
254 system().domain().unitCell().parameters();
255 UTIL_CHECK(sysParams.size() == parameters_.size());
256 bool identical = true; // are the two arrays identical?
257 for (int i = 0; i < parameters_.size(); i++) {
258 if (fabs(sysParams[i] - parameters_[i]) > 1e-10) {
259 identical = false;
260 break;
261 }
262 }
263
264 // Regenerate wall fields if the lattice parameters have changed
265 if (!identical) {
266 generateWallFields();
267 } else {
268 // If we did not regenerate wall fields, check if we need to
269 // regenerate external field (if chi array has changed)
270 bool newExternalFields = false; // Do we need new external fields?
271 for (int i = 0; i < chi_.capacity1(); i++) {
272 if ((chi_(i,0) != chiFields_(i,0)) ||
273 (chi_(i,1) != chiFields_(i,1))) {
274 newExternalFields = true;
275 break;
276 }
277 }
278 if (newExternalFields) {
279 generateExternalFields();
280 }
281 }
282 }

284 /*
285 * Generate the external fields that are imposed as a result of chemical
286 * interactions between the walls and the monomer species.
287 */
288 template <int D, typename IteratorType>
289 void FilmIteratorBase<D, IteratorType>::generateExternalFields()
290 {
291 // Set chiFields_ equal to the current chi array associated with
292 // these fields
293 chiFields_ = chi_;
294
295 if (isAthermal()) return; // no external field if walls are athermal
296
297 // Otherwise, external field must be generated
298 hasExternalFields_ = true;
299
300 int nm = system().mixture().nMonomer();
301 int nb = system().domain().basis().nBasis();
302
303 DArray< DArray <double> > eField;
304 eField.allocate(nm);
305 for (int i = 0; i < nm; i++) {
306 eField[i].allocate(nb);
307 }
308
309 if (isSymmetric()) {
310 // If walls are identical, external field is just maskBasis*chi
311
312 for (int i = 0; i < nm; i++) {
313 UTIL_CHECK(fabs(chi(i,0)-chi(i,1) < 1e-7));
314 for (int j = 0; j < nb; j++) {
315 eField[i][j] = maskBasis()[j] * chi(i,0);
316 }
317 }
318
319 } else {
320 // Need to generate a whole new field, walls are not identical
321
322 // Get length L of the lattice basis vector normal to the walls
323 RealVec<D> a;
324 a = system().domain().unitCell().rBasis(normalVecId_);
325 double norm_sqd; // norm squared
326 for (int i = 0; i < D; i++) {
327 norm_sqd = a[i]*a[i];
328 }
329 double L(sqrt(norm_sqd));
330
331 // Create a 3 element vector 'dim' that contains the grid
332 // dimensions. If system is 2D (1D), then the z (y and z)
333 // dimensions are set to 1.
334 IntVec<3> dim;
335 for (int ind = 0; ind < 3; ind++) {
336 if (ind < D) {
337 dim[ind] = system().domain().mesh().dimensions()[ind];
338 } else {
339 dim[ind] = 1;
340 }
341 }
342
343 // Generate an r-grid representation of the external fields
344 DArray< RField<D> > rGrid;
345 rGrid.allocate(nm);
346 for (int i = 0; i < nm; i++) {
347 rGrid[i].allocate(system().domain().mesh().dimensions());
348 }
349
350 int i, x, y, z;
351 int counter = 0;
352 FArray<int,3> coords;
353 double d, rho;
354
355 for (i = 0; i < nm; i++) {
356 for (x = 0; x < dim[0]; x++) {
357 coords[0] = x;
358 for (y = 0; y < dim[1]; y++) {
359 coords[1] = y;
360 for (z = 0; z < dim[2]; z++) {
361 coords[2] = z;
362
363 // Get the distance 'd' traveled along the lattice
364 // basis vector that is orthogonal to the walls
365 d = coords[normalVecId_] * L / dim[normalVecId_];
366
367 // Calculate wall volume fraction (rho) at gridpoint
368 // (x,y,z)
369 rho = 0.5*(1+tanh(4*(((.5*(T_-L))+fabs(d-(L/2)))/t_)));
370 if (d < (L/2)) {
371 rGrid[i][counter++] = rho * chi(i,0);
372 } else {
373 rGrid[i][counter++] = rho * chi(i,1);
374 }
375 }
376 }
377 }
378 counter = 0;
379 }
380
381 // Convert r-grid representation of walls to basis format
382 system().domain().fieldIo().convertRGridToBasis(rGrid, eField);
383 }
384
385 // Pass eField into the iterator
386 iterator_.setExternalFields(eField);
387
388 // Store chi array associated with these externalFields
389 chiFields_ = chi();
390 }

392 template <int D, typename IteratorType>
393 void FilmIteratorBase<D, IteratorType>::adjustPhiVals()
394 {
395 int np = system().mixture().nPolymer();
396 int ns = system().mixture().nSolvent();
397 UTIL_CHECK(np + ns > 0);
398
399 // Get the sum of the current values of phi of each species
400 double frac = 0;
401 for (int i = 0; i < np; i++) {
402 frac += system().mixture().polymer(i).phi();
403 }
404 for (int j = 0; j < ns; j++) {
405 frac += system().mixture().solvent(j).phi();
406 }
407
408 // Adjust phi for each species based on phi_walls and frac, if needed
409 double multiplier = (1 - maskPhi()) / frac;
410 if (fabs(multiplier - 1) > 1e-7) { // if multiplier != 1
411 Log::file() << "NOTE: phi values of all species are being scaled"
412 << " by a factor of" << std::endl << Dbl(multiplier)
413 << ". This is to correct for the volume occupied"
414 << " by the wall." << std::endl;
415
416 double phi;
417 for (int i = 0; i < np; i++) {
418 phi = system().mixture().polymer(i).phi();
419 system().mixture().polymer(i).setPhi(phi * multiplier);
420 }
421 for (int j = 0; j < ns; j++) {
422 phi = system().mixture().solvent(j).phi();
423 system().mixture().solvent(j).setPhi(phi * multiplier);
424 }
425 }
426 }

428 /*
429 * Check that user-defined space group is compatible
430 * with the thin film constraint
431 */
432 template <int D, typename IteratorType>
433 void FilmIteratorBase<D, IteratorType>::checkSpaceGroup() const
434 {
435 // Setup
436 std::string groupName = system().groupName();
437 SpaceGroup<D> group;
438 std::string fileName = makeGroupFileName(D, groupName);
439 std::ifstream in;
440
441 // Open and read file containing space group's symmetry operations
442 in.open(fileName);
443 if (in.is_open()) {
444 in >> group;
445 UTIL_CHECK(group.isValid());
446 } else {
447 Log::file() << "\nFailed to open group file: "
448 << fileName << "\n";
449 Log::file() << "\n Error: Unknown space group\n";
450 UTIL_THROW("Unknown space group");
451 }
452
453 // Make sure all symmetry operations are allowed
454 int nv = normalVecId();
455 bool symmetric = isSymmetric();
456 std::string msg = "Space group contains forbidden symmetry operations";
457 for (int i = 0; i < group.size(); i++) {
458 for (int j = 0; j < D; j++) {
459 int r = group[i].R(nv,j);
460 if (j == nv) {
461 if (r != 1) {
462 if ((r != -1) || (!symmetric)) {
463 UTIL_THROW(msg.c_str());
464 }
465 }
466 } else { // j != nv
467 if (r != 0) {
468 UTIL_THROW(msg.c_str());
469 }
470 }
471 }
472 if (group[i].t(nv) != 0) {
473 UTIL_THROW(msg.c_str());
474 }
475 }
476 }

478 /*
479 * Check whether or not the two walls are chemically identical using
480 * the chi array.
481 */
482 template <int D, typename IteratorType>
483 bool FilmIteratorBase<D, IteratorType>::isSymmetric() const
484 {
485 int nm = system().mixture().nMonomer();
486
487 UTIL_CHECK(nm > 0);
488 UTIL_CHECK(chi().capacity1() == nm);
489 UTIL_CHECK(chi().capacity2() == 2);
490
491 for (int i = 0; i < nm; i++) {
492 if (fabs(chi(i,0)-chi(i,1)) > 1e-7) {
493 return false;
494 }
495 }
496 return true;
497 }

513 if ((chi(i,0) != 0) || (chi(i,1) != 0)) {
