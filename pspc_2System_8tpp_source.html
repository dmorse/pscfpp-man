120 // Read program arguments

122 opterr = 0;

124 switch (c) {
125 case 'e':
126 eflag = true;
127 break;
128 case 'p': // parameter file
129 pFlag = true;

131 break;
132 case 'c': // command file

135 break;
136 case 'i': // input prefix

138 iArg = optarg;
139 break;

143 break;
144 case '?':
145 Log::file() << "Unknown option -" << optopt << std::endl;

150 // Set flag to echo parameters as they are read.
151 if (eflag) {

180 template <int D>
181 void System<D>::readParameters(std::istream& in)
182 {
183 readParamComposite(in, mixture_);
184 hasMixture_ = true;
185
186 int nm = mixture_.nMonomer();
187 int np = mixture_.nPolymer();
188 int ns = mixture_.nSolvent();
189
190 // Initialize homogeneous object NOTE: THIS OBJECT IS NOT USED AT ALL.
191 homogeneous_.setNMolecule(np+ns);
192 homogeneous_.setNMonomer(nm);
193 initHomogeneous();
194
195 interaction().setNMonomer(mixture_.nMonomer());
196 readParamComposite(in, interaction());
197
198 readParamComposite(in, domain_);
199
200 mixture_.setMesh(mesh());
201 mixture_.setupUnitCell(unitCell());
202
203 allocate();
204
205 // Initialize iterator through the factory and mediator
206 std::string className;
207 bool isEnd;
208 iteratorPtr_ = iteratorFactoryPtr_->readObject(in, *this, className, isEnd);
209 if (!iteratorPtr_) {
210 std::string msg = "Unrecognized Iterator subclass name ";
211 msg += className;
212 UTIL_THROW(msg.c_str());
213 }
214 iterator().setup();
215
216 // Optionally instantiate a Sweep object
217 readOptional<bool>(in, "hasSweep", hasSweep_);
218 if (hasSweep_) {
219 bool isEnd;
220 sweepPtr_ =
221 sweepFactoryPtr_->readObject(in, *this, className, isEnd);
222 if (!sweepPtr_) {
223 UTIL_THROW("Unrecognized Sweep subclass name");
224 }
225 sweepPtr_->setSystem(*this);
226 }

294 /*
295 * Read and execute commands from a specified command file.
296 */
297 template <int D>
298 void System<D>::readCommands(std::istream &in)
299 {
300 UTIL_CHECK(isAllocated_);
301 std::string command, filename, inFileName, outFileName;
302
303 bool readNext = true;
304 while (readNext) {
305
306 in >> command;
307 Log::file() << command <<std::endl;
308
309 if (command == "FINISH") {
310 Log::file() << std::endl;
311 readNext = false;
312 } else
313 if (command == "READ_W_BASIS") {
314 readEcho(in, filename);
315 readWBasis(filename);
316 } else
317 if (command == "READ_W_RGRID") {
318 readEcho(in, filename);
319 readWRGrid(filename);
320 } else
321 if (command == "COMPUTE") {
322 // Read w (chemical potential fields) if not done previously
323 if (!hasWFields_) {
324 readEcho(in, filename);
325 readWBasis(filename);
326 }
327 // Solve the modified diffusion equation, without iteration
328 compute();
329 } else
330 if (command == "ITERATE") {
331 // Read w (chemical potential) fields if not done previously
332 if (!hasWFields_) {
333 readEcho(in, filename);
334 readWBasis(filename);
335 }
336 // Attempt iteration to convergence
337 int fail = iterate();
338 if (fail) {
339 readNext = false;
340 }
341 } else
342 if (command == "SWEEP") {
343 // Do a series of iterations.
344 sweep();
345 } else
346 if (command == "COMPARE_BASIS") {
347
348 // Get two filenames for comparison
349 std::string filecompare1, filecompare2;
350 DArray< DArray<double> > Bfield1, Bfield2;
351 readEcho(in, filecompare1);
352 readEcho(in, filecompare2);
353
354 // Store fields. Bfield1 and Bfield2 are unallocated, and thus will be
355 // allocated by the readFieldsBasis function.
356 fieldIo().readFieldsBasis(filecompare1, Bfield1, domain_.unitCell());
357 fieldIo().readFieldsBasis(filecompare2, Bfield2, domain_.unitCell());
358
359 // Compare and output
360 compare(Bfield1, Bfield2);
361
362 } else
363 if (command == "COMPARE_RGRID") {
364 // Get two filenames for comparison
365 std::string filecompare1, filecompare2;
366 DArray< RField<D> > Rfield1, Rfield2;
367 readEcho(in, filecompare1);
368 readEcho(in, filecompare2);
369
370 // Store fields. Rfield1 and Rfield2 are unallocated, but will be
371 // allocated by the readFieldsBasis function.
372 fieldIo().readFieldsRGrid(filecompare1, Rfield1, domain_.unitCell());
373 fieldIo().readFieldsRGrid(filecompare2, Rfield2, domain_.unitCell());
374
375 // Compare and output
376 compare(Rfield1, Rfield2);
377
378 } else
379 if (command == "WRITE_W_BASIS") {
380 readEcho(in, filename);
381 writeWBasis(filename);
382 } else
383 if (command == "WRITE_W_RGRID") {
384 readEcho(in, filename);
385 writeWRGrid(filename);
386 } else
387 if (command == "WRITE_C_BASIS") {
388 readEcho(in, filename);
389 writeCBasis(filename);
390 } else
391 if (command == "WRITE_C_RGRID") {
392 readEcho(in, filename);
393 writeCRGrid(filename);
394 } else
395 if (command == "WRITE_PROPAGATOR") {
396 int polymerID, blockID;
397 readEcho(in, filename);
398 in >> polymerID;
399 in >> blockID;
400 Log::file() << Str("polymer ID ", 21) << polymerID << "\n"
401 << Str("block ID ", 21) << blockID << std::endl;
402 writePropagatorRGrid(filename, polymerID, blockID);
403 } else
404 if (command == "BASIS_TO_RGRID") {
405 readEcho(in, inFileName);
406 readEcho(in, outFileName);
407 basisToRGrid(inFileName, outFileName);
408 } else
409 if (command == "RGRID_TO_BASIS") {
410 readEcho(in, inFileName);
411 readEcho(in, outFileName);
412 rGridToBasis(inFileName, outFileName);
413 } else
414 if (command == "KGRID_TO_RGRID") {
415 readEcho(in, inFileName);
416 readEcho(in, outFileName);
417 kGridToRGrid(inFileName, outFileName);
418 } else
419 if (command == "RGRID_TO_KGRID") {
420 readEcho(in, inFileName);
421 readEcho(in, outFileName);
422 rGridToKGrid(inFileName, outFileName);
423 } else
424 if (command == "CHECK_RGRID_SYMMETRY") {
425 readEcho(in, inFileName);
426 checkRGridFieldSymmetry(inFileName);
427 } else
428 if (command == "RHO_TO_OMEGA") {
429 readEcho(in, inFileName);
430 readEcho(in, outFileName);
431 rhoToOmega(inFileName, outFileName);
432 } else
433 if (command == "OUTPUT_STARS") {
434 readEcho(in, outFileName);
435 outputStars(outFileName);
436 } else
437 if (command == "OUTPUT_WAVES") {
438 readEcho(in, outFileName);
439 outputWaves(outFileName);
440 } else {
441 Log::file() << "Error: Unknown command "
442 << command << std::endl;
443 readNext = false;
444 }
445 }
446 }

460 /*
461 * Compute Helmoltz free energy and pressure
462 */
463 template <int D>
464 void System<D>::computeFreeEnergy()
465 {
466 UTIL_CHECK(hasWFields_);
467 UTIL_CHECK(hasCFields_);
468
469 // Initialize to zero
470 fHelmholtz_ = 0.0;
471
472 double phi, mu;
473 int np = mixture_.nPolymer();
474 int ns = mixture_.nSolvent();
475
476 // Compute polymer ideal gas contributions to fHelhmoltz_
477 if (np > 0) {
478 Polymer<D>* polymerPtr;
479 double length;
480 for (int i = 0; i < np; ++i) {
481 polymerPtr = &mixture_.polymer(i);
482 phi = polymerPtr->phi();
483 mu = polymerPtr->mu();
484 length = polymerPtr->length();
485 // Recall: mu = ln(phi/q)
486 fHelmholtz_ += phi*( mu - 1.0 )/length;
487 }
488 }
489
490 // Compute solvent ideal gas contributions to fHelhmoltz_
491 if (ns > 0) {
492 Solvent<D>* solventPtr;
493 double size;
494 for (int i = 0; i < ns; ++i) {
495 solventPtr = &mixture_.solvent(i);
496 phi = solventPtr->phi();
497 mu = solventPtr->mu();
498 size = solventPtr->size();
499 fHelmholtz_ += phi*( mu - 1.0 )/size;
500 }
501 }
502
503 int nm = mixture_.nMonomer();
504 int nBasis = basis().nBasis();
505
506 // Compute Legendre transform subtraction
507 // Use expansion in symmetry-adapted orthonormal basis
508 double temp = 0.0;
509 for (int i = 0; i < nm; ++i) {
510 for (int k = 0; k < nBasis; ++k) {
511 temp += wFields_[i][k] * cFields_[i][k];
512 }
513 }
514 fHelmholtz_ -= temp;
515
516 // Compute excess interaction free energy [ phi^{T}*chi*phi ]
517 double chi;
518 for (int i = 0; i < nm; ++i) {
519 for (int j = i + 1; j < nm; ++j) {
520 chi = interaction().chi(i,j);
521 for (int k = 0; k < nBasis; ++k) {
522 fHelmholtz_+= chi * cFields_[i][k] * cFields_[j][k];
523 }
524 }
525 }
526
527 // Initialize pressure
528 pressure_ = -fHelmholtz_;
529
530 // Polymer corrections to pressure
531 if (np > 0) {
532 Polymer<D>* polymerPtr;
533 double length;
534 for (int i = 0; i < np; ++i) {
535 polymerPtr = &mixture_.polymer(i);
536 phi = polymerPtr->phi();
537 mu = polymerPtr->mu();
538 length = polymerPtr->length();
539 pressure_ += mu * phi /length;
540 }
541 }
542
543 // Solvent corrections to pressure
544 if (ns > 0) {
545 Solvent<D>* solventPtr;
546 double size;
547 for (int i = 0; i < ns; ++i) {
548 solventPtr = &mixture_.solvent(i);
549 phi = solventPtr->phi();
550 mu = solventPtr->mu();
551 size = solventPtr->size();
552 pressure_ += mu * phi /size;
553 }
554 }
555
556 }

601 template <int D>
602 void System<D>::initHomogeneous()
603 {
604
605 // Set number of molecular species and monomers
606 int nm = mixture_.nMonomer();
607 int np = mixture_.nPolymer();
608 int ns = mixture_.nSolvent();
609 UTIL_CHECK(homogeneous_.nMolecule() == np + ns);
610 UTIL_CHECK(homogeneous_.nMonomer() == nm);
611
612 // Allocate c_ work array, if necessary
613 if (c_.isAllocated()) {
614 UTIL_CHECK(c_.capacity() == nm);
615 } else {
616 c_.allocate(nm);
617 }
618
619 int i; // molecule index
620 int j; // monomer index
621 int k; // block or clump index
622 int nb; // number of blocks
623 int nc; // number of clumps
624
625 // Loop over polymer molecule species
626 if (np > 0) {
627 for (i = 0; i < np; ++i) {
628
629 // Initial array of clump sizes
630 for (j = 0; j < nm; ++j) {
631 c_[j] = 0.0;
632 }
633
634 // Compute clump sizes for all monomer types.
635 nb = mixture_.polymer(i).nBlock();
636 for (k = 0; k < nb; ++k) {
637 Block<D>& block = mixture_.polymer(i).block(k);
638 j = block.monomerId();
639 c_[j] += block.length();
640 }
641
642 // Count the number of clumps of nonzero size
643 nc = 0;
644 for (j = 0; j < nm; ++j) {
645 if (c_[j] > 1.0E-8) {
646 ++nc;
647 }
648 }
649 homogeneous_.molecule(i).setNClump(nc);
650
651 // Set clump properties for this Homogeneous::Molecule
652 k = 0; // Clump index
653 for (j = 0; j < nm; ++j) {
654 if (c_[j] > 1.0E-8) {
655 homogeneous_.molecule(i).clump(k).setMonomerId(j);
656 homogeneous_.molecule(i).clump(k).setSize(c_[j]);
657 ++k;
658 }
659 }
660 homogeneous_.molecule(i).computeSize();
661
662 }
663 }
664
665 // Add solvent contributions
666 if (ns > 0) {
667 double size;
668 int monomerId;
669 for (int is = 0; is < ns; ++is) {
670 i = is + np;
671 monomerId = mixture_.solvent(is).monomerId();
672 size = mixture_.solvent(is).size();
673 homogeneous_.molecule(i).setNClump(1);
674 homogeneous_.molecule(i).clump(0).setMonomerId(monomerId);
675 homogeneous_.molecule(i).clump(0).setSize(size);
676 homogeneous_.molecule(i).computeSize();
677 }
678 }

710 /*
711 * Set new w-field values.
712 */
713 template <int D>
714 void System<D>::setWBasis(DArray< DArray<double> > const & fields)
715 {
716 // Update system wFields
717 int nMonomer = mixture_.nMonomer();
718 int nBasis = domain_.basis().nBasis();
719 for (int i = 0; i < nMonomer; ++i) {
720 DArray<double> const & f = fields[i];
721 DArray<double> & w = wFields_[i];
722 for (int j = 0; j < nBasis; ++j) {
723 w[j] = f[j];
724 }
725 }
726
727 // Update system wFieldsRgrid
728 domain_.fieldIo().convertBasisToRGrid(wFields_, wFieldsRGrid_);
729
730 hasWFields_ = true;
731 hasCFields_ = false;
732 }

734 /*
735 * Set new w-field values, using r-grid fields as inputs.
736 */
737 template <int D>
738 void System<D>::setWRGrid(DArray<WField> const & fields)
739 {
740 // Update system wFieldsRGrid
741 int nMonomer = mixture_.nMonomer();
742 int meshSize = domain_.mesh().size();
743 for (int i = 0; i < nMonomer; ++i) {
744 WField const & f = fields[i];
745 WField& w = wFieldsRGrid_[i];
746 for (int j = 0; j < meshSize; ++j) {
747 w[j] = f[j];
748 }
749 }
750
751 // Update system wFieldsRgrid
752 domain_.fieldIo().convertRGridToBasis(wFieldsRGrid_, wFields_);
753
754 hasWFields_ = true;
755 hasCFields_ = false;
756 }

758 /*
759 * Solve MDE for current w-fields, without iteration.
760 */
761 template <int D>
762 void System<D>::compute(bool needStress)
763 {
764 UTIL_CHECK(hasWFields_);
765
766 // Solve the modified diffusion equation (without iteration)
767 mixture_.compute(wFieldsRGrid(), cFieldsRGrid_);
768
769 // Convert c fields from r-grid to basis
770 fieldIo().convertRGridToBasis(cFieldsRGrid_, cFields_);
771 hasCFields_ = true;
772
773 if (needStress) {
774 mixture_.computeStress();
775 }
776 }

778 /*
779 * Iteratively solve a SCFT problem for specified parameters.
780 */
781 template <int D>
782 int System<D>::iterate()
783 {
784 UTIL_CHECK(hasWFields_);
785 hasCFields_ = false;
786
787 Log::file() << std::endl;
788 Log::file() << std::endl;
789
790 // Call iterator
791 int error = iterator().solve();
792
793 hasCFields_ = true;
794
795 if (!error) {
796 if (!iterator().isFlexible()) {
797 mixture().computeStress();
798 }
799 computeFreeEnergy();
800 outputThermo(Log::file());
801 }
802 return error;
803 }

983 /*
984 * Construct guess for omega (w-field) from rho (c-field).
985 *
986 * Modifies wFields and wFieldsRGrid and outputs wFields.
987 */
988 template <int D>
989 void System<D>::rhoToOmega(const std::string & inFileName,
990 const std::string & outFileName)
991 {
992 hasCFields_ = false;
993 hasWFields_ = false;
994
995 fieldIo().readFieldsBasis(inFileName, tmpFields_, domain_.unitCell());
996
997 // Compute w fields from c fields
998 for (int i = 0; i < basis().nBasis(); ++i) {
999 for (int j = 0; j < mixture_.nMonomer(); ++j) {
1000 wFields_[j][i] = 0.0;
1001 for (int k = 0; k < mixture_.nMonomer(); ++k) {
1002 wFields_[j][i] += interaction().chi(j,k) * tmpFields_[k][i];
1003 }
1004 }
1005 }
1006
1007 // Convert to r-grid format
1008 fieldIo().convertBasisToRGrid(wFields_, wFieldsRGrid_);
1009 hasWFields_ = true;
1010 hasCFields_ = false;
1011
1012 // Write w field in basis format
1013 fieldIo().writeFieldsBasis(outFileName, wFields(), unitCell());
1014 }
