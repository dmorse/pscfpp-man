50 template <int D>
51 void Block<D>::setDiscretization(double ds, const Mesh<D>& mesh)
52 {
53 // Preconditions
54 UTIL_CHECK(mesh.size() > 1);
55 UTIL_CHECK(ds > 0.0);
56 UTIL_CHECK(!isAllocated_);
57
58 // Set contour length discretization
59 int tempNs;
60 tempNs = floor( length()/(2.0 *ds) + 0.5 );
61 if (tempNs == 0) {
62 tempNs = 1;
63 }
64 ns_ = 2*tempNs + 1;
65 ds_ = length()/double(ns_-1);
66
67 // Set association to mesh
68 meshPtr_ = &mesh;
69
70 fft_.setup(mesh.dimensions());
71
72 // Compute Fourier space kMeshDimensions_
73 for (int i = 0; i < D; ++i) {
74 if (i < D - 1) {
75 kMeshDimensions_[i] = mesh.dimensions()[i];
76 } else {
77 kMeshDimensions_[i] = mesh.dimensions()[i]/2 + 1;
78 }
79 }
80
81 // Set number kSize of points in k-space mesh
82 int kSize = 1;
83 for (int i = 0; i < D; ++i) {
84 kSize *= kMeshDimensions_[i];
85 }
86
87 // Allocate work arrays for MDE solution
88 expKsq_.allocate(kMeshDimensions_);
89 expKsq2_.allocate(kMeshDimensions_);
90 expW_.allocate(mesh.dimensions());
91 expW2_.allocate(mesh.dimensions());
92 qr_.allocate(mesh.dimensions());
93 qk_.allocate(mesh.dimensions());
94 qr2_.allocate(mesh.dimensions());
95 qk2_.allocate(mesh.dimensions());
96 qf_.allocate(mesh.dimensions());
97
98 // Allocate work array for stress calculation
99 dGsq_.allocate(kSize, 6);
100
101 // Allocate block concentration field
102 cField().allocate(mesh.dimensions());
103
104 // Allocate memory for solutions to MDE (requires ns_)
105 propagator(0).allocate(ns_, mesh);
106 propagator(1).allocate(ns_, mesh);
107
108 isAllocated_ = true;
109 hasExpKsq_ = false;
110 }

146 template <int D>
147 void Block<D>::computeExpKsq()
148 {
149 UTIL_CHECK(isAllocated_);
150
151 MeshIterator<D> iter;
152 iter.setDimensions(kMeshDimensions_);
153 IntVec<D> G, Gmin;
154 double Gsq;
155 double factor = -1.0*kuhn()*kuhn()*ds_/6.0;
156 int i;
157 for (iter.begin(); !iter.atEnd(); ++iter) {
158 i = iter.rank();
159 G = iter.position();
160 Gmin = shiftToMinimum(G, mesh().dimensions(), unitCell());
161 Gsq = unitCell().ksq(Gmin);
162 expKsq_[i] = exp(Gsq*factor);
163 expKsq2_[i] = exp(Gsq*factor*0.5);
164 }
165
166 hasExpKsq_ = true;
167 }

172 template <int D>
173 void
174 Block<D>::setupSolver(RField<D> const& w)
175 {
176 // Preconditions
177 int nx = mesh().size();
178 UTIL_CHECK(nx > 0);
179 UTIL_CHECK(isAllocated_);
180
181 // Compute expW arrays
182 for (int i = 0; i < nx; ++i) {
183
184 // First, check that w[i]*ds_ is not unreasonably large:
185 // (if this condition is not met, solution will have large
186 // error, and user should consider using a smaller ds_)
187 //UTIL_CHECK(std::abs(w[i]*ds_) < 1.0);
188
189 // Calculate values
190 expW_[i] = exp(-0.5*w[i]*ds_);
191 expW2_[i] = exp(-0.5*0.5*w[i]*ds_);
192 }
193
194 // Compute expKsq arrays if necessary
195 if (!hasExpKsq_) {
196 computeExpKsq();
197 }
198
199 }

201 /*
202 * Integrate to calculate monomer concentration for this block
203 */
204 template <int D>
205 void Block<D>::computeConcentration(double prefactor)
206 {
207 // Preconditions
208 UTIL_CHECK(isAllocated_);
209 int nx = mesh().size();
210 UTIL_CHECK(nx > 0);
211 UTIL_CHECK(ns_ > 0);
212 UTIL_CHECK(ds_ > 0);
213 UTIL_CHECK(propagator(0).isAllocated());
214 UTIL_CHECK(propagator(1).isAllocated());
215 UTIL_CHECK(cField().capacity() == nx);
216
217 // Initialize cField to zero at all points
218 int i;
219 for (i = 0; i < nx; ++i) {
220 cField()[i] = 0.0;
221 }
222
223 Propagator<D> const & p0 = propagator(0);
224 Propagator<D> const & p1 = propagator(1);
225
226 // Evaluate unnormalized integral
227 for(i = 0; i < nx; ++i) {
228 cField()[i] += p0.q(0)[i]*p1.q(ns_ - 1)[i];
229 cField()[i] += p0.q(ns_ -1)[i]*p1.q(0)[i];
230 }
231
232 //odd indices
233 for(int j = 1; j < (ns_ -1); j += 2) {
234 for(int i = 0; i < nx; ++i) {
235 cField()[i] += p0.q(j)[i] * p1.q(ns_ - 1 - j)[i] * 4.0;
236 }
237 }
238
239 //even indices
240 for(int j = 2; j < (ns_ -2); j += 2) {
241 for(int i = 0; i < nx; ++i) {
242 cField()[i] += p0.q(j)[i] * p1.q(ns_ - 1 - j)[i] * 2.0;
243 }
244 }
245
246 prefactor *= ds_ / 3.0;
247 for (i = 0; i < nx; ++i) {
248 cField()[i] *= prefactor;
249 }
250
251 }

253 /*
254 * Integrate to Stress exerted by the chain for this block
255 */
256 template <int D>
257 void Block<D>::computeStress(double prefactor)
258 {
259 // Preconditions
260 int nx = mesh().size();
261 UTIL_CHECK(nx > 0);
262 UTIL_CHECK(ns_ > 0);
263 UTIL_CHECK(ds_ > 0);
264 UTIL_CHECK(propagator(0).isAllocated());
265 UTIL_CHECK(propagator(1).isAllocated());
266
267 stress_.clear();
268
269 double dels, normal, increment;
270 int nParam, c, m;
271
272 normal = 3.0*6.0;
273
274 int kSize_ = 1;
275 for (int i = 0; i < D; ++i) {
276 kSize_ *= kMeshDimensions_[i];
277 }
278 nParam = unitCell().nParameter();
279 c = kSize_;
280
281 FSArray<double, 6> dQ;
282
283 // Initialize work array and stress_ to zero at all points
284 int i;
285 for (i = 0; i < nParam; ++i) {
286 dQ.append(0.0);
287 stress_.append(0.0);
288 }
289
290 computedGsq();
291
292 Propagator<D> const & p0 = propagator(0);
293 Propagator<D> const & p1 = propagator(1);
294
295 // Evaluate unnormalized integral
296 for (int j = 0; j < ns_ ; ++j) {
297
298 qr_ = p0.q(j);
299 fft_.forwardTransform(qr_, qk_);
300
301 qr2_ = p1.q(ns_ - 1 - j);
302 fft_.forwardTransform(qr2_, qk2_);
303
304 dels = ds_;
305
306 if (j != 0 && j != ns_ - 1) {
307 if (j % 2 == 0) {
308 dels = dels*2.0;
309 } else {
310 dels = dels*4.0;
311 }
312 }
313
314 for (int n = 0; n < nParam ; ++n) {
315 increment = 0.0;
316
317 for (m = 0; m < c ; ++m) {
318 double prod = 0;
319 prod = (qk2_[m][0] * qk_[m][0]) + (qk2_[m][1] * qk_[m][1]);
320 prod *= dGsq_(m,n);
321 increment += prod;
322 }
323 increment = (increment * kuhn() * kuhn() * dels)/normal;
324 dQ[n] = dQ[n] - increment;
325 }
326 }
327
328 // Normalize
329 for (i = 0; i < nParam; ++i) {
330 stress_[i] = stress_[i] - (dQ[i] * prefactor);
331 }
332
333 }

335 /*
336 * Compute dGsq_ array (derivatives of Gsq for all wavevectors)
337 */
338 template <int D>
339 void Block<D>::computedGsq()
340 {
341 IntVec<D> temp;
342 IntVec<D> vec;
343 IntVec<D> Partner;
344 MeshIterator<D> iter;
345 iter.setDimensions(kMeshDimensions_);
346
347 for (int n = 0; n < unitCell().nParameter() ; ++n) {
348 for (iter.begin(); !iter.atEnd(); ++iter) {
349 temp = iter.position();
350 vec = shiftToMinimum(temp, mesh().dimensions(), unitCell());
351 dGsq_(iter.rank(), n) = unitCell().dksq(vec, n);
352 for (int p = 0; p < D; ++p) {
353 if (temp [p] != 0) {
354 Partner[p] = mesh().dimensions()[p] - temp[p];
355 } else {
356 Partner[p] = 0;
357 }
358 }
359 if (Partner[D-1] > kMeshDimensions_[D-1]) {
360 dGsq_(iter.rank(), n) *= 2;
361 }
362 }
363 }
364 }

367 * Propagate solution by one step.
368 */
369 template <int D>
370 void Block<D>::step(RField<D> const & q, RField<D>& qNew)
371 {
372 // Preconditions
373 UTIL_CHECK(isAllocated_);
374 UTIL_CHECK(hasExpKsq_);
375
376 // Check real space mesh sizes
377 int nx = mesh().size();
378 UTIL_CHECK(nx > 0);
379 UTIL_CHECK(q.isAllocated());
380 UTIL_CHECK(q.capacity() == nx);
381 UTIL_CHECK(qNew.isAllocated());
382 UTIL_CHECK(qNew.capacity() == nx);
383 UTIL_CHECK(qr_.capacity() == nx);
384 UTIL_CHECK(expW_.capacity() == nx);
385
386 // Check Fourier-space mesh sizes
387 int nk = qk_.capacity();
388 UTIL_CHECK(expKsq_.capacity() == nk);
389
390 // Apply pseudo-spectral algorithm
391 int i;
392 for (i = 0; i < nx; ++i) {
393 qr_[i] = q[i]*expW_[i];
394 qr2_[i] = q[i]*expW2_[i];
395 }
396
397 fft_.forwardTransform(qr_, qk_);
398 fft_.forwardTransform(qr2_, qk2_);
399 for (i = 0; i < nk; ++i) {
400 qk_[i][0] *= expKsq_[i];
401 qk_[i][1] *= expKsq_[i];
402 qk2_[i][0] *= expKsq2_[i];
403 qk2_[i][1] *= expKsq2_[i];
404 }
405 fft_.inverseTransform(qk_, qr_);
406 fft_.inverseTransform(qk2_, qr2_);
407 for (i = 0; i < nx; ++i) {
408 qf_[i] = qr_[i]*expW_[i];
409 qr2_[i] = qr2_[i]*expW_[i];
410 }
411
412 fft_.forwardTransform(qr2_, qk2_);
413 for (i = 0; i < nk; ++i) {
414 qk2_[i][0] *= expKsq2_[i];
415 qk2_[i][1] *= expKsq2_[i];
416 }
417 fft_.inverseTransform(qk2_, qr2_);
418 for (i = 0; i < nx; ++i) {
419 qr2_[i] = qr2_[i]*expW2_[i];
420 }
421 for (i = 0; i < nx; ++i) {
422 qNew[i] = (4.0*qr2_[i] - qf_[i])/3.0;
423 }
424 }
425
426}
