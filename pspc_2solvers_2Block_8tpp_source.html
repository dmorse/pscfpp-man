196  // Preconditions

198  UTIL_CHECK(nx > 0);
199  UTIL_CHECK(ns_ > 0);
200  UTIL_CHECK(ds_ > 0);
201  UTIL_CHECK(propagator(0).isAllocated());
202  UTIL_CHECK(propagator(1).isAllocated());
203  UTIL_CHECK(cField().capacity() == nx)
204 
205  // Initialize cField to zero at all points
206  int i;
207  for (i = 0; i < nx; ++i) {
208  cField()[i] = 0.0;
209  }
210 
211  Propagator<D> const & p0 = propagator(0);
212  Propagator<D> const & p1 = propagator(1);
213 
214  // Evaluate unnormalized integral
215  for(i = 0; i < nx; ++i) {
216  cField()[i] += p0.q(0)[i]*p1.q(ns_ - 1)[i];
217  cField()[i] += p0.q(ns_ -1)[i]*p1.q(0)[i];
218  }
219 
220  //odd indices
221  for(int j = 1; j < (ns_ -1); j += 2) {
222  for(int i = 0; i < nx; ++i) {
223  cField()[i] += p0.q(j)[i] * p1.q(ns_ - 1 - j)[i] * 4.0;
224  }
225  }
226 
227  //even indices
228  for(int j = 2; j < (ns_ -2); j += 2) {
229  for(int i = 0; i < nx; ++i) {
230  cField()[i] += p0.q(j)[i] * p1.q(ns_ - 1 - j)[i] * 2.0;
231  }
232  }
233 
234  prefactor *= ds_ / 3.0;
235  for (i = 0; i < nx; ++i) {
236  cField()[i] *= prefactor;

244  template <int D>
245  void Block<D>::computeStress(double prefactor)
246  {
247  // Preconditions
248  int nx = mesh().size();
249  UTIL_CHECK(nx > 0);
250  UTIL_CHECK(ns_ > 0);
251  UTIL_CHECK(ds_ > 0);
252  UTIL_CHECK(propagator(0).isAllocated());
253  UTIL_CHECK(propagator(1).isAllocated());
254 
255  stress_.clear();
256 
257  double dels, normal, increment;
258  int r, c, m;
259 
260  normal = 3.0*6.0;
261 
262  int kSize_ = 1;
263  for (int i = 0; i < D; ++i) {
264  kSize_ *= kMeshDimensions_[i];
265  }
266  r = unitCellPtr_->nParameter();
267  c = kSize_;
268 
269  FSArray<double, 6> dQ;
270 
271  // Initialize work array and stress_ to zero at all points
272  int i;
273  for (i = 0; i < r; ++i) {
274  dQ.append(0.0);
275  stress_.append(0.0);
276  }
277 
278  computedGsq();
279 
280  Propagator<D> const & p0 = propagator(0);
281  Propagator<D> const & p1 = propagator(1);
282 
283  // Evaluate unnormalized integral
284  for (int j = 0; j < ns_ ; ++j) {
285 
286  qr_ = p0.q(j);
287  fft_.forwardTransform(qr_, qk_);
288 
289  qr2_ = p1.q(ns_ - 1 - j);
290  fft_.forwardTransform(qr2_, qk2_);
291 
292  dels = ds_;
293 
294  if (j != 0 && j != ns_ - 1) {
295  if (j % 2 == 0) {
296  dels = dels*2.0;
297  } else {
298  dels = dels*4.0;
299  }
300  }
301 
302  for (int n = 0; n < r ; ++n) {
303  increment = 0;
304 
305  for (m = 0; m < c ; ++m) {
306  double prod = 0;
307  prod = (qk2_[m][0] * qk_[m][0]) + (qk2_[m][1] * qk_[m][1]);
308  prod *= dGsq_(m,n);
309  increment += prod;
310  }
311  increment = (increment * kuhn() * kuhn() * dels)/normal;
312  dQ [n] = dQ[n]-increment;
313  }
314  }
315 
316  // Normalize
317  for (i = 0; i < r; ++i) {
318  stress_[i] = stress_[i] - (dQ[i] * prefactor);
319  }
320 
321  }

327  void Block<D>::computedGsq()

330  IntVec<D> vec;

335  for (int n = 0; n < unitCellPtr_->nParameter() ; ++n) {
336  for (iter.begin(); !iter.atEnd(); ++iter) {
337  temp = iter.position();
338  vec = shiftToMinimum(temp, mesh().dimensions(), *unitCellPtr_);
339  dGsq_(iter.rank(), n) = unitCellPtr_->dksq(vec, n);
340  for (int p = 0; p < D; ++p) {
341  if (temp [p] != 0) {
342  Partner[p] = mesh().dimensions()[p] - temp[p];
343  } else {
344  Partner[p] = 0;
345  }
346  }
347  if (Partner[D-1] > kMeshDimensions_[D-1]) {
348  dGsq_(iter.rank(), n) *= 2;
349  }
350  }

355  * Propagate solution by one step.
356  */
357  template <int D>
358  void Block<D>::step(const QField& q, QField& qNew)
359  {
360  // Check real-space mesh sizes`
361  int nx = mesh().size();
362  UTIL_CHECK(nx > 0);
363  UTIL_CHECK(q.isAllocated());
364  UTIL_CHECK(qNew.isAllocated());
365  UTIL_CHECK(q.capacity() == nx);
366  UTIL_CHECK(qNew.capacity() == nx);
367  UTIL_CHECK(qr_.capacity() == nx);
368  UTIL_CHECK(expW_.capacity() == nx);
369 
370  // Fourier-space mesh sizes
371  int nk = qk_.capacity();
372  UTIL_CHECK(expKsq_.capacity() == nk);
373 
374  // Apply pseudo-spectral algorithm
375  int i;
376  for (i = 0; i < nx; ++i) {
377  qr_[i] = q[i]*expW_[i];
378  qr2_[i] = q[i]*expW2_[i];
379  }
380  fft_.forwardTransform(qr_, qk_);
381  fft_.forwardTransform(qr2_, qk2_);
382  for (i = 0; i < nk; ++i) {
383  qk_[i][0] *= expKsq_[i];
384  qk_[i][1] *= expKsq_[i];
385  qk2_[i][0] *= expKsq2_[i];
386  qk2_[i][1] *= expKsq2_[i];
387  }
388  fft_.inverseTransform(qk_, qr_);
389  fft_.inverseTransform(qk2_, qr2_);
390  for (i = 0; i < nx; ++i) {
391  qf_[i] = qr_[i]*expW_[i];
392  qr2_[i] = qr2_[i]*expW_[i];
393  }
394 
395  fft_.forwardTransform(qr2_, qk2_);
396  for (i = 0; i < nk; ++i) {
397  qk2_[i][0] *= expKsq2_[i];
398  qk2_[i][1] *= expKsq2_[i];
399  }
400  fft_.inverseTransform(qk2_, qr2_);
401  for (i = 0; i < nx; ++i) {
402  qr2_[i] = qr2_[i]*expW2_[i];
403  }
404  for (i = 0; i < nx; ++i) {
405  qNew[i] = (4.0*qr2_[i] - qf_[i])/3.0;
406  }
407  }
408 
409 }
