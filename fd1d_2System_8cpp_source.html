226  void System::readCommands(std::istream &in)
227  {
228  UTIL_CHECK(hasMixture_);
229  UTIL_CHECK(hasDomain_);
230 
231  std::string command;
232  std::string filename;
233  FieldIo fieldIo(*this);
234 
235  std::istream& inBuffer = in;
236 
237  bool readNext = true;
238  while (readNext) {
239 
240  inBuffer >> command;
241  Log::file() << command;
242 
243  if (command == "FINISH") {
244  Log::file() << std::endl;
245  readNext = false;
246  } else
247  if (command == "READ_W") {
248  inBuffer >> filename;
249  Log::file() << " " << Str(filename, 20) << std::endl;
250  fieldIo.readFields(wFields(), filename);
251  } else
252  if (command == "ITERATE") {
253  Log::file() << std::endl;
254  iterator().solve();
255  outputThermo(Log::file());
256  } else
257  if (command == "COMPARE_HOMOGENEOUS") {
258  int mode;
259  inBuffer >> mode;
260  Log::file() << std::endl;
261  Log::file() << "mode = " << mode << std::endl;
262 
263  HomogeneousComparison comparison(*this);
264  comparison.compute(mode);
265  comparison.output(mode, Log::file());
266  } else
267  if (command == "SWEEP") {
268  UTIL_CHECK(hasSweep_);
269  UTIL_CHECK(sweepPtr_);
270  sweepPtr_->solve();
271  } else
272  if (command == "WRITE_W") {
273  inBuffer >> filename;
274  Log::file() << " " << Str(filename, 20) << std::endl;
275  fieldIo.writeFields(wFields(), filename);
276  } else
277  if (command == "WRITE_C") {
278  inBuffer >> filename;
279  Log::file() << " " << Str(filename, 20) << std::endl;
280  fieldIo.writeFields(cFields(), filename);
281  } else
282  if (command == "WRITE_BLOCK_C") {
283  inBuffer >> filename;
284  Log::file() << " " << Str(filename, 20) << std::endl;
285  fieldIo.writeBlockCFields(filename);
286  } else
287  if (command == "WRITE_VERTEX_Q") {
288  int polymerId, vertexId;
289  inBuffer >> filename;
290  Log::file() << " " << Str(filename, 20) << std::endl;
291  inBuffer >> polymerId;
292  Log::file() << "polymerId = "
293  << Int(polymerId, 5) << std::endl;
294  inBuffer >> vertexId;
295  Log::file() << "vertexId = "
296  << Int(vertexId, 5) << std::endl;
297  fieldIo.writeVertexQ(polymerId, vertexId, filename);
298  } else
299  if (command == "REMESH_W") {
300  int nx;
301  inBuffer >> nx;
302  Log::file() << std::endl;
303  Log::file() << "nx = " << Int(nx, 20) << std::endl;
304  inBuffer >> filename;
305  Log::file() << "outfile = " << Str(filename, 20) << std::endl;
306  fieldIo.remesh(wFields(), nx, filename);
307  } else
308  if (command == "EXTEND_W") {
309  int m;
310  inBuffer >> m;
311  Log::file() << std::endl;
312  Log::file() << "m = " << Int(m, 20) << std::endl;
313  inBuffer >> filename;
314  Log::file() << "outfile = " << Str(filename, 20) << std::endl;
315  fieldIo.extend(wFields(), m, filename);
316  } else {
317  Log::file() << " Error: Unknown command " << command << std::endl;
318  readNext = false;
319  }
320 
321  }

335  /*
336  * Compute Helmoltz free energy and pressure
337  */
338  void System::computeFreeEnergy()
339  {
340  fHelmholtz_ = 0.0;
341 
342  // Compute ideal gas contributions to fHelhmoltz_
343  Polymer* polymerPtr;
344  double phi, mu, length;
345  int np = mixture().nPolymer();
346  for (int i = 0; i < np; ++i) {
347  polymerPtr = &mixture().polymer(i);
348  phi = polymerPtr->phi();
349  mu = polymerPtr->mu();
350  // Recall: mu = ln(phi/q)
351  length = polymerPtr->length();
352  fHelmholtz_ += phi*( mu - 1.0 )/length;
353  }
354 
355  // Apply Legendre transform subtraction
356  int nm = mixture().nMonomer();
357  for (int i = 0; i < nm; ++i) {
358  fHelmholtz_ -=
359  domain().innerProduct(wFields_[i], cFields_[i]);
360  }
361 
362  // Add average interaction free energy density per monomer
363  int nx = domain().nx();
364  if (!f_.isAllocated()) f_.allocate(nx);
365  if (!c_.isAllocated()) c_.allocate(nm);
366  int j;
367  for (int i = 0; i < nx; ++i) {
368  // Store c_[j] = local concentration of species j
369  for (j = 0; j < nm; ++j) {
370  c_[j] = cFields_[j][i];
371  }
372  // Compute f_[i] = excess free eenrgy at grid point i
373  f_[i] = interaction().fHelmholtz(c_);
374  }
375  fHelmholtz_ += domain().spatialAverage(f_);
376 
377  // Compute pressure
378  pressure_ = -fHelmholtz_;
379  for (int i = 0; i < np; ++i) {
380  polymerPtr = & mixture().polymer(i);
381  phi = polymerPtr->phi();
382  mu = polymerPtr->mu();
383  length = polymerPtr->length();
384  pressure_ += phi*mu/length;
385  }
386 
387  }

440  void System::initHomogeneous()
441  {
442 
443  // Set number of molecular species and monomers
444  int nm = mixture().nMonomer();
445  int np = mixture().nPolymer();
446  //int ns = mixture().nSolvent();
447  int ns = 0;
448  UTIL_CHECK(homogeneous_.nMolecule() == np + ns);
449  UTIL_CHECK(homogeneous_.nMonomer() == nm);
450 
451  // Allocate c_ work array, if necessary
452  if (c_.isAllocated()) {
453  UTIL_CHECK(c_.capacity() == nm);
454  } else {
455  c_.allocate(nm);
456  }
457 
458  int i; // molecule index
459  int j; // monomer index
460  int k; // block or clump index
461  int nb; // number of blocks
462  int nc; // number of clumps
463 
464  // Loop over polymer molecule species
465  for (i = 0; i < np; ++i) {
466 
467  // Initial array of clump sizes
468  for (j = 0; j < nm; ++j) {
469  c_[j] = 0.0;
470  }
471 
472  // Compute clump sizes for all monomer types.
473  nb = mixture().polymer(i).nBlock();
474  for (k = 0; k < nb; ++k) {
475  Block& block = mixture().polymer(i).block(k);
476  j = block.monomerId();
477  c_[j] += block.length();
478  }
479 
480  // Count the number of clumps of nonzero size
481  nc = 0;
482  for (j = 0; j < nm; ++j) {
483  if (c_[j] > 1.0E-8) {
484  ++nc;
485  }
486  }
487  homogeneous_.molecule(i).setNClump(nc);
488 
489  // Set clump properties for this Homogeneous::Molecule
490  k = 0; // Clump index
491  for (j = 0; j < nm; ++j) {
492  if (c_[j] > 1.0E-8) {
493  homogeneous_.molecule(i).clump(k).setMonomerId(j);
494  homogeneous_.molecule(i).clump(k).setSize(c_[j]);
495  ++k;
496  }
497  }
498  homogeneous_.molecule(i).computeSize();
499 
500  }
