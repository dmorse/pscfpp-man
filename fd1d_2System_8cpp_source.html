93 /*
94 * Process command line options.
95 */
96 void System::setOptions(int argc, char **argv)
97 {
98 bool eflag = false; // echo
99 bool pFlag = false; // param file
100 bool cFlag = false; // command file
101 bool iFlag = false; // input prefix
102 bool oFlag = false; // output prefix
103 char* pArg = 0;
104 char* cArg = 0;
105 char* iArg = 0;
106 char* oArg = 0;
107
108 // Read program arguments
109 int c;
110 opterr = 0;
111 while ((c = getopt(argc, argv, "er:p:c:i:o:f")) != -1) {
112 switch (c) {
113 case 'e':
114 eflag = true;
115 break;
116 case 'p': // parameter file
117 pFlag = true;
118 pArg = optarg;
119 break;
120 case 'c': // command file
121 cFlag = true;
122 cArg = optarg;
123 break;
124 case 'i': // input prefix
125 iFlag = true;
126 iArg = optarg;
127 break;
128 case 'o': // output prefix
129 oFlag = true;
130 oArg = optarg;
131 break;
132 case '?':
133 Log::file() << "Unknown option -" << optopt << std::endl;
134 UTIL_THROW("Invalid command line option");
135 }
136 }
137
138 // Set flag to echo parameters as they are read.
139 if (eflag) {
140 Util::ParamComponent::setEcho(true);
141 }
142
143 // If option -p, set parameter file name
144 if (pFlag) {
145 fileMaster().setParamFileName(std::string(pArg));
146 }
147
148 // If option -c, set command file name
149 if (cFlag) {
150 fileMaster().setCommandFileName(std::string(cArg));
151 }
152
153 // If option -i, set path prefix for input files
154 if (iFlag) {
155 fileMaster().setInputPrefix(std::string(iArg));
156 }
157
158 // If option -o, set path prefix for output files
159 if (oFlag) {
160 fileMaster().setOutputPrefix(std::string(oArg));
161 }

163 }

165 /*
166 * Read parameters and initialize.
167 */
168 void System::readParameters(std::istream& in)
169 {
170 readParamComposite(in, mixture());
171 hasMixture_ = true;
172
173 // Initialize homogeneous object
174 int nm = mixture().nMonomer();
175 int np = mixture().nPolymer();
176 int ns = mixture().nSolvent();
177 homogeneous_.setNMolecule(np+ns);
178 homogeneous_.setNMonomer(nm);
179 initHomogeneous();
180
181 interaction().setNMonomer(mixture().nMonomer());
182 readParamComposite(in, interaction());
183
184 readParamComposite(in, domain());
185 hasDomain_ = true;
186 allocateFields();
187
188 // Instantiate and initialize an Iterator
189 std::string className;
190 bool isEnd;
191 iteratorPtr_ = iteratorFactoryPtr_->readObject(in, *this,
192 className, isEnd);
193 if (!iteratorPtr_) {
194 std::string msg = "Unrecognized Iterator subclass name ";
195 msg += className;
196 UTIL_THROW(msg.c_str());
197 }
198
199 // Optionally instantiate and initialize a Sweep object
200 sweepPtr_ =
201 sweepFactoryPtr_->readObjectOptional(in, *this, className, isEnd);
202 if (sweepPtr_) {
203 hasSweep_ = true;
204 sweepPtr_->setSystem(*this);
205 } else {
206 hasSweep_ = false;
207 }
208 }

263 /*
264 * Read and execute commands from a specified command file.
265 */
266 void System::readCommands(std::istream &in)
267 {
268 UTIL_CHECK(hasMixture_);
269 UTIL_CHECK(hasDomain_);
270
271 std::string command;
272 std::string filename;
273
274 std::istream& inBuffer = in;
275
276 bool readNext = true;
277 while (readNext) {
278
279 inBuffer >> command;
280
281 if (inBuffer.eof()) {
282 break;
283 } else {
284 Log::file() << command;
285 }
286
287 if (command == "FINISH") {
288 Log::file() << std::endl;
289 readNext = false;
290 } else
291 if (command == "READ_W") {
292 readEcho(inBuffer, filename);
293 fieldIo_.readFields(wFields(), filename);
294 } else
295 if (command == "COMPUTE") {
296 // Solve the modified diffusion equation, without iteration
297 Log::file() << std::endl;
298 compute();
299 } else
300 if (command == "ITERATE") {
301 // Attempt iteratively solve a single SCFT problem
302 bool isContinuation = false;
303 int fail = iterate(isContinuation);
304 if (fail) {
305 readNext = false;
306 }
307 } else
308 if (command == "SWEEP") {
309 // Attempt to solve a sequence of SCFT problems along a line
310 // through parameter space.
311 sweep();
312 } else
313 if (command == "COMPARE_HOMOGENEOUS") {
314 int mode;
315 inBuffer >> mode;
316 Log::file() << std::endl;
317 Log::file() << "mode = " << mode << std::endl;
318
319 HomogeneousComparison comparison(*this);
320 comparison.compute(mode);
321 comparison.output(mode, Log::file());
322 } else
323 if (command == "WRITE_W") {
324 readEcho(inBuffer, filename);
325 fieldIo_.writeFields(wFields(), filename);
326 } else
327 if (command == "WRITE_C") {
328 readEcho(inBuffer, filename);
329 fieldIo_.writeFields(cFields(), filename);
330 } else
331 if (command == "WRITE_BLOCK_C") {
332 readEcho(inBuffer, filename);
333 fieldIo_.writeBlockCFields(mixture_, filename);
334 } else
335 if (command == "WRITE_PARAM") {
336 readEcho(inBuffer, filename);
337 std::ofstream file;
338 fileMaster().openOutputFile(filename, file);
339 writeParam(file);
340 file.close();
341 } else
342 if (command == "WRITE_THERMO") {
343 readEcho(inBuffer, filename);
344 std::ofstream file;
345 fileMaster().openOutputFile(filename, file, std::ios_base::app);
346 writeThermo(file);
347 file.close();
348 } else
349 if (command == "WRITE_VERTEX_Q") {
350 int polymerId, vertexId;
351 inBuffer >> polymerId;
352 Log::file() << std::endl;
353 Log::file() << "polymerId = "
354 << Int(polymerId, 5) << std::endl;
355 inBuffer >> vertexId;
356 Log::file() << "vertexId = "
357 << Int(vertexId, 5) << std::endl;
358 inBuffer >> filename;
359 Log::file() << "outfile = "
360 << Str(filename, 20) << std::endl;
361 fieldIo_.writeVertexQ(mixture_, polymerId, vertexId, filename);
362 } else
363 if (command == "REMESH_W") {
364 int nx;
365 inBuffer >> nx;
366 Log::file() << std::endl;
367 Log::file() << "nx = " << Int(nx, 20) << std::endl;
368 inBuffer >> filename;
369 Log::file() << "outfile = " << Str(filename, 20) << std::endl;
370 fieldIo_.remesh(wFields(), nx, filename);
371 } else
372 if (command == "EXTEND_W") {
373 int m;
374 inBuffer >> m;
375 Log::file() << std::endl;
376 Log::file() << "m = " << Int(m, 20) << std::endl;
377 inBuffer >> filename;
378 Log::file() << "outfile = " << Str(filename, 20) << std::endl;
379 fieldIo_.extend(wFields(), m, filename);
380 } else {
381 Log::file() << " Error: Unknown command "
382 << command << std::endl;
383 readNext = false;
384 }
385
386 }
387 }

462 /*
463 * Compute Helmoltz free energy and pressure
464 */
465 void System::computeFreeEnergy()
466 {
467 fHelmholtz_ = 0.0;
468 int np = mixture().nPolymer();
469 int ns = mixture().nSolvent();
470 int nm = mixture().nMonomer();
471 double phi, mu;
472
473 // Polymeric ideal gas contributions to fHelhmoltz_
474 if (np > 0) {
475 Polymer* polymerPtr;
476 double length;
477 for (int i = 0; i < np; ++i) {
478 polymerPtr = &mixture().polymer(i);
479 phi = polymerPtr->phi();
480 mu = polymerPtr->mu();
481 // Recall: mu = ln(phi/q)
482 length = polymerPtr->length();
483 if (phi > 1E-08) {
484 fHelmholtz_ += phi*( mu - 1.0 )/length;
485 }
486 }
487 }
488
489 // Solvent ideal gas contributions to fHelhmoltz_
490 if (ns > 0) {
491 Solvent* solventPtr;
492 double size;
493 for (int i = 0; i < ns; ++i) {
494 solventPtr = &mixture().solvent(i);
495 phi = solventPtr->phi();
496 mu = solventPtr->mu();
497 // Recall: mu = ln(phi/q)
498 size = solventPtr->size();
499 if (phi > 1E-08) {
500 fHelmholtz_ += phi*( mu - 1.0 )/size;
501 }
502 }
503 }
504
505 // Apply Legendre transform subtraction
506 for (int i = 0; i < nm; ++i) {
507 fHelmholtz_ -=
508 domain().innerProduct(wFields_[i], cFields_[i]);
509 }
510
511 // Add average interaction free energy density per monomer
512 int nx = domain().nx();
513 if (!f_.isAllocated()) f_.allocate(nx);
514 if (!c_.isAllocated()) c_.allocate(nm);
515 int j;
516 for (int i = 0; i < nx; ++i) {
517 // Store c_[j] = local concentration of species j
518 for (j = 0; j < nm; ++j) {
519 c_[j] = cFields_[j][i];
520 }
521 // Compute f_[i] = excess free eenrgy at grid point i
522 f_[i] = interaction().fHelmholtz(c_);
523 }
524 fHelmholtz_ += domain().spatialAverage(f_);
525
526 // Compute pressure
527 pressure_ = -fHelmholtz_;
528 if (np > 0) {
529 double length;
530 Polymer* polymerPtr;
531 for (int i = 0; i < np; ++i) {
532 polymerPtr = & mixture().polymer(i);
533 phi = polymerPtr->phi();
534 mu = polymerPtr->mu();
535 length = polymerPtr->length();
536 if (phi > 1E-08) {
537 pressure_ += phi*mu/length;
538 }
539 }
540 }
541 if (ns > 0) {
542 double size;
543 Solvent* solventPtr;
544 for (int i = 0; i < ns; ++i) {
545 solventPtr = & mixture().solvent(i);
546 phi = solventPtr->phi();
547 mu = solventPtr->mu();
548 size = solventPtr->size();
549 if (phi > 1E-08) {
550 pressure_ += phi*mu/size;
551 }
552 }
553 }
554
555 }

557 void System::initHomogeneous()
558 {
559
560 // Set number of molecular species and monomers
561 int nm = mixture().nMonomer();
562 int np = mixture().nPolymer();
563 int ns = mixture().nSolvent();
564 UTIL_CHECK(homogeneous_.nMolecule() == np + ns);
565 UTIL_CHECK(homogeneous_.nMonomer() == nm);
566
567 // Allocate c_ work array, if necessary
568 if (c_.isAllocated()) {
569 UTIL_CHECK(c_.capacity() == nm);
570 } else {
571 c_.allocate(nm);
572 }
573
574 int i; // molecule index
575 int j; // monomer index
576 int k; // block or clump index
577 int nb; // number of blocks
578 int nc; // number of clumps
579
580 // Loop over polymer molecule species
581 if (np > 0) {
582 for (i = 0; i < np; ++i) {
583
584 // Initial array of clump sizes
585 for (j = 0; j < nm; ++j) {
586 c_[j] = 0.0;
587 }
588
589 // Compute clump sizes for all monomer types.
590 nb = mixture().polymer(i).nBlock();
591 for (k = 0; k < nb; ++k) {
592 Block& block = mixture().polymer(i).block(k);
593 j = block.monomerId();
594 c_[j] += block.length();
595 }
596
597 // Count the number of clumps of nonzero size
598 nc = 0;
599 for (j = 0; j < nm; ++j) {
600 if (c_[j] > 1.0E-8) {
601 ++nc;
602 }
603 }
604 homogeneous_.molecule(i).setNClump(nc);
605
606 // Set clump properties for this Homogeneous::Molecule
607 k = 0; // Clump index
608 for (j = 0; j < nm; ++j) {
609 if (c_[j] > 1.0E-8) {
610 homogeneous_.molecule(i).clump(k).setMonomerId(j);
611 homogeneous_.molecule(i).clump(k).setSize(c_[j]);
612 ++k;
613 }
614 }
615 homogeneous_.molecule(i).computeSize();
616
617 }
618 }
619
620 // Add solvent contributions
621 if (np > 0) {
622 double size;
623 int monomerId;
624 for (int is = 0; is < ns; ++is) {
625 i = is + np;
626 monomerId = mixture().solvent(is).monomerId();
627 size = mixture().solvent(is).size();
628 homogeneous_.molecule(i).setNClump(1);
629 homogeneous_.molecule(i).clump(0).setMonomerId(monomerId);
630 homogeneous_.molecule(i).clump(0).setSize(size);
631 homogeneous_.molecule(i).computeSize();
632 }
633 }
