90 /*
91 * Extract a Edge from an istream.
92 */
93 std::istream& operator >> (std::istream& in, Edge& block)
94 {
95 // Read monomer type id
96 in >> block.monomerId_;
97
98 // Read length or nBead
99 if (PolymerModel::isThread()) {
100 in >> block.length_;
101 } else
102 if (PolymerModel::isBead()) {
103 in >> block.nBead_;
104 }
105
106 // For branched polyimers, read topology information
107 if (block.polymerType_ == PolymerType::Branched) {
108 in >> block.vertexIds_[0];
109 in >> block.vertexIds_[1];
110 if (PolymerModel::isBead()) {
111 in >> block.ownsVertex_[0];
112 in >> block.ownsVertex_[1];
113 }
114 }
115
116 return in;
117 }

119 /*
120 * Output a Edge to an ostream, without line breaks.
121 */
122 std::ostream& operator << (std::ostream& out,
123 Edge const & block)
124 {
125 // Write monomer type id
126 out << " " << block.monomerId_;
127 out << " ";
128
129 // Write length or nBead
130 if (PolymerModel::isThread()) {
131 out.setf(std::ios::scientific);
132 out.width(20);
133 out.precision(12);
134 out << block.length_;
135 } else
136 if (PolymerModel::isBead()) {
137 out << block.nBead_;
138 }
139
140 // For branched polymer, write topology information
141 if (block.polymerType_ == PolymerType::Branched) {
142 out << " " << block.vertexIds_[0];
143 out << " " << block.vertexIds_[1];
144 if (PolymerModel::isBead()) {
145 out << " " << block.ownsVertex_[0];
146 out << " " << block.ownsVertex_[1];
147 }
148 }
149
150 return out;
151 }
