300 template <class Block>
301 void PolymerTmpl<Block>::readParameters(std::istream& in)
302 {
303
304 PolymerSpecies::readParameters(in);
305
306 // Set sources for all propagators
307 Vertex const * vertexPtr = 0;
308 Propagator const * sourcePtr = 0;
309 Propagator * propagatorPtr = 0;
310 Pair<int> propId;
311 int blockId, directionId, vertexId, i;
312 for (blockId = 0; blockId < nBlock(); ++blockId) {
313 // Add sources
314 for (directionId = 0; directionId < 2; ++directionId) {
315 vertexId = block(blockId).vertexId(directionId);
316 vertexPtr = &vertex(vertexId);
317 propagatorPtr = &block(blockId).propagator(directionId);
318 for (i = 0; i < vertexPtr->size(); ++i) {
319 propId = vertexPtr->inPropagatorId(i);
320 if (propId[0] == blockId) {
321 UTIL_CHECK(propId[1] != directionId);
322 } else {
323 sourcePtr =
324 &block(propId[0]).propagator(propId[1]);
325 propagatorPtr->addSource(*sourcePtr);
326 }
327 } // end loop over inputs to head vertex
328 } // end loop over directionId
329 } // end loop over blockId
330
331 // If using bead model, set propagator vertex ownership
332 Block* blockPtr;
333 if (PolymerModel::isBead()) {
334 bool own0, own1;
335 for (int blockId = 0; blockId < nBlock(); ++blockId) {
336 blockPtr = &(blocks_[blockId]);
337 own0 = blockPtr->ownsVertex(0);
338 own1 = blockPtr->ownsVertex(1);
339 blockPtr->propagator(0).setVertexOwnership(own0, own1);
340 blockPtr->propagator(1).setVertexOwnership(own1, own0);
341 }
342 }
343
344 // If using bead model, check ownership of vertex beads
345 if (PolymerModel::isBead()) {
346 int vSize, ib, id, nOwner;
347 Pair<int> propId;
348
349 // Loop over vertices
350 for (int vertexId = 0; vertexId < nVertex(); ++vertexId) {
351 vSize = vertex(vertexId).size();
352
353 // Incoming propagators
354 nOwner = 0;
355 for (int ip = 0; ip < vSize; ++ip) {
356 propId = vertex(vertexId).inPropagatorId(ip);
357 ib = propId[0];
358 id = propId[1];
359 if (block(ib).propagator(id).ownsTail()) {
360 ++nOwner;
361 }
362 }
363 UTIL_CHECK(nOwner == 1);
364
365 // Outgoing propagators
366 nOwner = 0;
367 for (int i = 0; i < vSize; ++i) {
368 propId = vertex(vertexId).outPropagatorId(i);
369 ib = propId[0];
370 id = propId[1];
371 if (block(ib).propagator(id).ownsHead()) {
372 ++nOwner;
373 }
374 }
375 UTIL_CHECK(nOwner == 1);
376
377 } // end loop over vertices
378
379 // Check consistency of block and propagator ownership flags
380 bool own0, own1;
381 for (int blockId = 0; blockId < nBlock(); ++blockId) {
382 blockPtr = &(block(blockId));
383 own0 = blockPtr->ownsVertex(0);
384 own1 = blockPtr->ownsVertex(1);
385 UTIL_CHECK(blockPtr->propagator(0).ownsHead() == own0);
386 UTIL_CHECK(blockPtr->propagator(1).ownsTail() == own0);
387 UTIL_CHECK(blockPtr->propagator(0).ownsTail() == own1);
388 UTIL_CHECK(blockPtr->propagator(1).ownsHead() == own1);
389 }
390
391 } // end if (PolymerModel::isBead())
392
393 }

395 /*
396 * Compute a solution to the MDE and block concentrations.
397 */
398 template <class Block>
399 void PolymerTmpl<Block>::solve(double phiTot)
400 {
401
402 // Clear all propagators
403 for (int j = 0; j < nPropagator(); ++j) {
404 propagator(j).setIsSolved(false);
405 }
406
407 // Solve modified diffusion equation for all propagators in
408 // the order specified by function makePlan.
409 for (int j = 0; j < nPropagator(); ++j) {
410 UTIL_CHECK(propagator(j).isReady());
411 propagator(j).solve();
412 }
413
414 // Compute molecular partition function q_
415 q_ = block(0).propagator(0).computeQ();
416
417 // The Propagator::computeQ function returns a spatial average.
418 // Correct for partial occupation of the unit cell.
419 q_ = q_/phiTot;
420
421 // Compute mu_ or phi_, depending on ensemble
422 if (ensemble() == Species::Closed) {
423 mu_ = log(phi_/q_);
424 } else
425 if (ensemble() == Species::Open) {
426 phi_ = exp(mu_)*q_;
427 }
428
429 #if 0
430 // Compute block concentration fields
431 double prefactor;
432 if (PolymerModel::isThread()) {
433 prefactor = phi_ / ( q_ * length() );
434 } else
435 if (PolymerModel::isBead()) {
436 prefactor = phi_ / ( q_ * (double)nBead() );
437 }
438 for (int i = 0; i < nBlock(); ++i) {
439 block(i).computeConcentration(prefactor);
440 }
441 #endif
442
443 }
