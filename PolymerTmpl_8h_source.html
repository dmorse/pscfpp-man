311  template <class Block>
312  void PolymerTmpl<Block>::readParameters(std::istream& in)
313  {
314  read<int>(in, "nBlock", nBlock_);
315  read<int>(in, "nVertex", nVertex_);
316 
317  // Allocate all arrays
318  blocks_.allocate(nBlock_);
319  vertices_.allocate(nVertex_);
320  propagatorIds_.allocate(2*nBlock_);
321 
322  readDArray<Block>(in, "blocks", blocks_, nBlock_);
323 
324  // Set vertex indices
325  for (int vertexId = 0; vertexId < nVertex_; ++vertexId) {
326  vertices_[vertexId].setId(vertexId);
327  }
328 
329  // Add blocks to vertices
330  int vertexId0, vertexId1;
331  Block* blockPtr;
332  for (int blockId = 0; blockId < nBlock_; ++blockId) {
333  blockPtr = &(blocks_[blockId]);
334  vertexId0 = blockPtr->vertexId(0);
335  vertexId1 = blockPtr->vertexId(1);
336  vertices_[vertexId0].addBlock(*blockPtr);
337  vertices_[vertexId1].addBlock(*blockPtr);
338  }
339 
340  makePlan();
341 
342  // Read ensemble and phi or mu
343  ensemble_ = Species::Closed;
344  readOptional<Species::Ensemble>(in, "ensemble", ensemble_);
345  if (ensemble_ == Species::Closed) {
346  read(in, "phi", phi_);
347  } else {
348  read(in, "mu", mu_);
349  }
350 
351  // Set sources for all propagators
352  Vertex const * vertexPtr = 0;
353  Propagator const * sourcePtr = 0;
354  Propagator * propagatorPtr = 0;
355  Pair<int> propagatorId;
356  int blockId, directionId, vertexId, i;
357  for (blockId = 0; blockId < nBlock(); ++blockId) {
358  // Add sources
359  for (directionId = 0; directionId < 2; ++directionId) {
360  vertexId = block(blockId).vertexId(directionId);
361  vertexPtr = &vertex(vertexId);
362  propagatorPtr = &block(blockId).propagator(directionId);
363  for (i = 0; i < vertexPtr->size(); ++i) {
364  propagatorId = vertexPtr->inPropagatorId(i);
365  if (propagatorId[0] == blockId) {
366  UTIL_CHECK(propagatorId[1] != directionId);
367  } else {
368  sourcePtr =
369  &block(propagatorId[0]).propagator(propagatorId[1]);
370  propagatorPtr->addSource(*sourcePtr);
371  }
372  }
373  }

378  template <class Block>
379  void PolymerTmpl<Block>::makePlan()
380  {
381  if (nPropagator_ != 0) {
382  UTIL_THROW("nPropagator !=0 on entry");
383  }
384 
385  // Allocate and initialize isFinished matrix
386  DMatrix<bool> isFinished;
387  isFinished.allocate(nBlock_, 2);
388  for (int iBlock = 0; iBlock < nBlock_; ++iBlock) {
389  for (int iDirection = 0; iDirection < 2; ++iDirection) {
390  isFinished(iBlock, iDirection) = false;
391  }
392  }
393 
394  Pair<int> propagatorId;
395  Vertex* inVertexPtr = 0;
396  int inVertexId = -1;
397  bool isReady;
398  while (nPropagator_ < nBlock_*2) {
399  for (int iBlock = 0; iBlock < nBlock_; ++iBlock) {
400  for (int iDirection = 0; iDirection < 2; ++iDirection) {
401  if (isFinished(iBlock, iDirection) == false) {
402  inVertexId = blocks_[iBlock].vertexId(iDirection);
403  inVertexPtr = &vertices_[inVertexId];
404  isReady = true;
405  for (int j = 0; j < inVertexPtr->size(); ++j) {
406  propagatorId = inVertexPtr->inPropagatorId(j);
407  if (propagatorId[0] != iBlock) {
408  if (!isFinished(propagatorId[0], propagatorId[1])) {
409  isReady = false;
410  break;
411  }
412  }
413  }
414  if (isReady) {
415  propagatorIds_[nPropagator_][0] = iBlock;
416  propagatorIds_[nPropagator_][1] = iDirection;
417  isFinished(iBlock, iDirection) = true;
418  ++nPropagator_;
419  }
420  }
421  }
422  }
423  }

430  template <class Block>
431  void PolymerTmpl<Block>::solve()
432  {
433 
434  // Clear all propagators
435  for (int j = 0; j < nPropagator(); ++j) {
436  propagator(j).setIsSolved(false);
437  }
438 
439  // Solve modified diffusion equation for all propagators
440  for (int j = 0; j < nPropagator(); ++j) {
441  UTIL_CHECK(propagator(j).isReady());
442  propagator(j).solve();
443  }
444 
445  // Compute molecular partition function
446  double q = block(0).propagator(0).computeQ();
447  if (ensemble() == Species::Closed) {
448  mu_ = log(phi_/q);
449  }
450  else if (ensemble() == Species::Open) {
451  phi_ = exp(mu_)*q;
452  }
453 
454  // Compute block concentration fields
455  double prefactor = phi_ / (q *length() );
456  for (int i = 0; i < nBlock(); ++i) {
457  block(i).computeConcentration(prefactor);
458  }
459 
460  }
