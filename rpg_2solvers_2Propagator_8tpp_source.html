40 /*
41 * Allocate memory used by this propagator.
42 */
43 template <int D>
44 void Propagator<D>::allocate(int ns, const Mesh<D>& mesh)
45 {
46 ns_ = ns;
47 meshPtr_ = &mesh;
48
49 // Allocate memory in qFieldsAll_ using value of ns
50 int meshSize = meshPtr_->size();
51 qFieldsAll_.allocate(ns * meshSize);
52
53 // Set up array of associated RField<D> arrays
54 qFields_.allocate(ns);
55 for (int i = 0; i < ns; ++i) {
56 qFields_[i].associate(qFieldsAll_, i*meshSize, meshPtr_->dimensions());
57 }
58 isAllocated_ = true;
59 }

66 {

69 ns_ = ns;

71 // Deallocate memory previously used by this propagator.
72 qFields_.deallocate(); // Destroys associated RField<D> objects
73 // but not the underlying data
74 qFieldsAll_.deallocate(); // Destroy actual propagator data
75
76 // Allocate memory in qFieldsAll_ using new value of ns
77 int meshSize = meshPtr_->size();
78 qFieldsAll_.allocate(ns * meshSize);
79
80 // Set up array of associated RField<D> arrays
81 qFields_.allocate(ns);
82 for (int i = 0; i < ns; ++i) {
83 qFields_[i].associate(qFieldsAll_, i*meshSize, meshPtr_->dimensions());
84 }

89 /*
90 * Compute initial head QField from final tail QFields of sources.
91 */
92 template <int D>
93 void Propagator<D>::computeHeadThread()
94 {
95
96 // Initialize head field (s=0) to 1.0 at all grid points
97 int nx = meshPtr_->size();
98 VecOp::eqS(qFields_[0], 1.0);
99
100 // Multiply head q-field by tail q-fields of all sources
101 if (nSource() > 0) {
102 DArray<DeviceArray<cudaReal> const *> tails;
103 tails.allocate(nSource()+1);
104 tails[0] = &qFields_[0];
105 for (int is = 0; is < nSource(); ++is) {
106 if (!source(is).isSolved()) {
107 UTIL_THROW("Source not solved in computeHeadThread");
108 }
109 tails[is+1] = &(source(is).tail());
110 }
111 VecOp::mulVMany(qFields_[0], tails);
112 }
113 }

134 /*
135 * Solve the modified diffusion equation for this block.
136 */
137 template <int D>
138 void Propagator<D>::solve()
139 {
140 UTIL_CHECK(isAllocated());
141
142 if (PolymerModel::isThread()) {
143
144 // Initialize head as pointwise product of source propagators
145 computeHeadThread();
146
147 // MDE step loop for thread model
148 for (int iStep = 0; iStep < ns_ - 1; ++iStep) {
149 block().stepThread(qFields_[iStep], qFields_[iStep + 1]);
150 }
151
152 } else
153 if (PolymerModel::isBead()) {
154
155 // Initialize head, starting with product of source propagators
156 computeHeadBead();
157
158 // MDE step loop for bead model (stop before tail vertex)
159 int iStep;
160 for (iStep = 0; iStep < ns_ - 2; ++iStep) {
161 block().stepBead(qFields_[iStep], qFields_[iStep + 1]);
162 }
163
164 // Compute q for the tail vertex
165 iStep = ns_ - 2;
166 if (ownsTail()) {
167 // Full step, including bead weight for tail vertex
168 block().stepBead(qFields_[iStep], qFields_[iStep + 1]);
169 } else {
170 // Bond operator, excluding bead weight for tail vertex
171 block().stepBondBead(qFields_[iStep], qFields_[iStep + 1]);
172 }
173
174 } else {
175 // This should be impossible
176 UTIL_THROW("Unexpected PolymerModel type");
177 }
178
179 setIsSolved(true);
180 }

182 /*
183 * Solve the modified diffusion equation with specified initial field.
184 */
185 template <int D>
186 void Propagator<D>::solve(RField<D> const & head)
187 {
188 UTIL_CHECK(isAllocated());
189
190 // Initialize initial (head) field
191 VecOp::eqV(qFields_[0], head);
192
193 for (int iStep = 0; iStep < ns_ - 1; ++iStep) {
194 block().stepThread(qFields_[iStep], qFields_[iStep+1]);
195 }
196 setIsSolved(true);
197 }

199 /*
200 * Integrate to calculate monomer concentration for this block
201 */
202 template <int D>
203 double Propagator<D>::computeQ()
204 {
205 // Preconditions
206 if (!isSolved()) {
207 UTIL_THROW("Propagator is not solved.");
208 }
209 if (!hasPartner()) {
210 UTIL_THROW("Propagator has no partner set.");
211 }
212 if (!partner().isSolved()) {
213 UTIL_THROW("Partner propagator is not solved");
214 }
215
216 QField const & qh = head();
217 QField const & qt = partner().tail();
218 int nx = meshPtr_->size();
219 UTIL_CHECK(nx == qh.capacity());
220 UTIL_CHECK(nx == qt.capacity());
221
222 // Compute average product of head slice and partner tail slice
223 double Q = 1.0;
224 if (PolymerModel::isThread()) {
225 Q = block().averageProduct(qh, qt);
226 } else {
227 UTIL_CHECK(ownsHead() == partner().ownsTail());
228 if (ownsHead()) {
229 Q = block().averageProductBead(qh, qt);
230 } else {
231 Q = block().averageProduct(qh, qt);
232 }
233 }
234
235 return Q;
236 }
