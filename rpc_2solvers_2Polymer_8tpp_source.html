63 template <int D>
64 void Polymer<D>::compute(DArray< RField<D> > const & wFields,
65 double phiTot)
66 {
67 // Setup solvers for all blocks
68 int monomerId;
69 for (int j = 0; j < nBlock(); ++j) {
70 monomerId = block(j).monomerId();
71 block(j).setupSolver(wFields[monomerId]);
72 }
73
74 // Call base class PolymerTmpl solve() function
75 // Solve MDE for all propagators
76 solve(phiTot);
77
78 // Compute block concentration fields
79 double prefactor;
80 if (PolymerModel::isThread()) {
81 prefactor = phi() / ( q() * length() );
82 for (int i = 0; i < nBlock(); ++i) {
83 block(i).computeConcentrationThread(prefactor);
84 }
85 } else
86 if (PolymerModel::isBead()) {
87 prefactor = phi() / ( q() * (double)nBead() );
88 for (int i = 0; i < nBlock(); ++i) {
89 block(i).computeConcentrationBead(prefactor);
90 }
91 }
92
93 }

98 template <int D>
99 void Polymer<D>::computeStress()
100 {
101
102 // Compute stress contributions for all blocks
103 double prefactor;
104 if (PolymerModel::isThread()) {
105 prefactor = phi() / ( q() * length() );
106 for (int i = 0; i < nBlock(); ++i) {
107 block(i).computeStressThread(prefactor);
108 }
109 } else {
110 prefactor = phi() / ( q() * (double)nBead() );
111 for (int i = 0; i < nBlock(); ++i) {
112 block(i).computeStressBead(prefactor);
113 }
114 }
115
116 // Initialize all stress_ elements zero
117 for (int i = 0; i < 6; ++i) {
118 stress_[i] = 0.0;
119 }
120
121 // Sum over all block stress contributions
122 for (int i = 0; i < nBlock(); ++i) {
123 for (int j = 0; j < nParam_ ; ++j){
124 stress_[j] += block(i).stress(j);
125 }
126 }
127
128 }
