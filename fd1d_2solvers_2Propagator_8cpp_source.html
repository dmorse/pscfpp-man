48 /*
49 * Compute initial head QField from final tail QFields of sources.
50 */
51 void Propagator::computeHead()
52 {
53
54 // Reference to head of this propagator
55 QField& qh = qFields_[0];
56
57 // Initialize qh field to 1.0 at all grid points
58 int ix;
59 for (ix = 0; ix < nx_; ++ix) {
60 qh[ix] = 1.0;
61 }
62
63 // Pointwise multiply tail QFields of all sources
64 for (int is = 0; is < nSource(); ++is) {
65 if (!source(is).isSolved()) {
66 UTIL_THROW("Source not solved in computeHead");
67 }
68 QField const& qt = source(is).tail();
69 for (ix = 0; ix < nx_; ++ix) {
70 qh[ix] *= qt[ix];
71 }
72 }
73 }

75 /*
76 * Solve the modified diffusion equation for this block.
77 */
78 void Propagator::solve()
79 {
80 computeHead();
81 for (int iStep = 0; iStep < ns_ - 1; ++iStep) {
82 block().step(qFields_[iStep], qFields_[iStep + 1]);
83 }
84 setIsSolved(true);
85 }

87 /*
88 * Solve the modified diffusion equation with specified initial field.
89 */
90 void Propagator::solve(const Propagator::QField& head)
91 {
92 // Initialize initial (head) field
93 QField& qh = qFields_[0];
94 for (int i = 0; i < nx_; ++i) {
95 qh[i] = head[i];
96 }
97
98 // Setup solver and solve
99 for (int iStep = 0; iStep < ns_ - 1; ++iStep) {
100 block().step(qFields_[iStep], qFields_[iStep + 1]);
101 }
102 setIsSolved(true);
103 }

105 /*
106 * Integrate to calculate monomer concentration for this block
107 */
108 double Propagator::computeQ()
109 {
110 if (!isSolved()) {
111 UTIL_THROW("Propagator is not solved.");
112 }
113 if (!hasPartner()) {
114 UTIL_THROW("Propagator has no partner set.");
115 }
116 if (!partner().isSolved()) {
117 UTIL_THROW("Partner propagator is not solved");
118 }
119 QField const& qh = head();
120 QField const& qt = partner().tail();
121 return block().domain().innerProduct(qh, qt);
122 }
