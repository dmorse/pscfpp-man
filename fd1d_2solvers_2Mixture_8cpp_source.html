57  * Compute concentrations (but not total free energy).
58  */
59  void Mixture::compute(DArray<Mixture::WField> const & wFields,
60  DArray<Mixture::CField>& cFields)
61  {
62  UTIL_CHECK(domainPtr_);
63  UTIL_CHECK(domain().nx() > 0);
64  UTIL_CHECK(nMonomer() > 0);
65  UTIL_CHECK(nPolymer() + nSolvent() > 0);
66  UTIL_CHECK(wFields.capacity() == nMonomer());
67  UTIL_CHECK(cFields.capacity() == nMonomer());
68 
69  int nx = domain().nx();
70  int nm = nMonomer();
71  int i, j, k;
72 
73  // Clear all monomer concentration fields
74  for (i = 0; i < nm; ++i) {
75  UTIL_CHECK(cFields[i].capacity() == nx);
76  UTIL_CHECK(wFields[i].capacity() == nx);
77  for (j = 0; j < nx; ++j) {
78  cFields[i][j] = 0.0;
79  }
80  }
81 
82  // Solve MDE for all polymers
83  for (i = 0; i < nPolymer(); ++i) {
84  polymer(i).compute(wFields);
85  }
86 
87  // Accumulate monomer concentration fields
88  for (i = 0; i < nPolymer(); ++i) {
89  for (j = 0; j < polymer(i).nBlock(); ++j) {
90  int monomerId = polymer(i).block(j).monomerId();
91  UTIL_CHECK(monomerId >= 0);
92  UTIL_CHECK(monomerId < nm);
93  CField& monomerField = cFields[monomerId];
94  CField& blockField = polymer(i).block(j).cField();
95  for (k = 0; k < nx; ++k) {
96  monomerField[k] += blockField[k];
97  }
98  }
99  }
