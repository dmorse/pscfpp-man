484  * Initialize Pscf::Homogeneous::Mixture homogeneous_ member.

486  template <int D>
487  void System<D>::initHomogeneous()
488  {
489 
490  // Set number of molecular species and monomers
491  int nm = mixture().nMonomer();
492  int np = mixture().nPolymer();
493  //int ns = mixture().nSolvent();
494  int ns = 0;
495  UTIL_CHECK(homogeneous_.nMolecule() == np + ns);
496  UTIL_CHECK(homogeneous_.nMonomer() == nm);
497 
498  // Allocate c_ work array, if necessary
499  if (c_.isAllocated()) {
500  UTIL_CHECK(c_.capacity() == nm);
501  } else {
502  c_.allocate(nm);
503  }
504 
505  int i; // molecule index
506  int j; // monomer index
507  int k; // block or clump index
508  int nb; // number of blocks
509  int nc; // number of clumps
510 
511  // Loop over polymer molecule species
512  for (i = 0; i < np; ++i) {
513 
514  // Initial array of clump sizes
515  for (j = 0; j < nm; ++j) {
516  c_[j] = 0.0;
517  }
518 
519  // Compute clump sizes for all monomer types.
520  nb = mixture().polymer(i).nBlock();
521  for (k = 0; k < nb; ++k) {
522  Block<D>& block = mixture().polymer(i).block(k);
523  j = block.monomerId();
524  c_[j] += block.length();
525  }
526 
527  // Count the number of clumps of nonzero size
528  nc = 0;
529  for (j = 0; j < nm; ++j) {
530  if (c_[j] > 1.0E-8) {
531  ++nc;
532  }
533  }
534  homogeneous_.molecule(i).setNClump(nc);
535 
536  // Set clump properties for this Homogeneous::Molecule
537  k = 0; // Clump index
538  for (j = 0; j < nm; ++j) {
539  if (c_[j] > 1.0E-8) {
540  homogeneous_.molecule(i).clump(k).setMonomerId(j);
541  homogeneous_.molecule(i).clump(k).setSize(c_[j]);
542  ++k;
543  }
544  }
545  homogeneous_.molecule(i).computeSize();
546  }

549  /*
550  * Compute Helmoltz free energy and pressure
551  */
552  template <int D>
553  void System<D>::computeFreeEnergy()
554  {
555  fHelmholtz_ = 0.0;
556 
557  // Compute ideal gas contributions to fHelhmoltz_
558  Polymer<D>* polymerPtr;
559  double phi, mu, length;
560  int np = mixture().nPolymer();
561  for (int i = 0; i < np; ++i) {
562  polymerPtr = &mixture().polymer(i);
563  phi = polymerPtr->phi();
564  mu = polymerPtr->mu();
565  // Recall: mu = ln(phi/q)
566  length = polymerPtr->length();
567  fHelmholtz_ += phi*( mu - 1.0 )/length;
568  }
569 
570  int nm = mixture().nMonomer();
571  int nx = mesh().size();
572  //RDField<D> workArray;
573  //workArray.allocate(nx);
574  float temp = 0;
575  for (int i = 0; i < nm; ++i) {
576  for (int j = i + 1; j < nm; ++j) {
577  assignUniformReal << <NUMBER_OF_BLOCKS, THREADS_PER_BLOCK >> >(workArray.cDField(), interaction().chi(i, j), nx);
578  inPlacePointwiseMul << <NUMBER_OF_BLOCKS, THREADS_PER_BLOCK >> > (workArray.cDField(), cFieldsRGrid_[i].cDField(), nx);
579  inPlacePointwiseMul << <NUMBER_OF_BLOCKS, THREADS_PER_BLOCK >> > (workArray.cDField(), cFieldsRGrid_[j].cDField(), nx);
580  fHelmholtz_ += (reductionH(workArray, nx) / nx);
581  }
582 
583  assignReal << <NUMBER_OF_BLOCKS, THREADS_PER_BLOCK >> > (workArray.cDField(), wFieldsRGrid_[i].cDField(), nx);
584  inPlacePointwiseMul << <NUMBER_OF_BLOCKS, THREADS_PER_BLOCK >> > (workArray.cDField(), cFieldsRGrid_[i].cDField(), nx);
585  temp += reductionH(workArray, nx);
586  }
587  fHelmholtz_ -= (temp / nx);
588 
589  // Compute pressure
590  pressure_ = -fHelmholtz_;
591  for (int i = 0; i < np; ++i) {
592  polymerPtr = &mixture().polymer(i);
593  phi = polymerPtr->phi();
594  mu = polymerPtr->mu();
595  length = polymerPtr->length();
596  pressure_ += mu * phi /length;
597  }
598 
599  }

624  template <int D>
625  cudaReal System<D>::innerProduct(const RDField<D>& a, const RDField<D>& b, int size) {
626 
627  switch(THREADS_PER_BLOCK){
628  case 512:
629  deviceInnerProduct<512><<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK, THREADS_PER_BLOCK * sizeof(cudaReal)>>>(d_kernelWorkSpace_, a.cDField(), b.cDField(), size);
630  break;
631  case 256:
632  deviceInnerProduct<256><<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK, THREADS_PER_BLOCK * sizeof(cudaReal)>>>(d_kernelWorkSpace_, a.cDField(), b.cDField(), size);
633  break;
634  case 128:
635  deviceInnerProduct<128><<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK, THREADS_PER_BLOCK * sizeof(cudaReal)>>>(d_kernelWorkSpace_, a.cDField(), b.cDField(), size);
636  break;
637  case 64:
638  deviceInnerProduct<64><<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK, THREADS_PER_BLOCK * sizeof(cudaReal)>>>(d_kernelWorkSpace_, a.cDField(), b.cDField(), size);
639  break;
640  case 32:
641  deviceInnerProduct<32><<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK, THREADS_PER_BLOCK * sizeof(cudaReal)>>>(d_kernelWorkSpace_, a.cDField(), b.cDField(), size);
642  break;
643  case 16:
644  deviceInnerProduct<16><<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK, THREADS_PER_BLOCK * sizeof(cudaReal)>>>(d_kernelWorkSpace_, a.cDField(), b.cDField(), size);
645  break;
646  case 8:
647  deviceInnerProduct<8><<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK, THREADS_PER_BLOCK * sizeof(cudaReal)>>>(d_kernelWorkSpace_, a.cDField(), b.cDField(), size);
648  break;
649  case 4:
650  deviceInnerProduct<4><<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK, THREADS_PER_BLOCK * sizeof(cudaReal)>>>(d_kernelWorkSpace_, a.cDField(), b.cDField(), size);
651  break;
652  case 2:
653  deviceInnerProduct<2><<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK, THREADS_PER_BLOCK * sizeof(cudaReal)>>>(d_kernelWorkSpace_, a.cDField(), b.cDField(), size);
654  break;
655  case 1:
656  deviceInnerProduct<1><<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK, THREADS_PER_BLOCK * sizeof(cudaReal)>>>(d_kernelWorkSpace_, a.cDField(), b.cDField(), size);
657  break;
658  }
659  cudaMemcpy(kernelWorkSpace_, d_kernelWorkSpace_, NUMBER_OF_BLOCKS * sizeof(cudaReal), cudaMemcpyDeviceToHost);
660  cudaReal final = 0;
661  cudaReal c = 0;
662  //use kahan summation to reduce error
663  for (int i = 0; i < NUMBER_OF_BLOCKS; ++i) {
664  cudaReal y = kernelWorkSpace_[i] - c;
665  cudaReal t = final + y;
666  c = (t - final) - y;
667  final = t;
668 
669  }
670 
671  return final;
672  }

674  template<int D>
675  cudaReal System<D>::reductionH(const RDField<D>& a, int size) {
676  reduction << < NUMBER_OF_BLOCKS / 2, THREADS_PER_BLOCK, THREADS_PER_BLOCK * sizeof(cudaReal) >> > (d_kernelWorkSpace_, a.cDField(), size);
677  cudaMemcpy(kernelWorkSpace_, d_kernelWorkSpace_, NUMBER_OF_BLOCKS / 2 * sizeof(cudaReal), cudaMemcpyDeviceToHost);
